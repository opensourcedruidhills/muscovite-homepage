81auto InMemorySubscriber::deliver(const MessageEnvelope& envelope)
82 -> std::vector<HandlerResult> {
83 const std::scoped_lock lock(m_mutex);
84
85 std::vector<HandlerResult> results;
86 for (auto& entry : m_subscriptions) {
87 if (!entry.active->load()) {
88 continue;
89 }
90 if (subject_matches(entry.subject, envelope.subject)) {
91 auto result = entry.handler(envelope);
92 results.push_back(result);
93 ++m_messages_received;
94 }
95 }
96
97 // Clean up inactive subscriptions
98 std::erase_if(m_subscriptions,
99 [](const auto& entry) { return !entry.active->load(); });
100
101 return results;
102}

120auto InMemorySubscriber::subject_matches(
121 const std::string& pattern,
122 const std::string& subject) -> bool {
123 // Exact match
124 if (pattern == subject) {
125 return true;
126 }
127
128 // ">" wildcard: matches any tokens after the prefix
129 // e.g., "muscovite.patient.>" matches "muscovite.patient.PatientCreated"
130 if (pattern.size() >= 2 && pattern.back() == '>') {
131 auto prefix = pattern.substr(0, pattern.size() - 1);
132 return subject.starts_with(prefix);
133 }
134
135 // "*" wildcard: matches a single token
136 // e.g., "muscovite.*.PatientCreated" matches "muscovite.patient.PatientCreated"
137 if (pattern.find('*') != std::string::npos) {
138 // Split both on '.'
139 auto split = [](const std::string& s) -> std::vector<std::string> {
140 std::vector<std::string> tokens;
141 std::size_t start = 0;
142 while (start < s.size()) {
143 auto pos = s.find('.', start);
144 if (pos == std::string::npos) {
145 tokens.push_back(s.substr(start));
146 break;
147 }
148 tokens.push_back(s.substr(start, pos - start));
149 start = pos + 1;
150 }
151 return tokens;
152 };
153
154 auto pattern_tokens = split(pattern);
155 auto subject_tokens = split(subject);
156
157 if (pattern_tokens.size() != subject_tokens.size()) {
158 return false;
159 }
160
161 for (std::size_t i = 0; i < pattern_tokens.size(); ++i) {
162 if (pattern_tokens[i] != "*" && pattern_tokens[i] != subject_tokens[i]) {
163 return false;
164 }
165 }
166 return true;
167 }
168
169 return false;
170}
