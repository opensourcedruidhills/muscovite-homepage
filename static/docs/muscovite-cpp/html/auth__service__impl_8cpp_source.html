16// Helper to extract principal_id from context metadata (set by interceptor)
17std::optional<std::string> get_principal_id_from_context(::grpc::ServerContext* context) {
18 const auto& metadata = context->client_metadata();
19 auto it = metadata.find("x-principal-id");
20 if (it != metadata.end()) {
21 return std::string(it->second.data(), it->second.length());
22 }
23 return std::nullopt;
24}

26// Helper to get IP address from context
27std::string get_ip_address(::grpc::ServerContext* context) {
28 std::string peer = context->peer();
29 // Peer format is usually "ipv4:1.2.3.4:port" or "ipv6:[::1]:port"
30 const size_t last_colon = peer.rfind(':');
31 if (last_colon != std::string::npos) {
32 const std::string addr = peer.substr(0, last_colon);
33 // Remove protocol prefix
34 const size_t first_colon = addr.find(':');
35 if (first_colon != std::string::npos) {
36 return addr.substr(first_colon + 1);
37 }
38 }
39 return peer;
40}

42// Convert Session to proto Session
43void session_to_proto(
44 const Session& session,
45 ::muscovite::security::v1::Session* proto) {
46
47 proto->mutable_id()->set_value(session.id);
48 proto->mutable_principal_id()->set_value(session.principal_id);
49 proto->set_username(session.username);
50 proto->set_display_name(session.display_name);
51
52 for (const auto& group : session.groups) {
53 proto->add_groups(group);
54 }
55 for (const auto& priv : session.privileges) {
56 proto->add_privileges(priv);
57 }
58
59 proto->set_access_token(session.access_token);
60 proto->set_refresh_token(session.refresh_token);
61
62 // Parse ISO 8601 timestamps
63 ::google::protobuf::Timestamp ts;
64 if (!session.expires_at.empty() &&
65 ::google::protobuf::util::TimeUtil::FromString(session.expires_at, &ts)) {
66 *proto->mutable_expires_at() = ts;
67 }
68 if (!session.refresh_expires_at.empty() &&
69 ::google::protobuf::util::TimeUtil::FromString(session.refresh_expires_at, &ts)) {
70 *proto->mutable_refresh_expires_at() = ts;
71 }
72}

96::grpc::Status AuthServiceImpl::Login(
97 ::grpc::ServerContext* context,
98 const ::muscovite::security::v1::LoginRequest* request,
99 ::muscovite::security::v1::LoginResponse* response) {
100
101 std::string ip_address = request->ip_address();
102 if (ip_address.empty()) {
103 ip_address = get_ip_address(context);
104 }
105
106 auto result = impl_->auth_service->login(
107 request->username(),
108 request->password(),
109 ip_address,
110 request->user_agent()
111 );
112
113 response->set_success(result.success);
114 if (result.success && result.session) {
115 session_to_proto(*result.session, response->mutable_session());
116 } else {
117 response->set_error_message(result.error_message);
118 }
119
120 return ::grpc::Status::OK;
121}

141

144 return {::grpc::StatusCode::UNAUTHENTICATED, "Not authenticated"};

182::grpc::Status AuthServiceImpl::ValidateToken(
183 [[maybe_unused]] ::grpc::ServerContext* context,
184 const ::muscovite::security::v1::ValidateTokenRequest* request,
185 ::muscovite::security::v1::ValidateTokenResponse* response) {
186
187 auto principal = impl_->auth_service->validate_token(request->access_token());
188
189 response->set_valid(principal.has_value());
190 if (principal) {
191 // Populate basic session info from the validated principal
192 auto* session = response->mutable_session();
193 session->mutable_principal_id()->set_value(principal->id);
194 session->set_username(principal->username);
195 if (!principal->display_name.empty()) {
196 session->set_display_name(principal->display_name);
197 }
198 }
199
200 return ::grpc::Status::OK;
201}

228::grpc::Status AuthServiceImpl::GetCurrentUser(
229 ::grpc::ServerContext* context,
230 [[maybe_unused]] const ::google::protobuf::Empty* request,
231 ::muscovite::security::v1::Principal* response) {
232
233 auto principal_id = get_principal_id_from_context(context);
234 if (!principal_id) {
235 return {::grpc::StatusCode::UNAUTHENTICATED, "Not authenticated"};
236 }
237
238 // Get user info from session in context metadata
239 const auto& metadata = context->client_metadata();
240
241 response->mutable_id()->set_value(*principal_id);
242
243 auto username_it = metadata.find("x-username");
244 if (username_it != metadata.end()) {
245 response->set_username(std::string(username_it->second.data(), username_it->second.length()));
246 }
247
248 auto display_name_it = metadata.find("x-display-name");
249 if (display_name_it != metadata.end()) {
250 response->set_display_name(std::string(display_name_it->second.data(), display_name_it->second.length()));
251 }
252
253 response->set_is_active(true); // Must be active to be authenticated
254
255 return ::grpc::Status::OK;
256}
