21void init_telemetry(const TelemetryConfig& config) {
22 if (!config.enabled) {
23 return; // Zero overhead when not opted in
24 }
25
26#ifdef MUSCOVITE_TELEMETRY_AVAILABLE
27 // Configure OTLP HTTP exporter (switches to gRPC in FTR-483 when grpc is conan-managed)
28 otlp::OtlpHttpExporterOptions opts;
29 opts.url = config.otlp_endpoint;
30
31 auto exporter = otlp::OtlpHttpExporterFactory::Create(opts);
32
33 // Batch processor for production use
34 auto processor = trace_sdk::BatchSpanProcessorFactory::Create(
35 std::move(exporter), trace_sdk::BatchSpanProcessorOptions{});
36
37 // Service resource attributes
38 auto resource_attrs = resource::ResourceAttributes{
39 {"service.name", config.service_name}
40 };
41 auto res = resource::Resource::Create(resource_attrs);
42
43 // Create and register the tracer provider
44 auto sdk_provider = trace_sdk::TracerProviderFactory::Create(
45 std::move(processor), res);
46
47 trace_api::Provider::SetTracerProvider(
48 nostd::shared_ptr<trace_api::TracerProvider>(sdk_provider.release()));
49#endif
50}

52void shutdown_telemetry() {
53#ifdef MUSCOVITE_TELEMETRY_AVAILABLE
54 auto provider = trace_api::Provider::GetTracerProvider();
55 if (provider) {
56 // Flush all pending spans
57 if (auto* sdk_provider =
58 dynamic_cast<trace_sdk::TracerProvider*>(provider.get())) {
59 sdk_provider->ForceFlush();
60 sdk_provider->Shutdown();
61 }
62 }
63#endif
64}
