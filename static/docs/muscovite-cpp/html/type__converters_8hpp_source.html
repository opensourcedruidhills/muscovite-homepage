40// =============================================================================
41// DecimalParam: Wrapper for std::optional<Decimal> to work with pqxx::params
42// =============================================================================
43// The pqxx library uses `if constexpr (nullness<T>::always_null)` which requires
44// compile-time constexpr. boost::multiprecision provides non-constexpr static bool,
45// causing pqxx instantiation failures. This wrapper provides the required trait.
46struct DecimalParam {
47 std::optional<Decimal> value;
48 explicit DecimalParam(const std::optional<Decimal>& v) : value(v) {}
49 explicit DecimalParam(std::optional<Decimal>&& v) : value(std::move(v)) {}
50};

187// =============================================================================
188// pqxx Specialization for Decimal (boost::multiprecision::cpp_dec_float_50)
189// =============================================================================
190// Boost's multiprecision library doesn't provide a pqxx nullness specialization,
191// so we add one here. Decimal values are never null on their own.
192namespace pqxx {
193 template<>
194 struct nullness<muscovite::dba::Decimal> {
195 // CRITICAL: Must be static constexpr for pqxx's if constexpr checks
196 static constexpr bool always_null = false;
197
198 // pqxx also requires is_null(const T&) static method
199 [[nodiscard]] static constexpr auto is_null(const muscovite::dba::Decimal&) noexcept -> bool {
200 return false; // Decimal values are never null
201 }
202 };
203}

205// =============================================================================
206// pqxx Specialization for DecimalParam (wraps optional<Decimal>)
207// =============================================================================
208namespace pqxx {
209 template<>

211 // CRITICAL: Must be static constexpr for pqxx's if constexpr checks
212 static constexpr bool always_null = false;
213
214 // pqxx also requires is_null(const T&) static method

219
220 template<>
221 struct string_traits<muscovite::dba::DecimalParam> {
222 static constexpr bool converts_to_string = true;
223 static constexpr bool converts_from_string = false;
224
226 [[nodiscard]] static std::size_t size_buffer(const muscovite::dba::DecimalParam& param) noexcept {
227 if (!param.value.has_value()) {
228 return 5; // "NULL" + null terminator
229 }
230 // Conservative estimate: 50 digits + sign + decimal point + null
231 return 55;
232 }
233
235 static char* into_buf(char* begin, const char* end, const muscovite::dba::DecimalParam& param) {
236 if (!param.value.has_value()) {
237 // For NULL, we write empty (pqxx handles NULLs separately via nullness)
238 *begin = '\0';
239 return begin + 1; // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
240 }
241 // Convert Decimal to string and copy to buffer
242 std::string str = param.value->str(0, std::ios_base::fixed);
243 if (static_cast<std::size_t>(end - begin) < str.size() + 1) {
244 throw conversion_overrun{"Buffer too small for DecimalParam"};
245 }
246 std::ranges::copy(str, begin);
247 begin[str.size()] = '\0'; // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
248 return begin + str.size() + 1; // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
249 }
250
252 [[nodiscard]] static zview to_buf(char* begin, const char* end, const muscovite::dba::DecimalParam& param) {
253 const char* written_end = into_buf(begin, end, param);
254 return zview{begin, static_cast<std::size_t>(written_end - begin - 1)};
255 }
256 };
257}
