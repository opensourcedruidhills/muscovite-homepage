class invalid_value_error : public std::invalid_argument {
public:
invalid_value_error(std::string_view type_name, std::string_view value)
: std::invalid_argument(
std::format("Invalid {} value: '{}'", type_name, value)) {}
invalid_value_error(std::string_view type_name, std::string_view value,
std::string_view pattern)
: std::invalid_argument(std::format(
"Invalid {} value: '{}' (must match: {})", type_name, value, pattern)) {}
};
template <typename Derived>
class strong_type_base {
public:
// NOLINTBEGIN(bugprone-crtp-constructor-accessibility)
// Public constructors required: derived classes inherit via
// `using strong_type_base::strong_type_base;` which needs public access.
// CRTP misuse is prevented by the Derived-only validation call.
strong_type_base() = default;
explicit strong_type_base(std::string_view value)
: value_(value) {
if (!Derived::is_valid(value)) {
throw invalid_value_error(Derived::type_name(), value);
}
}
// NOLINTEND(bugprone-crtp-constructor-accessibility)
[[nodiscard]] static auto from_trusted(std::string value) noexcept -> Derived {
Derived d;
d.value_ = std::move(value);
return d;
}
[[nodiscard]] auto value() const noexcept -> const std::string& {
return value_;
}
[[nodiscard]] operator std::string_view() const noexcept {
return value_;
}
[[nodiscard]] auto empty() const noexcept -> bool {
return value_.empty();
}
[[nodiscard]] auto operator<=>(const strong_type_base& other) const noexcept = default;
[[nodiscard]] auto operator==(const strong_type_base& other) const noexcept -> bool = default;
protected:
std::string value_;
};
} // namespace muscovite::types
