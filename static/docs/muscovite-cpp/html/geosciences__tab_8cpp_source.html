90void GeosciencesTab::setupConnections() {
91 if (!serviceClient || !authMgr) return;
92
93 // UI actions
94 connect(fetchButton, &QPushButton::clicked, this, &GeosciencesTab::onFetchSamples);
95 connect(createButton, &QPushButton::clicked, this, &GeosciencesTab::onCreateClicked);
96 connect(editButton, &QPushButton::clicked, this, &GeosciencesTab::onEditClicked);
97 connect(deleteButton, &QPushButton::clicked, this, &GeosciencesTab::onDeleteClicked);
98
99 // Service results
100 connect(serviceClient, &services::GeosciencesServiceClient::samplesLoaded,
101 this, &GeosciencesTab::onSamplesLoaded);

103 this, &GeosciencesTab::onSamplesLoadFailed);
104 connect(serviceClient, &services::GeosciencesServiceClient::sampleCreated,
105 this, &GeosciencesTab::onSampleCreated);
106 connect(serviceClient, &services::GeosciencesServiceClient::sampleCreationFailed,
107 this, &GeosciencesTab::onSampleCreationFailed);
108 connect(serviceClient, &services::GeosciencesServiceClient::sampleUpdated,
109 this, &GeosciencesTab::onSampleUpdated);
110 connect(serviceClient, &services::GeosciencesServiceClient::sampleUpdateFailed,
111 this, &GeosciencesTab::onSampleUpdateFailed);
112 connect(serviceClient, &services::GeosciencesServiceClient::sampleDeleted,
113 this, &GeosciencesTab::onSampleDeleted);
114 connect(serviceClient, &services::GeosciencesServiceClient::sampleDeletionFailed,
115 this, &GeosciencesTab::onSampleDeletionFailed);

117 this, &GeosciencesTab::onAuthenticationRequired);

119 // Auth state

124void GeosciencesTab::updateAuthorizationState() {
125 if (!authMgr || !authMgr->isAuthenticated()) {
126 createButton->setEnabled(false);
127 editButton->setEnabled(false);
128 deleteButton->setEnabled(false);
129 return;
130 }
131
132 createButton->setEnabled(authMgr->hasPrivilege("CREATE_SAMPLE"));
133 editButton->setEnabled(authMgr->hasPrivilege("UPDATE_SAMPLE"));
134 deleteButton->setEnabled(authMgr->hasPrivilege("DELETE_SAMPLE"));
135}

144void GeosciencesTab::onCreateClicked() {
145 if (!serviceClient) return;
146
147 QString name = nameEdit->text().trimmed();
148 QString description = descriptionEdit->text().trimmed();
149
150 if (name.isEmpty()) {
151 errorLabel->setText("Please enter a sample name");
152 return;
153 }
154
155 statusLabel->setText("Creating sample...");
156 errorLabel->setText("");
157 serviceClient->createSample(name, description);
158}
159
160void GeosciencesTab::onEditClicked() {
161 if (!serviceClient || samplesTable->currentRow() < 0) {
162 errorLabel->setText("Please select a sample to edit");
163 return;
164 }
165
166 auto item = samplesTable->item(samplesTable->currentRow(), 0);
167 if (!item) return;
168
169 QString id = item->text();
170 QString name = nameEdit->text().trimmed();
171 QString description = descriptionEdit->text().trimmed();
172
173 if (name.isEmpty()) {
174 errorLabel->setText("Please enter a sample name");
175 return;
176 }
177
178 statusLabel->setText("Updating sample...");
179 errorLabel->setText("");
180 serviceClient->updateSample(id, name, description);
181}
182
183void GeosciencesTab::onDeleteClicked() {
184 if (!serviceClient || samplesTable->currentRow() < 0) {
185 errorLabel->setText("Please select a sample to delete");
186 return;
187 }
188
189 auto item = samplesTable->item(samplesTable->currentRow(), 0);
190 if (!item) return;
191
192 QString id = item->text();
193
194 if (QMessageBox::question(this, "Delete Sample",
195 "Are you sure you want to delete this sample?") != QMessageBox::Yes) {
196 return;
197 }
198
199 statusLabel->setText("Deleting sample...");
200 errorLabel->setText("");
201 serviceClient->deleteSample(id);
202}

271void GeosciencesTab::loadTable() {
272 samplesTable->setRowCount(samples.size());
273
274 for (int row = 0; row < samples.size(); ++row) {
275 const auto& sample = samples[row];
276 samplesTable->setItem(row, 0, new QTableWidgetItem(sample.id));
277 samplesTable->setItem(row, 1, new QTableWidgetItem(sample.name));
278 samplesTable->setItem(row, 2, new QTableWidgetItem(sample.description));
279 }
