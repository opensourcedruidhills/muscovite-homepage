10void OAuth2Config::validate() const {
11 if (authority.empty()) {
12 throw std::invalid_argument("OAuth2Config: authority cannot be empty");
13 }
14
15 if (client_id.empty()) {
16 throw std::invalid_argument("OAuth2Config: client_id cannot be empty");
17 }
18
19 if (redirect_uri.empty()) {
20 throw std::invalid_argument("OAuth2Config: redirect_uri cannot be empty");
21 }
22
23 if (scopes.empty()) {
24 throw std::invalid_argument("OAuth2Config: scopes cannot be empty");
25 }
26
27 if (provider_type != ProviderType::BuiltIn && token_endpoint.empty()) {
28 throw std::invalid_argument("OAuth2Config: token_endpoint required for external providers");
29 }
30
31 if (provider_type != ProviderType::BuiltIn && authorize_endpoint.empty()) {
32 throw std::invalid_argument("OAuth2Config: authorize_endpoint required for external providers");
33 }
34
35 if (provider_type == ProviderType::BuiltIn && !builtin_auth_endpoint) {
36 throw std::invalid_argument("OAuth2Config: builtin_auth_endpoint required for BuiltIn provider");
37 }
38
39 if (access_token_ttl_seconds <= 0) {
40 throw std::invalid_argument("OAuth2Config: access_token_ttl_seconds must be positive");
41 }
42
43 if (refresh_token_ttl_seconds <= 0) {
44 throw std::invalid_argument("OAuth2Config: refresh_token_ttl_seconds must be positive");
45 }
46
47 if (token_refresh_margin_seconds < 0) {
48 throw std::invalid_argument("OAuth2Config: token_refresh_margin_seconds cannot be negative");
49 }
50
51 if (use_pkce && (code_challenge_method != "S256" && code_challenge_method != "plain")) {
52 throw std::invalid_argument("OAuth2Config: code_challenge_method must be 'S256' or 'plain'");
53 }
54}

56OAuth2Config OAuth2Config::forKeycloak(
57 const std::string& realm_url,
58 const std::string& client_id,
59 const std::string& redirect_uri
60) {
61 OAuth2Config config;
62 config.authority = realm_url;
63 config.provider_type = ProviderType::Keycloak;
64 config.client_id = client_id;
65 config.redirect_uri = redirect_uri;
66
67 // Keycloak endpoints
68 config.token_endpoint = std::format("{}/protocol/openid-connect/token", realm_url);
69 config.authorize_endpoint = std::format("{}/protocol/openid-connect/auth", realm_url);
70 config.userinfo_endpoint = std::format("{}/protocol/openid-connect/userinfo", realm_url);
71 config.revocation_endpoint = std::format("{}/protocol/openid-connect/revoke", realm_url);
72
73 // Keycloak requires PKCE for public clients
74 config.use_pkce = true;
75 config.code_challenge_method = "S256";
76
77 config.validate();
78 return config;
79}

129OAuth2Config OAuth2Config::forBuiltIn(
130 const std::string& auth_endpoint
131) {
132 OAuth2Config config;
133 config.authority = auth_endpoint;
134 config.provider_type = ProviderType::BuiltIn;
135 config.builtin_auth_endpoint = auth_endpoint;
136 config.client_id = "builtin-client";
137 config.redirect_uri = "http://localhost:8080/callback";
138
139 // Built-in doesn't use standard OAuth2 endpoints
140 config.token_endpoint = std::format("{}/token", auth_endpoint);
141 config.authorize_endpoint = ""; // Not used for built-in
142 config.userinfo_endpoint = ""; // Not used for built-in
143
144 // Built-in doesn't use PKCE
145 config.use_pkce = false;
146
147 config.validate();
148 return config;
