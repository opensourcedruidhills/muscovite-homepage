35class GrpcMetadataCarrier : public oprop::TextMapCarrier {
36public:
37 explicit GrpcMetadataCarrier(::grpc::experimental::InterceptorBatchMethods* methods)
38 : methods_(methods) {}
39
40 auto Get(nostd::string_view key) const noexcept
41 -> nostd::string_view override {
42 auto* metadata = methods_->GetRecvInitialMetadata();
43 if (!metadata) {
44 return "";
45 }
46 auto it = metadata->find(std::string(key.data(), key.size()));
47 if (it != metadata->end()) {
48 return nostd::string_view(
49 it->second.data(), it->second.size());
50 }
51 return "";
52 }
53
54 void Set(nostd::string_view, nostd::string_view) noexcept override {
55 // Read-only carrier for incoming metadata
56 }
57
58private:
59 ::grpc::experimental::InterceptorBatchMethods* methods_;
60};

73class OtelServerInterceptor : public ::grpc::experimental::Interceptor {
74public:
75 explicit OtelServerInterceptor(::grpc::experimental::ServerRpcInfo* info)
76 : method_name_(info ? info->method() : "unknown") {}
77
78 void Intercept(::grpc::experimental::InterceptorBatchMethods* methods) override {
79 if (methods->QueryInterceptionHookPoint(
80 ::grpc::experimental::InterceptionHookPoints::POST_RECV_INITIAL_METADATA)) {
81 // Start span on receiving initial metadata
82 auto tracer = otrace::Provider::GetTracerProvider()->GetTracer("grpc-server");
83
84 // Extract trace context from incoming metadata
85 GrpcMetadataCarrier carrier(methods);
86 auto propagator = oprop::GlobalTextMapPropagator::GetGlobalPropagator();
87 auto current_ctx = ocontext::RuntimeContext::GetCurrent();
88 auto context = propagator->Extract(carrier, current_ctx);
89
90 otrace::StartSpanOptions options;
91 options.parent = otrace::GetSpan(context)->GetContext();
92
93 span_ = tracer->StartSpan(method_name_, {}, options);
94
95 // Set standard gRPC span attributes
96 span_->SetAttribute("rpc.system", "grpc");
97 span_->SetAttribute("rpc.method", method_name_);
98 }
99
100 if (methods->QueryInterceptionHookPoint(
101 ::grpc::experimental::InterceptionHookPoints::PRE_SEND_STATUS)) {
102 // End span on sending status
103 if (span_) {
104 auto status = methods->GetSendStatus();
105 span_->SetAttribute("rpc.grpc.status_code", static_cast<int>(status.error_code()));
106 if (status.error_code() != ::grpc::StatusCode::OK) {
107 span_->SetStatus(otrace::StatusCode::kError,
108 std::string(status.error_message()));
109 } else {
110 span_->SetStatus(otrace::StatusCode::kOk);
111 }
112 span_->End();
113 }
114 }
115
116 methods->Proceed();
117 }
118
119private:
120 std::string method_name_;
121 nostd::shared_ptr<otrace::Span> span_;
122};
123#endif
