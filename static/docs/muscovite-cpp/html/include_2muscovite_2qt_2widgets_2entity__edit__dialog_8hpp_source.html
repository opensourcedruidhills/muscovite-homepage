29class EntityEditDialog : public QDialog {
30 Q_OBJECT
31
32public:
33 explicit EntityEditDialog(QWidget* parent = nullptr)
34 : QDialog(parent) {
35 formLayout_ = new QFormLayout();
36
37 auto* mainLayout = new QVBoxLayout(this);
38 mainLayout->addLayout(formLayout_);
39
40 buttonBox_ = new QDialogButtonBox(
41 QDialogButtonBox::Ok | QDialogButtonBox::Cancel,
42 this);
43
44 connect(buttonBox_, &QDialogButtonBox::accepted, this, &QDialog::accept);
45 connect(buttonBox_, &QDialogButtonBox::rejected, this, &QDialog::reject);
46
47 mainLayout->addWidget(buttonBox_);
48 }
49
50protected:
51 QLineEdit* addLineEdit(const QString& label, const QString& objectName) {
52 auto* lineEdit = new QLineEdit(this);
53 lineEdit->setObjectName(objectName);
54 formLayout_->addRow(label + ":", lineEdit);
55 return lineEdit;
56 }
57
58 QTextEdit* addTextEdit(const QString& label, const QString& objectName) {
59 auto* textEdit = new QTextEdit(this);
60 textEdit->setObjectName(objectName);
61 textEdit->setMaximumHeight(100);
62 formLayout_->addRow(label + ":", textEdit);
63 return textEdit;
64 }
65
66 QComboBox* addComboBox(const QString& label, const QString& objectName) {
67 auto* comboBox = new QComboBox(this);
68 comboBox->setObjectName(objectName);
69 formLayout_->addRow(label + ":", comboBox);
70 return comboBox;
71 }
72
73 QCheckBox* addCheckBox(const QString& label, const QString& objectName) {
74 auto* checkBox = new QCheckBox(this);
75 checkBox->setObjectName(objectName);
76 formLayout_->addRow(label + ":", checkBox);
77 return checkBox;
78 }
79
80 QDateEdit* addDateEdit(const QString& label, const QString& objectName) {
81 auto* dateEdit = new QDateEdit(this);
82 dateEdit->setObjectName(objectName);
83 dateEdit->setCalendarPopup(true);
84 formLayout_->addRow(label + ":", dateEdit);
85 return dateEdit;
86 }
87
88 QFormLayout* formLayout_;
89 QDialogButtonBox* buttonBox_;
90};
