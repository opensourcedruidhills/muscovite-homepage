168 [[nodiscard]] static auto escape_json(std::string_view input) -> std::string {
169 std::string result;
170 result.reserve(input.size());
171 for (char const ch : input) {
172 switch (ch) {
173 case '"': result += R"(\")"; break;
174 case '\\': result += R"(\\)"; break;
175 case '\n': result += R"(\n)"; break;
176 case '\t': result += R"(\t)"; break;
177 default: result += ch;
178 }
