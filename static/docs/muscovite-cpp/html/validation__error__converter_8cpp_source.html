65validation::ValidationMessage toValidationMessage(const ValidationError& error) {
66 QString const codeStr = QString::fromStdString(error.code);
67 QString const fieldName = QString::fromStdString(error.field);
68 QString const detail = QString::fromStdString(error.message);
69
70 validation::ValidationCode const code = parseValidationCode(codeStr);
71
72 // If we parsed as VAL500_Custom, use the backend message as detail
73 // Otherwise, the standard message from validationMessage() is used
74 if (code == validation::ValidationCode::VAL500_Custom && !detail.isEmpty()) {
75 return validation::ValidationMessage::custom(fieldName, detail, false);
76 }
77
78 return {code, fieldName, detail};
79}

91std::optional<ValidationError> extractValidationErrorFromContext(const ::grpc::ClientContext& ctx) {
92 const auto& metadata = ctx.GetServerTrailingMetadata();
93
94 auto codeIt = metadata.find("x-validation-code");
95 auto fieldIt = metadata.find("x-validation-field");
96 auto msgIt = metadata.find("x-validation-message");
97
98 if (codeIt != metadata.end() || msgIt != metadata.end()) {
99 ValidationError error;
100 if (codeIt != metadata.end()) {
101 error.code = std::string(codeIt->second.data(), codeIt->second.size());
102 }
103 if (fieldIt != metadata.end()) {
104 error.field = std::string(fieldIt->second.data(), fieldIt->second.size());
105 }
106 if (msgIt != metadata.end()) {
107 error.message = std::string(msgIt->second.data(), msgIt->second.size());
108 }
109 return error;
110 }
111
112 return std::nullopt;
113}

115bool handleValidationMetadata(const ::grpc::ClientContext& ctx, const ::grpc::Status& status,
116 const std::function<void(const validation::ValidationResult&)>& showResult,
117 const std::function<void(bool)>& setLoading) {
118
119 if (status.ok()) {
120 return true; // Success, no errors
121 }
122
123 auto validationError = extractValidationErrorFromContext(ctx);
124 if (validationError) {
125 auto result = fromBackendError(*validationError);
126 showResult(result);
127 setLoading(false);
128 return false; // Validation error shown
129 }
130
131 return true; // Error, but not a validation error - let caller handle
132}
