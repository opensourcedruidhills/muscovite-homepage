25auto parse_timestamp(const std::string& iso_str)
26 -> std::chrono::system_clock::time_point {
27 // Simple parser for format: 2026-02-03T12:34:56.123456Z
28 std::tm tm{};
29 std::istringstream ss(iso_str);
30 ss >> std::get_time(&tm, "%Y-%m-%dT%H:%M:%S");
31 if (ss.fail()) {
32 return std::chrono::system_clock::now();
33 }
34 auto tp = std::chrono::system_clock::from_time_t(std::mktime(&tm));
35
36 // Parse fractional seconds if present
37 if (ss.peek() == '.') {
38 ss.get(); // consume '.'
39 int microseconds = 0;
40 ss >> microseconds;
41 tp += std::chrono::microseconds(microseconds);
42 }
43 return tp;
44}

89auto NatsEventConsumer::start() -> bool {
90 if (m_running) {
91 spdlog::warn("Consumer already running");
92 return true;
93 }
94
95 spdlog::info("Starting NATS event consumer for stream: {}", m_stream_name);
96
97 // FTR-208: RAII-managed NATS options (auto-destroyed at end of scope)
98 auto [opts_status, opts] = make_nats_options();
99 if (opts_status != NATS_OK) {
100 const std::scoped_lock lock(m_error_mutex);
101 m_last_error = "Failed to create NATS options: " +
102 std::string(natsStatus_GetText(opts_status));
103 spdlog::error(m_last_error);
104 return false;
105 }
106
107 auto s = natsOptions_SetURL(opts.get(), m_nats_url.c_str());
108 if (s != NATS_OK) {
109 const std::scoped_lock lock(m_error_mutex);
110 m_last_error = "Failed to set NATS URL: " +
111 std::string(natsStatus_GetText(s));
112 spdlog::error(m_last_error);
113 return false;
114 }
115
116 // Set reconnect options
117 natsOptions_SetMaxReconnect(opts.get(), -1); // Unlimited reconnects
118 natsOptions_SetReconnectWait(opts.get(), 1000); // 1 second between attempts
119
120 // FTR-208: factory returns RAII-managed connection
121 auto [conn_status, conn] = make_nats_connection(opts.get());
122 // opts RAII will auto-destroy at scope exit
123
124 if (conn_status != NATS_OK) {
125 const std::scoped_lock lock(m_error_mutex);
126 m_last_error = "Failed to connect to NATS: " +
127 std::string(natsStatus_GetText(conn_status));
128 spdlog::error(m_last_error);
129 return false;
130 }
131 m_conn = std::move(conn);
132
133 // Create JetStream context
134 jsOptions jsOpts;
135 jsOptions_Init(&jsOpts);
136
137 auto [js_status, js] = make_js_context(m_conn.get(), &jsOpts);
138 if (js_status != NATS_OK) {
139 m_conn.reset();
140 const std::scoped_lock lock(m_error_mutex);
141 m_last_error = "Failed to create JetStream context: " +
142 std::string(natsStatus_GetText(js_status));
143 spdlog::error(m_last_error);
144 return false;
145 }
146 m_js = std::move(js);
147
148 // Build filter subjects from config; use first subject or default pattern
149 const std::string filter_subject = !m_config.subjects.empty()
150 ? m_config.subjects[0]
151 : "muscovite." + m_stream_name + ".>";
152
153 spdlog::debug("Consumer filter subject: {}", filter_subject);
154
155 // Create durable push consumer
156 // The consumer will be created if it doesn't exist, or updated if it does
157 jsConsumerConfig cc;
158 jsConsumerConfig_Init(&cc);
159 cc.Durable = m_config.consumer_name.c_str();
160 cc.AckPolicy = js_AckExplicit;
161 cc.AckWait = static_cast<int64_t>(m_config.ack_wait.count()) * 1000000000LL; // nanoseconds
162 cc.MaxDeliver = static_cast<int64_t>(m_config.max_deliver);
163 cc.FilterSubject = filter_subject.c_str();
164 cc.DeliverPolicy = m_config.deliver_all ? js_DeliverAll : js_DeliverNew;
165 cc.MaxAckPending = static_cast<int64_t>(m_config.max_pending);
166
167 jsSubOptions subOpts;
168 jsSubOptions_Init(&subOpts);
169 subOpts.Stream = m_stream_name.c_str();
170 // Don't set Consumer - we want NATS to create/update it from Config.Durable
171 subOpts.Config = cc;
172
173 jsErrCode errCode{};
174 // FTR-208: factory returns RAII-managed subscription
175 auto [sub_status, sub] = make_js_subscription(
176 m_js.get(), filter_subject.c_str(),
177 message_handler, this, &jsOpts, &subOpts, &errCode);
178 if (sub_status != NATS_OK) {
179 m_js.reset();
180 m_conn.reset();
181 const std::scoped_lock lock(m_error_mutex);
182 m_last_error = std::format("Failed to create JetStream subscription: {} (errCode={})",
183 natsStatus_GetText(sub_status), static_cast<int>(errCode));
184 spdlog::error(m_last_error);
185 return false;
186 }
187 m_sub = std::move(sub);
188
189 // Flush connection to ensure subscription is fully established on server
190 // This prevents race conditions where publish happens before subscription is ready
191 s = natsConnection_Flush(m_conn.get());
192 if (s != NATS_OK) {
193 spdlog::warn("Failed to flush connection after subscribe: {}",
194 natsStatus_GetText(s));
195 }
196
197 m_running = true;
198 m_should_stop = false;
199
200 spdlog::info("NATS event consumer started for stream: {} with consumer: {}",
201 m_stream_name, m_config.consumer_name);
202 return true;
203}

205void NatsEventConsumer::stop() {
206 if (!m_running) {
207 return;
208 }
209
210 spdlog::info("Stopping NATS event consumer");
211 m_should_stop.store(true, std::memory_order_release);
212
213 // Drain subscription and wait for completion to avoid callback races
214 // during teardown. Destroying handles before drain completion can leave
215 // in-flight callbacks touching freed state.
216 if (m_sub) {
217 auto const drain_status = natsSubscription_DrainTimeout(m_sub.get(), 5 * 1000000000LL);
218 if (drain_status != NATS_OK) {
219 spdlog::warn("Subscription drain timeout/status: {}", natsStatus_GetText(drain_status));
220 }
221
222 auto const wait_status = natsSubscription_WaitForDrainCompletion(m_sub.get(), 5 * 1000000000LL);
223 if (wait_status != NATS_OK && wait_status != NATS_TIMEOUT) {
224 spdlog::warn("Subscription wait-for-drain status: {}", natsStatus_GetText(wait_status));
225 }
226
227 // BFX-002: Spin-wait for in-flight message_handler callbacks to
228 // finish before destroying handles. DrainCompletion may return
229 // while a callback is still executing process_message().
230 for (int attempt = 0; attempt < 500; ++attempt) {
231 if (m_in_flight_callbacks.load(std::memory_order_acquire) == 0) {
232 break;
233 }
234 std::this_thread::sleep_for(std::chrono::milliseconds(10));
235 }
236 if (m_in_flight_callbacks.load(std::memory_order_acquire) != 0) {
237 spdlog::warn("BFX-002: {} callbacks still in-flight after 5s wait",
238 m_in_flight_callbacks.load());
239 }
240
241 // FTR-208: RAII reset() calls NatsSubscriptionDeleter → natsSubscription_Destroy
242 m_sub.reset();
243 }
244
245 // FTR-208: RAII reset() calls JsCtxDeleter → jsCtx_Destroy
246 m_js.reset();
247
248 // BFX-001: Close the connection and spin-wait for IsClosed before
249 // Destroy. This ensures cnats internal close handlers have quiesced.
250 // The remaining cnats library threads are cleaned up by the global
251 // nats_CloseAndWait() call (see NatsLibraryGuard in the framework).
252 if (m_conn) {
253 natsConnection_Close(m_conn.get());
254
255 for (int attempt = 0; attempt < 100; ++attempt) {
256 if (natsConnection_IsClosed(m_conn.get())) {
257 break;
258 }
259 std::this_thread::sleep_for(std::chrono::milliseconds(10));
260 }
261
262 // FTR-208: RAII reset() calls NatsConnectionDeleter → natsConnection_Destroy
263 m_conn.reset();
264 }
265
266 m_running = false;
267 spdlog::info("NATS event consumer stopped");
268}

287void NatsEventConsumer::message_handler(natsConnection* /*nc*/,
288 natsSubscription* /*sub*/,
289 natsMsg* msg,
290 void* closure) {
291 auto* self = static_cast<NatsEventConsumer*>(closure);
292
293 // BFX-002: Track in-flight callbacks so stop() can wait for completion.
294 // If stop is already in progress and draining has finished, bail out
295 // immediately to avoid touching member state during teardown.
296 if (self->m_should_stop.load(std::memory_order_acquire)) {
297 natsMsg_Nak(msg, nullptr);
298 natsMsg_Destroy(msg);
299 return;
300 }
301
302 self->m_in_flight_callbacks.fetch_add(1, std::memory_order_acq_rel);
303
304 // RAII guard: decrement in_flight on all exit paths
305 struct InFlightGuard {
306 std::atomic<int>& counter; // NOLINT(cppcoreguidelines-avoid-const-or-ref-data-members)
307 ~InFlightGuard() { counter.fetch_sub(1, std::memory_order_acq_rel); }
308 };
309 const InFlightGuard guard{self->m_in_flight_callbacks};
310
311 self->process_message(msg);
312}

314void NatsEventConsumer::process_message(natsMsg* msg) {
315 if (m_should_stop) {
316 // Nak during shutdown so message is redelivered
317 natsMsg_Nak(msg, nullptr);
318 natsMsg_Destroy(msg);
319 return;
320 }
321
322 spdlog::debug("Processing message on subject: {}", natsMsg_GetSubject(msg));
323
324 // Parse the event
325 auto ctx_opt = parse_event(msg);
326 if (!ctx_opt) {
327 // Invalid message, term (don't retry)
328 spdlog::warn("Failed to parse event message, terminating");
329 natsMsg_Term(msg, nullptr);
330 m_events_failed++;
331 natsMsg_Destroy(msg);
332 return;
333 }
334
335 auto& ctx = *ctx_opt;
336
337 spdlog::debug("Parsed event: id={}, type={}", ctx.event_id, ctx.event_type);
338
339 // Check for duplicate (if deduplication is enabled)
340 if (m_config.dedupe_window.count() > 0 && is_duplicate(ctx.event_id)) {
341 spdlog::debug("Skipping duplicate event: {}", ctx.event_id);
342 natsMsg_Ack(msg, nullptr);
343 natsMsg_Destroy(msg);
344 return;
345 }
346
347 // Dispatch to handler
348 HandlerResult result{};
349 try {
350 result = dispatch_event(ctx);
351 } catch (const std::exception& e) {
352 spdlog::debug("Handler threw exception: {}", e.what());
353 // Call error handler
354 const ErrorContext err_ctx{.event = ctx, .error_message = e.what(), .exception_type = typeid(e).name()};
355 try {
356 const std::scoped_lock lock(m_handlers_mutex);
357 if (m_error_handler) {
358 result = m_error_handler(err_ctx);
359 } else {
360 result = HandlerResult::Nak;
361 }
362 } catch (...) {
363 spdlog::error("Error handler threw exception");
364 result = HandlerResult::Nak;
365 }
366 }
367
368 // Acknowledge based on result
369 acknowledge(msg, result);
370
371 // Record for deduplication
372 if (result == HandlerResult::Ack && m_config.dedupe_window.count() > 0) {
373 record_processed(ctx.event_id);
374 }
375
376 // Update counters
377 if (result == HandlerResult::Ack) {
378 m_events_processed++;
379 } else if (result == HandlerResult::Nak || result == HandlerResult::Term) {
380 m_events_failed++;
381 }
382
383 natsMsg_Destroy(msg);
384
385 // Periodic dedupe cache cleanup
386 cleanup_dedupe_cache();
387}

389auto NatsEventConsumer::parse_event(natsMsg* msg) -> std::optional<EventContext> {
390 const char* data = natsMsg_GetData(msg);
391 const int len = natsMsg_GetDataLength(msg);
392
393 if (data == nullptr || len == 0) {
394 spdlog::warn("Empty message received");
395 return std::nullopt;
396 }
397
398 try {
399 auto json = nlohmann::json::parse(std::string(data, static_cast<size_t>(len)));
400
401 EventContext ctx;
402 ctx.event_id = json.value("id", "");
403 ctx.event_type = json.value("event_type", "");
404 ctx.aggregate_type = json.value("aggregate_type", "");
405 ctx.aggregate_id = json.value("aggregate_id", "");
406 ctx.schema = json.value("schema", m_stream_name);
407 ctx.payload = json.value("payload", nlohmann::json::object());
408 ctx.occurred_at = parse_timestamp(json.value("occurred_at", ""));
409 ctx.subject = natsMsg_GetSubject(msg);
410
411 // Get JetStream metadata if available
412 jsMsgMetaData* meta = nullptr;
413 if (natsMsg_GetMetaData(&meta, msg) == NATS_OK && meta != nullptr) {
414 ctx.sequence = meta->Sequence.Stream;
415 ctx.delivery_count = meta->NumDelivered;
416 jsMsgMetaData_Destroy(meta);
417 }
418
419 // Reply-to (for request-reply patterns)
420 const char* reply = natsMsg_GetReply(msg);
421 if (reply != nullptr && *reply != '\0') {
422 ctx.reply_to = std::string(reply);
423 }
424
425 if (ctx.event_id.empty() || ctx.event_type.empty()) {
426 spdlog::warn("Event missing required fields: id={}, type={}",
427 ctx.event_id, ctx.event_type);
428 return std::nullopt;
429 }
430
431 return ctx;
432 } catch (const nlohmann::json::exception& e) {
433 spdlog::warn("Failed to parse event JSON: {}", e.what());
434 return std::nullopt;
435 }
436}

438auto NatsEventConsumer::dispatch_event(const EventContext& ctx) -> HandlerResult {
439 // Copy handler function under mutex, then release before calling
440 // This avoids deadlock if handler throws and we need to lock for error handler
441 EventHandler handler;
442 bool use_any_handler = false;
443
444 {
445 const std::scoped_lock lock(m_handlers_mutex);
446
447 // Check event type filter
448 if (!m_config.event_type_filter.empty()) {
449 bool matched = false;
450 for (const auto& filter : m_config.event_type_filter) {
451 if (ctx.event_type == filter) {
452 matched = true;
453 break;
454 }
455 }
456 if (!matched) {
457 spdlog::debug("Event type {} not in filter, auto-acking", ctx.event_type);
458 return HandlerResult::Ack;
459 }
460 }
461
462 // Try specific handler
463 auto it = m_handlers.find(ctx.event_type);
464 if (it != m_handlers.end()) {
465 handler = it->second;
466 } else if (m_any_handler) {
467 handler = m_any_handler;
468 use_any_handler = true;
469 }
470 }
471 // mutex released here
472
473 if (handler) {
474 spdlog::debug("Dispatching event {} to {} handler",
475 ctx.event_id,
476 use_any_handler ? "catch-all" : ctx.event_type);
477 return handler(ctx);
478 }
479
480 // No handler, auto-ack
481 spdlog::debug("No handler for event type {}, auto-acking", ctx.event_type);
482 return HandlerResult::Ack;
483}

485void NatsEventConsumer::acknowledge(natsMsg* msg, HandlerResult result) {
486 switch (result) {
487 case HandlerResult::Ack:
488 natsMsg_Ack(msg, nullptr);
489 break;
490 case HandlerResult::Nak:
491 natsMsg_Nak(msg, nullptr);
492 break;
493 case HandlerResult::Term:
494 natsMsg_Term(msg, nullptr);
495 break;
496 case HandlerResult::InProgress:
497 // Handler will ack/nak later
498 natsMsg_InProgress(msg, nullptr);
499 break;
500 }
501}

503auto NatsEventConsumer::is_duplicate(const std::string& event_id) -> bool {
504 const std::scoped_lock lock(m_dedupe_mutex);
505 auto it = m_dedupe_cache.find(event_id);
506 if (it == m_dedupe_cache.end()) {
507 return false;
508 }
509 // Check if entry has expired
510 auto now = std::chrono::steady_clock::now();
511 auto age = std::chrono::duration_cast<std::chrono::seconds>(now - it->second.timestamp);
512 return age < m_config.dedupe_window;
513}

520void NatsEventConsumer::cleanup_dedupe_cache() {
521 auto now = std::chrono::steady_clock::now();
522 if (now - m_last_dedupe_cleanup < DEDUPE_CLEANUP_INTERVAL) {
523 return;
524 }
525
526 const std::scoped_lock lock(m_dedupe_mutex);
527 auto it = m_dedupe_cache.begin();
528 while (it != m_dedupe_cache.end()) {
529 auto age = std::chrono::duration_cast<std::chrono::seconds>(now - it->second.timestamp);
530 if (age >= m_config.dedupe_window) {
531 it = m_dedupe_cache.erase(it);
532 } else {
533 ++it;
534 }
535 }
536 m_last_dedupe_cleanup = now;
537}
