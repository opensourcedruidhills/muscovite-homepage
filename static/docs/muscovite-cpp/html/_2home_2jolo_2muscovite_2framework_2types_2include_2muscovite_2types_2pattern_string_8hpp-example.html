namespace muscovite::types {
template <StringLiteral Pattern>
class PatternString {
public:
using value_type = std::string;
[[nodiscard]] static constexpr auto pattern() noexcept -> const char* {
return Pattern.c_str();
}
[[nodiscard]] static auto is_valid(std::string_view sv) noexcept -> bool {
try {
return std::regex_match(sv.begin(), sv.end(), cached_regex());
} catch (...) {
return false;
}
}
[[nodiscard]] static auto try_from(std::string_view sv)
-> std::expected<PatternString, std::string> {
if (!is_valid(sv)) {
return std::unexpected(
std::format("String '{}' does not match pattern '{}'",
sv, Pattern.c_str()));
}
return PatternString{std::string{sv}, trusted_tag{}};
}
[[nodiscard]] static auto from_trusted(std::string value) noexcept
-> PatternString {
return PatternString{std::move(value), trusted_tag{}};
}
[[nodiscard]] auto value() const noexcept -> const std::string& {
return value_;
}
[[nodiscard]] operator std::string_view() const noexcept {
return value_;
}
[[nodiscard]] auto operator<=>(const PatternString&) const noexcept = default;
[[nodiscard]] auto operator==(const PatternString&) const noexcept -> bool = default;
private:
struct trusted_tag {};
PatternString(std::string v, trusted_tag) noexcept
: value_(std::move(v)) {}
[[nodiscard]] static auto cached_regex() -> const std::regex& {
static const std::regex re{Pattern.c_str(), std::regex::optimize};
return re;
}
std::string value_;
};
} // namespace muscovite::types
