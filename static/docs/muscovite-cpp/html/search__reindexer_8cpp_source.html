45auto build_ndjson(
46 const std::string& index_name,
47 const std::vector<std::pair<std::string, nlohmann::json>>& documents)
48 -> std::string {
49 std::string ndjson;
50 // Pre-allocate: ~200 bytes per doc pair is a reasonable estimate
51 ndjson.reserve(documents.size() * 200);
52
53 for (const auto& [doc_id, doc_body] : documents) {
54 // Action line
55 nlohmann::json const action = {{"index", {{"_index", index_name}, {"_id", doc_id}}}};
56 ndjson += action.dump();
57 ndjson += '\n';
58
59 // Source line
60 ndjson += doc_body.dump();
61 ndjson += '\n';
62 }
63
64 return ndjson;
65}

78void SearchReindexer::register_mapper(std::shared_ptr<BulkDocumentMapper> mapper) {
79 if (!mapper) {
80 throw std::invalid_argument("SearchReindexer::register_mapper: mapper is null");
81 }
82
83 auto type = mapper->entity_type();
84 if (mappers_.contains(type)) {
85 throw std::logic_error(
86 "SearchReindexer::register_mapper: mapper already registered for '" + type + "'");
87 }
88
89 spdlog::info("SearchReindexer: registered mapper for '{}' (index: '{}')",
90 type, mapper->index_name());
91 mappers_[std::move(type)] = std::move(mapper);
92}

94auto SearchReindexer::reindex_all(const ReindexConfig& config, const ReindexProgressCallback& on_progress)
95 -> ReindexResult {
96
97 ReindexResult result;
98
99 // Sort entity types for deterministic ordering
100 auto types = registered_types();
101
102 for (const auto& type : types) {
103 auto entity_result = reindex_entity(type, config, on_progress);
104 result.entity_results.push_back(std::move(entity_result));
105 }
106
107 spdlog::info("SearchReindexer: reindex complete — {} indexed, {} failed across {} types",
108 result.total_indexed(), result.total_failed(), types.size());
109
110 return result;
111}

113auto SearchReindexer::reindex_entity(const std::string& entity_type, const ReindexConfig& config,
114 const ReindexProgressCallback& on_progress) -> ReindexEntityResult {
115
116 auto it = mappers_.find(entity_type);
117 if (it == mappers_.end()) {
118 throw std::invalid_argument(
119 "SearchReindexer::reindex_entity: no mapper for '" + entity_type + "'");
120 }
121
122 const auto& mapper = it->second;
123 const auto index = mapper->index_name();
124 const auto total_estimate = mapper->estimated_count();
125
126 ReindexEntityResult result;
127 result.entity_type = entity_type;
128 result.index_name = index;
129
130 spdlog::info("SearchReindexer: starting reindex of '{}' (index: '{}', estimated: {})",
131 entity_type, index, total_estimate);
132
133 try {
134 // Step 1: Drop + recreate index (if configured)
135 if (config.recreate_indexes) {
136 if (client_.index_exists(index)) {
137 client_.delete_index(index);
138 spdlog::debug("SearchReindexer: dropped index '{}'", index);
139 }
140 client_.create_index(index, mapper->index_mapping());
141 spdlog::debug("SearchReindexer: created index '{}' with fresh mapping", index);
142 } else {
143 // Ensure index exists
144 if (!client_.index_exists(index)) {
145 client_.create_index(index, mapper->index_mapping());
146 }
147 }
148
149 // Step 2: Paginated bulk index
150 std::size_t offset = 0;
151 while (true) {
152 auto page = mapper->fetch_page(offset, config.batch_size);
153 if (page.empty()) {
154 break; // No more documents
155 }
156
157 // Build NDJSON and submit
158 auto ndjson = build_ndjson(index, page);
159 auto bulk_response = client_.bulk(ndjson);
160
161 result.documents_indexed += static_cast<std::size_t>(bulk_response.success_count);
162 result.documents_failed += static_cast<std::size_t>(bulk_response.error_count);
163 result.batches++;
164
165 offset += page.size();
166
167 // Progress callback
168 if (on_progress) {
169 on_progress(entity_type, offset, total_estimate);
170 }
171
172 spdlog::debug("SearchReindexer: batch {} for '{}' — {} indexed, {} failed",
173 result.batches, entity_type,
174 bulk_response.success_count, bulk_response.error_count);
175
176 // If page smaller than batch_size, we've reached the end
177 if (page.size() < config.batch_size) {
178 break;
179 }
180 }
181
182 result.success = true;
183 spdlog::info("SearchReindexer: finished '{}' — {} indexed, {} failed in {} batches",
184 entity_type, result.documents_indexed, result.documents_failed, result.batches);
185
186 } catch (const SearchError& e) {
187 result.success = false;
188 result.error = e.what();
189 spdlog::error("SearchReindexer: failed to reindex '{}': {}", entity_type, e.what());
190
191 } catch (const std::exception& e) {
192 result.success = false;
193 result.error = e.what();
194 spdlog::error("SearchReindexer: unexpected error reindexing '{}': {}",
195 entity_type, e.what());
196 }
197
198 return result;
199}

201auto SearchReindexer::registered_types() const -> std::vector<std::string> {
202 std::vector<std::string> types;
203 types.reserve(mappers_.size());
204 for (const auto& [type, _] : mappers_) {
205 types.push_back(type);
206 }
207 std::ranges::sort(types);
208 return types;
209}
