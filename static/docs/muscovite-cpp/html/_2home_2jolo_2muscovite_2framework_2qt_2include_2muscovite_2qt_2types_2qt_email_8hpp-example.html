#pragma once
#include <muscovite/qt/types/qt_type_base.hpp>
#include <QRegularExpression>
#include <QString>
namespace muscovite::qt {
class QtEmail : public QtTypeBase<QtEmail> {
public:
using QtTypeBase::QtTypeBase;
[[nodiscard]] static auto typeName() -> QString {
return QStringLiteral("Email");
}
[[nodiscard]] static auto pattern() -> QString {
return QStringLiteral(R"(^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$)");
}
[[nodiscard]] static auto isValid(QStringView value) -> bool {
if (value.isEmpty()) return false;
static const QRegularExpression re(pattern());
return re.matchView(value).hasMatch();
}
[[nodiscard]] auto localPart() const -> QString {
auto pos = value_.indexOf('@');
if (pos < 0) return {};
return value_.left(static_cast<int>(pos));
}
[[nodiscard]] auto domain() const -> QString {
auto pos = value_.indexOf('@');
if (pos < 0) return {};
return value_.mid(static_cast<int>(pos) + 1);
}
};
} // namespace muscovite::qt
