Code/identifier with format validation.

Code/identifier with format validation Validates against code pattern: ^[A-Z][A-Z0-9_]{0,29}$

Requirements:

  • Starts with uppercase letter
  • Contains only uppercase letters, digits, underscores
  • Length 1-30 characters
auto status = Code("ACTIVE"); // OK
auto code = Code("TYPE_A1"); // OK
auto bad = Code("active"); // throws (lowercase)
auto bad2 = Code("123ABC"); // throws (starts with digit)

#pragma once
#include <algorithm>
#include <cctype>
#include <regex>
#include <string_view>
#include <muscovite/types/strong_type_base.hpp>
namespace muscovite::types {
class Code : public strong_type_base<Code> {
public:
using strong_type_base::strong_type_base;
[[nodiscard]] static constexpr auto type_name() noexcept -> std::string_view {
return "Code";
}
[[nodiscard]] static constexpr auto pattern() noexcept -> std::string_view {
return R"(^[A-Z][A-Z0-9_]{0,29}$)";
}
[[nodiscard]] static auto is_valid(std::string_view value) noexcept -> bool {
if (value.empty() || value.size() > 30) return false;
// First character must be uppercase letter
if (std::isupper(static_cast<unsigned char>(value[0])) == 0)
return false;
// Rest must be uppercase, digits, or underscore
return std::all_of(value.begin() + 1, value.end(), [](unsigned char c) { // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
return std::isupper(c) || std::isdigit(c) || c == '_';
});
}
[[nodiscard]] static auto from_string(std::string_view value) -> Code {
std::string upper;
upper.reserve(value.size());
std::ranges::transform(value, std::back_inserter(upper), [](unsigned char c) { return std::toupper(c); });
return Code{upper};
}
};
} // namespace muscovite::types
