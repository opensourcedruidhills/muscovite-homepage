27class MetricsStore {
28public:
29 static auto instance() -> MetricsStore& {
30 static MetricsStore store;
31 return store;
32 }
33
34 void record_request(const std::string& method) {
35 const std::scoped_lock lock{m_mutex};
36 ++m_requests[method];
37 ++m_total_requests;
38 }
39
40 void record_response(const std::string& method, ::grpc::StatusCode code,
41 std::chrono::microseconds duration) {
42 const std::scoped_lock lock{m_mutex};
43
44 // Count by status
45 if (code == ::grpc::StatusCode::OK) {
46 ++m_success[method];
47 ++m_total_success;
48 } else {
49 ++m_errors[method];
50 ++m_total_errors;
51 }
52
53 // Record duration (simple average for now)
54 auto& sum = m_duration_sum[method];
55 auto& count = m_duration_count[method];
56 sum += static_cast<uint64_t>(duration.count());
57 ++count;
58
59 m_total_duration_sum += static_cast<uint64_t>(duration.count());
60 ++m_total_duration_count;
61 }
62
64 [[nodiscard]] auto summary() const -> std::string {
65 const std::scoped_lock lock{m_mutex};
66
67 std::string result = "gRPC Metrics Summary:\n";
68 result += fmt::format(" Total requests: {}\n", m_total_requests.load());
69 result += fmt::format(" Total success: {}\n", m_total_success.load());
70 result += fmt::format(" Total errors: {}\n", m_total_errors.load());
71
72 if (m_total_duration_count > 0) {
73 auto avg = m_total_duration_sum.load() / m_total_duration_count.load();
74 result += fmt::format(" Average latency: {}µs\n", avg);
75 }
76
77 return result;
78 }
79
80private:
81 MetricsStore() = default;
82
83 mutable std::mutex m_mutex;
84 std::unordered_map<std::string, uint64_t> m_requests;
85 std::unordered_map<std::string, uint64_t> m_success;
86 std::unordered_map<std::string, uint64_t> m_errors;
87 std::unordered_map<std::string, uint64_t> m_duration_sum;
88 std::unordered_map<std::string, uint64_t> m_duration_count;
89
90 std::atomic<uint64_t> m_total_requests{0};
91 std::atomic<uint64_t> m_total_success{0};
92 std::atomic<uint64_t> m_total_errors{0};
93 std::atomic<uint64_t> m_total_duration_sum{0};
94 std::atomic<uint64_t> m_total_duration_count{0};
95};

97class MetricsInterceptor : public ::grpc::experimental::Interceptor {
98public:
99 explicit MetricsInterceptor(::grpc::experimental::ServerRpcInfo* info)
100 : m_method_name{info != nullptr ? info->method() : "unknown"}
101 , m_start_time{std::chrono::steady_clock::now()} {
102
103 MetricsStore::instance().record_request(m_method_name);
104 }
105
106 void Intercept(::grpc::experimental::InterceptorBatchMethods* methods) override {
107 // Record metrics when response status is set
108 if (methods->QueryInterceptionHookPoint(
109 ::grpc::experimental::InterceptionHookPoints::PRE_SEND_STATUS)) {
110 auto end_time = std::chrono::steady_clock::now();
111 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(
112 end_time - m_start_time);
113
114 auto status = methods->GetSendStatus();
115 auto code = status.error_code();
116
117 MetricsStore::instance().record_response(m_method_name, code, duration);
118 }
119
120 methods->Proceed();
121 }
122
123private:
124 std::string m_method_name;
125 std::chrono::steady_clock::time_point m_start_time;
126};

136auto create_default_interceptors()
137 -> std::vector<std::unique_ptr<::grpc::experimental::ServerInterceptorFactoryInterface>> {
138
139 std::vector<std::unique_ptr<::grpc::experimental::ServerInterceptorFactoryInterface>> factories;
140
141 // Order matters: CorrelationId → Error → Metrics → Logging (outermost to innermost)
142 // CorrelationId must be first so all other interceptors can access the ID
143 factories.push_back(std::make_unique<CorrelationIdInterceptorFactory>());
144 factories.push_back(std::make_unique<ErrorInterceptorFactory>());
145 factories.push_back(std::make_unique<MetricsInterceptorFactory>());
146 factories.push_back(std::make_unique<LoggingInterceptorFactory>());
147
148 return factories;
149}
