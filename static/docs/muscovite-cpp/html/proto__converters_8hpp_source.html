94[[nodiscard]] inline auto timestamp_to_iso8601(
95 const std::chrono::system_clock::time_point& ts) -> std::string {
96 auto time_t_val = std::chrono::system_clock::to_time_t(ts);
97 std::tm tm_buf{};
98 gmtime_r(&time_t_val, &tm_buf);
99
100 char buffer[32];
101 std::strftime(buffer, sizeof(buffer), "%Y-%m-%dT%H:%M:%SZ", &tm_buf);
102 return std::string{buffer};
103}

109[[nodiscard]] inline auto iso8601_to_timestamp(std::string_view s)
110 -> std::chrono::system_clock::time_point {
111 if (s.empty()) {
112 throw std::invalid_argument("Timestamp string cannot be empty");
113 }
114
115 int year{}, month{}, day{}, hour{}, minute{}, second{};
116 // NOLINTNEXTLINE(cert-err34-c) — sscanf is appropriate for fixed-format parsing
117 int matched = std::sscanf(std::string{s}.c_str(), "%d-%d-%dT%d:%d:%d",
118 &year, &month, &day, &hour, &minute, &second);
119 if (matched != 6) {
120 throw std::invalid_argument(
121 std::string{"Invalid timestamp format (expected ISO 8601): "} +
122 std::string{s});
123 }
124
125 std::tm tm{};
126 tm.tm_year = year - 1900;
127 tm.tm_mon = month - 1;
128 tm.tm_mday = day;
129 tm.tm_hour = hour;
130 tm.tm_min = minute;
131 tm.tm_sec = second;
132 tm.tm_isdst = 0;
133
134 auto time_t_val = timegm(&tm);
135 return std::chrono::system_clock::from_time_t(time_t_val);
136}

138// ============================================================================
139// Date Conversions (ISO 8601 ↔ year_month_day)
140// ============================================================================

145[[nodiscard]] inline auto date_to_iso8601(
146 const std::chrono::year_month_day& date) -> std::string {
147 auto y = static_cast<int>(date.year());
148 auto m = static_cast<unsigned>(date.month());
149 auto d = static_cast<unsigned>(date.day());
150
151 char buffer[16];
152 std::snprintf(buffer, sizeof(buffer), "%04d-%02u-%02u", y, m, d);
153 return std::string{buffer};
154}

160[[nodiscard]] inline auto iso8601_to_date(std::string_view s)
161 -> std::chrono::year_month_day {
162 if (s.empty()) {
163 throw std::invalid_argument("Date string cannot be empty");
164 }
165
166 int year{}, month{}, day{};
167 // NOLINTNEXTLINE(cert-err34-c) — sscanf is appropriate for fixed-format parsing
168 int matched = std::sscanf(std::string{s}.c_str(), "%d-%d-%d",
169 &year, &month, &day);
170 if (matched != 3) {
171 throw std::invalid_argument(
172 std::string{"Invalid date format (expected YYYY-MM-DD): "} +
173 std::string{s});
174 }
175
176 return std::chrono::year{year} /
177 std::chrono::month{static_cast<unsigned>(month)} /
178 std::chrono::day{static_cast<unsigned>(day)};
179}
