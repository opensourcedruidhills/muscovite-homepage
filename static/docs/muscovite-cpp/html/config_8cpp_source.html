30auto to_spdlog_level(std::string_view level_str) -> spdlog::level::level_enum {
31 std::string lower{level_str};
32 std::ranges::transform(lower, lower.begin(), ::tolower);
33
34 if (lower == "trace")
35 return spdlog::level::trace;
36 if (lower == "debug")
37 return spdlog::level::debug;
38 if (lower == "info")
39 return spdlog::level::info;
40 if (lower == "warn" || lower == "warning")
41 return spdlog::level::warn;
42 if (lower == "error" || lower == "err")
43 return spdlog::level::err;
44 if (lower == "critical" || lower == "fatal")
45 return spdlog::level::critical;
46 if (lower == "off")
47 return spdlog::level::off;
48
49 return spdlog::level::info; // Default
50}

53auto get_env(std::string_view prefix, std::string_view name) -> std::optional<std::string> {
54 std::string full_name{prefix};
55 full_name += name;
56
57 // Convert to uppercase
58 std::ranges::transform(full_name, full_name.begin(), ::toupper);
59
60 if (auto* value = std::getenv(full_name.c_str())) { // NOLINT(concurrency-mt-unsafe)
61 return std::string{value};
62 }
63 return std::nullopt;
64}
