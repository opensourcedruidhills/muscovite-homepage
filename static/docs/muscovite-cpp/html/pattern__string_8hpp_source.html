52 [[nodiscard]] static auto is_valid(std::string_view sv) noexcept -> bool {
53 try {
54 return std::regex_match(sv.begin(), sv.end(), cached_regex());
55 } catch (...) {
56 return false;
57 }
58 }

61 [[nodiscard]] static auto try_from(std::string_view sv)
62 -> std::expected<PatternString, std::string> {
63 if (!is_valid(sv)) {
64 return std::unexpected(
65 std::format("String '{}' does not match pattern '{}'",
66 sv, Pattern.c_str()));
67 }
68 return PatternString{std::string{sv}, trusted_tag{}};
69 }
