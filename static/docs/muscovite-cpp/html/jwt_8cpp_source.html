16// Simple base64 decode implementation
17std::string base64_decode(const std::string& encoded) {
18 static constexpr std::array<int, 256> decode_table = []() {
19 std::array<int, 256> t{};
20 t.fill(-1);
21 for (int i = 0; i < 26; ++i) {
22 t['A' + i] = i;
23 t['a' + i] = i + 26;
24 }
25 for (int i = 0; i < 10; ++i) {
26 t['0' + i] = i + 52;
27 }
28 t['+'] = 62;
29 t['/'] = 63;
30 t['='] = 0; // padding
31 return t;
32 }();
33
34 std::string decoded;
35 decoded.reserve((encoded.size() * 3) / 4);
36
37 std::uint32_t buf = 0;
38 int bits = 0;
39
40 for (char c : encoded) {
41 if (c == '=') break;
42 int val = decode_table[static_cast<unsigned char>(c)];
43 if (val < 0) continue; // Skip invalid characters
44 buf = (buf << 6) | static_cast<std::uint32_t>(val);
45 bits += 6;
46 if (bits >= 8) {
47 bits -= 8;
48 decoded.push_back(static_cast<char>((buf >> bits) & 0xFF));
49 }
50 }
51 return decoded;
52}
53} // anonymous namespace

55std::string JWT::decodeBase64Url(const std::string& encoded) {
56 // Base64url decode (convert - to +, _ to /)
57 std::string modified = encoded;
58 std::replace(modified.begin(), modified.end(), '-', '+');
59 std::replace(modified.begin(), modified.end(), '_', '/');
60
61 // Add padding if needed
62 switch (modified.length() % 4) {
63 case 2:
64 modified += "==";
65 break;
66 case 3:
67 modified += "=";
68 break;
69 }
70
71 return base64_decode(modified);
72}

74json JWT::extractClaims(const std::string& token) {
75 // JWT format: header.payload.signature
76 std::istringstream iss(token);
77 std::string header, payload, signature;
78
79 if (!std::getline(iss, header, '.') ||
80 !std::getline(iss, payload, '.') ||
81 !std::getline(iss, signature)) {
82 throw std::invalid_argument("Invalid JWT format");
83 }
84
85 try {
86 std::string decoded_payload = decodeBase64Url(payload);
87 return json::parse(decoded_payload);
88 } catch (const json::exception& e) {
89 throw std::invalid_argument(
90 fmt::format("Failed to parse JWT claims: {}", e.what())
91 );
92 }
93}

95bool JWT::validateStandardClaims(
96 const json& claims,
97 const std::string& expected_issuer,
98 const std::string& expected_audience
99) {
100 // Check issuer
101 if (claims.contains("iss")) {
102 if (claims.at("iss").get<std::string>() != expected_issuer) {
103 return false;
104 }
105 }
106
107 // Check audience
108 if (claims.contains("aud")) {
109 auto aud = claims.at("aud");
110 if (aud.is_string()) {
111 if (aud.get<std::string>() != expected_audience) {
112 return false;
113 }
114 } else if (aud.is_array()) {
115 bool found = false;
116 for (const auto& a : aud) {
117 if (a.get<std::string>() == expected_audience) {
118 found = true;
119 break;
120 }
121 }
122 if (!found) {
123 return false;
124 }
125 }
126 }
127
128 // Check expiration
129 if (isExpired(claims)) {
130 return false;
131 }
132
133 // Check not-before
134 if (claims.contains("nbf")) {
135 auto nbf = std::time(nullptr);
136 if (nbf < claims.at("nbf").get<std::time_t>()) {
137 return false;
138 }
139 }
140
141 return true;
142}

144bool JWT::isExpired(const json& claims, int clock_skew_seconds) {
145 if (!claims.contains("exp")) {
146 return false; // No expiration
147 }
148
149 auto now = std::time(nullptr);
150 auto exp = claims.at("exp").get<std::time_t>();
151 return now > (exp + clock_skew_seconds);
152}

164std::string JWT::extractSubject(const json& claims) {
165 if (claims.contains("sub")) {
166 return claims.at("sub").get<std::string>();
167 }
168 if (claims.contains("preferred_username")) {
169 return claims.at("preferred_username").get<std::string>();
170 }
171 if (claims.contains("name")) {
172 return claims.at("name").get<std::string>();
173 }
174 throw std::invalid_argument("No subject identifier found in JWT claims");
175}
