15void example_builtin_auth() {
16 std::cout << "=== Example 1: Built-in Authentication ===" << std::endl;
17
18 // Configure for built-in auth
19 auto config = OAuth2Config::forBuiltIn("http://localhost:8000/auth");
20
21 // Create authorization flow handler
22 auto flow = std::make_shared<AuthorizationCodeFlow>(config);
23
24 // Create gRPC interceptor
25 auto interceptor = std::make_shared<OAuth2Interceptor>(flow);
26
27 // Authenticate with username/password
28 try {
29 interceptor->authenticateBuiltIn("john.doe", "secure_password_123");
30 std::cout << "✓ Authenticated as: " << interceptor->getUsername().value() << std::endl;
31 } catch (const std::exception& e) {
32 std::cerr << "✗ Authentication failed: " << e.what() << std::endl;
33 }
34
35 // Create gRPC channel with OAuth2
36 auto channel = createSecureChannelWithOAuth2(
37 "localhost:50051",
38 interceptor
39 );
40 std::cout << "✓ gRPC channel created with OAuth2" << std::endl;
41
42 // ... use channel to create service stubs ...
43 // All RPC calls now include Authorization: Bearer <token>
44
45 // Logout
46 interceptor->logout();
47 std::cout << "✓ Logged out" << std::endl;
48}

170void example_jwt_claims() {
171 std::cout << "\n=== Example 6: JWT Claims ===" << std::endl;
172
173 // Parse JWT token (would be from interceptor->getAccessToken())
174 std::string id_token = "eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9..." // actual JWT
175
176 try {
177 auto claims = JWT::extractClaims(id_token);
178 std::cout << "✓ Extracted JWT claims" << std::endl;
179
180 // Validate standard OIDC claims
181 bool valid = JWT::validateStandardClaims(
182 claims,
183 "https://keycloak.example.com/realms/test", // expected issuer
184 "test-client" // expected audience
185 );
186
187 if (valid) {
188 std::cout << "✓ JWT claims valid" << std::endl;
189
190 // Extract subject (username)
191 auto subject = JWT::extractSubject(claims);
192 std::cout << "✓ Subject (user): " << subject << std::endl;
193 } else {
194 std::cout << "✗ JWT claims validation failed" << std::endl;
195 }
196 } catch (const std::exception& e) {
197 std::cerr << "✗ JWT parsing failed: " << e.what() << std::endl;
198 }
199}
