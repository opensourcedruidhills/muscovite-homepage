Postal code with format validation.

Postal code with format validation Validates against flexible postal code pattern: ^[A-Za-z0-9\s\-]{3,10}$

Accepts:

  • US 5-digit: 12345
  • US 9-digit: 12345-6789
  • Canadian: K1A 0B1
  • UK: SW1A 1AA
auto zip = PostalCode("12345"); // OK
auto canada = PostalCode("K1A 0B1"); // OK
auto bad = PostalCode("AB"); // throws (too short)

#pragma once
#include <muscovite/types/strong_type_base.hpp>
#include <regex>
#include <string_view>
namespace muscovite::types {
class PostalCode : public strong_type_base<PostalCode> {
public:
using strong_type_base::strong_type_base;
[[nodiscard]] static constexpr auto type_name() noexcept -> std::string_view {
return "PostalCode";
}
[[nodiscard]] static constexpr auto pattern() noexcept -> std::string_view {
return R"(^[A-Za-z0-9\s\-]{3,10}$)";
}
[[nodiscard]] static auto is_valid(std::string_view value) noexcept -> bool {
if (value.empty()) return false;
try {
static const std::regex pattern_{std::string{pattern()}, std::regex::optimize};
return std::regex_match(value.begin(), value.end(), pattern_);
} catch (...) {
return false;
}
}
[[nodiscard]] auto is_us_zip() const noexcept -> bool {
try {
static const std::regex us_zip{R"(^\d{5}(-\d{4})?$)", std::regex::optimize};
return std::regex_match(value_, us_zip);
} catch (...) {
return false;
}
}
[[nodiscard]] auto is_canadian() const noexcept -> bool {
try {
static const std::regex ca_postal{
R"(^[A-Za-z]\d[A-Za-z]\s?\d[A-Za-z]\d$)", std::regex::optimize};
return std::regex_match(value_, ca_postal);
} catch (...) {
return false;
}
}
};
} // namespace muscovite::types
