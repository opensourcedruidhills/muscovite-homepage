16PagedSearchResult PagedSearchResult::from_search_result(
17 const SearchResult& result,
18 std::int32_t page_,
19 std::int32_t page_size_) {
20 PagedSearchResult paged;
21 paged.hits = result.hits;
22 paged.total = result.total_hits;
23 paged.page = page_;
24 paged.page_size = page_size_;
25 paged.took_ms = result.took_ms;
26
27 // Calculate has_more — widen before multiply to avoid int overflow
28 auto offset = (static_cast<std::int64_t>(page_) - 1) * static_cast<std::int64_t>(page_size_);
29 paged.has_more = (offset + static_cast<std::int64_t>(result.hits.size()))
30 < result.total_hits;
31
32 // Parse aggregations from raw response
33 if (result.raw.contains("aggregations")) {
34 for (auto it = result.raw["aggregations"].begin();
35 it != result.raw["aggregations"].end(); ++it) {
36 SearchFacet facet;
37 facet.name = it.key();
38 if (it->contains("buckets") && (*it)["buckets"].is_array()) {
39 for (const auto& bucket : (*it)["buckets"]) {
40 FacetBucket fb;
41 if (bucket.contains("key")) {
42 if (bucket["key"].is_string()) {
43 fb.key = bucket["key"].get<std::string>();
44 } else {
45 fb.key = bucket["key"].dump();
46 }
47 }
48 fb.doc_count = bucket.value("doc_count", 0);
49 facet.buckets.push_back(std::move(fb));
50 }
51 }
52 paged.facets.push_back(std::move(facet));
53 }
54 }
55
56 return paged;
57}
