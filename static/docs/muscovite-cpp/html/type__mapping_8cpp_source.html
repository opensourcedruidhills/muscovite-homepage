34void to_proto(const std::chrono::system_clock::time_point& src, proto::Timestamp* dst) {
35 auto duration = src.time_since_epoch();
36 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(duration);
37 auto nanos = std::chrono::duration_cast<std::chrono::nanoseconds>(duration - seconds);
38
39 dst->set_seconds(seconds.count());
40 dst->set_nanos(static_cast<int32_t>(nanos.count()));
41}

55 dst->set_micros(static_cast<int32_t>(hms.subseconds().count()));

119auto spatial_to_wkt(const proto::LineString& src) -> std::string {
120 std::ostringstream oss;
121 oss << std::setprecision(15) << "LINESTRING(";
122 bool first = true;
123 for (const auto& pt : src.points()) {
124 if (!first) oss << ", ";
125 first = false;
126 oss << pt.longitude() << " " << pt.latitude();
127 }
128 oss << ")";
129 return oss.str();
130}

132auto spatial_to_wkt(const proto::Polygon& src) -> std::string {
133 std::ostringstream oss;
134 oss << std::setprecision(15) << "POLYGON((";
135 bool first = true;
136 for (const auto& pt : src.exterior_ring()) {
137 if (!first) oss << ", ";
138 first = false;
139 oss << pt.longitude() << " " << pt.latitude();
140 }
141 oss << "))";
142 // Note: holes (interior rings) are not handled in this basic implementation
143 return oss.str();
144}

146void wkt_to_spatial(const std::string& wkt, proto::Point* dst) {
147 // Basic WKT parsing: POINT(lon lat)
148 // More robust parsing would use a proper WKT parser
149 if (wkt.starts_with("POINT(") || wkt.starts_with("POINT (")) {
150 size_t const start = wkt.find('(') + 1;
151 size_t const end = wkt.find(')');
152 std::string const coords = wkt.substr(start, end - start);
153 std::istringstream iss(coords);
154 double lon = 0.0;
155 double lat = 0.0;
156 if (iss >> lon >> lat) {
157 dst->set_longitude(lon);
158 dst->set_latitude(lat);
159 dst->set_srid(4326); // Default to WGS84
160 }
161 }
162}

164void wkt_to_spatial(const std::string& wkt, proto::LineString* dst) {
165 // Basic WKT parsing: LINESTRING(lon1 lat1, lon2 lat2, ...)
166 if (wkt.starts_with("LINESTRING(") || wkt.starts_with("LINESTRING (")) {
167 size_t const start = wkt.find('(') + 1;
168 size_t const end = wkt.rfind(')');
169 std::string coords = wkt.substr(start, end - start);
170
171 // Split by comma and parse each point
172 size_t pos = 0;
173 while (pos < coords.size()) {
174 // Skip whitespace
175 while (pos < coords.size() && (coords[pos] == ' ' || coords[pos] == ',')) pos++;
176 if (pos >= coords.size()) break;
177
178 // Find end of this coordinate pair
179 size_t const next_comma = coords.find(',', pos);
180 std::string const pair =
181 coords.substr(pos, (next_comma != std::string::npos ? next_comma : coords.size()) - pos);
182
183 std::istringstream iss(pair);
184 double lon = 0.0;
185 double lat = 0.0;
186 if (iss >> lon >> lat) {
187 auto* pt = dst->add_points();
188 pt->set_longitude(lon);
189 pt->set_latitude(lat);
190 pt->set_srid(4326);
191 }
192
193 pos = (next_comma != std::string::npos) ? next_comma + 1 : coords.size();
194 }
195 dst->set_srid(4326);
196 }
197}

199void wkt_to_spatial(const std::string& wkt, proto::Polygon* dst) {
200 // Basic WKT parsing: POLYGON((lon1 lat1, lon2 lat2, ...))
201 if (wkt.starts_with("POLYGON((") || wkt.starts_with("POLYGON ((")) {
202 size_t const start = wkt.find("((") + 2;
203 size_t const end = wkt.find("))");
204 std::string coords = wkt.substr(start, end - start);
205
206 // Split by comma and parse each point
207 size_t pos = 0;
208 while (pos < coords.size()) {
209 // Skip whitespace

211 if (pos >= coords.size()) break;
