27// Convert proto Session to Qt Session
28Session fromProto(const ::muscovite::security::v1::Session& proto) {
29 Session session;
30 session.id = QString::fromStdString(proto.id().value());
31 session.principalId = QString::fromStdString(proto.principal_id().value());
32 session.username = QString::fromStdString(proto.username());
33 session.displayName = QString::fromStdString(proto.display_name());
34 session.accessToken = QString::fromStdString(proto.access_token());
35 session.refreshToken = QString::fromStdString(proto.refresh_token());
36
37 for (const auto& group : proto.groups()) {
38 session.groups.append(QString::fromStdString(group));
39 }
40 for (const auto& priv : proto.privileges()) {
41 session.privileges.append(QString::fromStdString(priv));
42 }
43
44 // Convert timestamps
45 const auto& expTs = proto.expires_at();
46 session.expiresAt = QDateTime::fromSecsSinceEpoch(expTs.seconds());
47
48 const auto& refExpTs = proto.refresh_expires_at();
49 session.refreshExpiresAt = QDateTime::fromSecsSinceEpoch(refExpTs.seconds());
50
51 return session;
52}

75 void createChannel() {
76 if (config.serverAddress.isEmpty()) {
77 return;
78 }
79
80 std::shared_ptr<grpc::ChannelCredentials> creds;
81 if (config.useTls) {
82 grpc::SslCredentialsOptions sslOpts;
83 if (!config.caCertPath.isEmpty()) {
84 // Read CA cert file
85 QFile certFile(config.caCertPath);
86 if (certFile.open(QIODevice::ReadOnly)) {
87 sslOpts.pem_root_certs = certFile.readAll().toStdString();
88 }
89 }
90 creds = grpc::SslCredentials(sslOpts);
91 } else {
92 creds = grpc::InsecureChannelCredentials();
93 }
94
95 channel = grpc::CreateChannel(config.serverAddress.toStdString(), creds);
96 authStub = ::muscovite::security::v1::AuthService::NewStub(channel);
97 }
98
99 void setupTimers() {
100 if (refreshTimer == nullptr) {
101 refreshTimer = new QTimer(q);
102 QObject::connect(refreshTimer, &QTimer::timeout, q, &AuthManager::refreshToken);
103 }
104
105 if (expiryWarningTimer == nullptr) {
106 expiryWarningTimer = new QTimer(q);
107 QObject::connect(expiryWarningTimer, &QTimer::timeout, [this]() {
108 if (currentSession) {
109 const auto secsRemaining = QDateTime::currentDateTime().secsTo(currentSession->expiresAt);
110 if (secsRemaining <= 60) {
111 emit q->sessionExpiring(static_cast<int>(secsRemaining));
112 }
113 if (secsRemaining <= 0) {
114 emit q->sessionExpired();
115 expiryWarningTimer->stop();
116 }
117 }
118 });
119 }
120 }
121
122 void scheduleRefresh() {
123 if (!currentSession || !config.autoRefresh) {
124 refreshTimer->stop();
125 return;
126 }
127
128 // Schedule refresh before expiry
129 const auto secsUntilExpiry = QDateTime::currentDateTime().secsTo(currentSession->expiresAt);
130 const auto refreshIn = std::max(qint64{1}, secsUntilExpiry - config.refreshBeforeExpirySecs);
131
132 refreshTimer->start(static_cast<int>(refreshIn * 1000));
133
134 // Also start expiry warning timer
135 expiryWarningTimer->start(10000); // Check every 10 seconds
136 }
137
138 void clearSession() {
139 const QMutexLocker lock(&mutex);
140 currentSession = std::nullopt;
141 refreshTimer->stop();
142 expiryWarningTimer->stop();
143 }
144};

245void AuthManager::login(const QString& username, const QString& password) {
246 if (impl_->loginInProgress.load()) {
247 return;
248 }
249
250 if (!impl_->authStub) {
251 impl_->lastError = "Authentication service not configured";
252 emit loginFailed(impl_->lastError);
253 emit errorOccurred(impl_->lastError);
254 return;
255 }
256
257 impl_->loginInProgress.store(true);
258 emit loginProgressChanged();
259
260 // Run gRPC call in a thread
261 QThread* thread = QThread::create([this, username, password]() {
262 ::muscovite::security::v1::LoginRequest request;
263 request.set_username(username.toStdString());
264 request.set_password(password.toStdString());
265
266 ::muscovite::security::v1::LoginResponse response;
267 grpc::ClientContext context;
268
269 const grpc::Status status = impl_->authStub->Login(&context, request, &response);
270
271 // Post result back to main thread
272 QMetaObject::invokeMethod(this, [this, status, response]() {
273 impl_->loginInProgress.store(false);
274 emit loginProgressChanged();
275
276 if (!status.ok()) {
277 impl_->lastError = QString::fromStdString(status.error_message());
278 emit loginFailed(impl_->lastError);
279 emit errorOccurred(impl_->lastError);
280 return;
281 }
282
283 if (!response.success()) {
284 impl_->lastError = QString::fromStdString(response.error_message());
285 emit loginFailed(impl_->lastError);
286 emit errorOccurred(impl_->lastError);
287 return;
288 }
289
290 // Store session
291 {
292 const QMutexLocker lock(&impl_->mutex);
293 impl_->currentSession = fromProto(response.session());
294 }
295
296 impl_->scheduleRefresh();
297
298 emit authStateChanged();
299 emit loginSucceeded(*impl_->currentSession);
300 }, Qt::QueuedConnection);
301 });
302
303 connect(thread, &QThread::finished, thread, &QThread::deleteLater);
304 thread->start();
305}

307void AuthManager::logout() {
308 if (!impl_->currentSession || !impl_->authStub) {
309 return;
310 }
311
312 const QString sessionId = impl_->currentSession->id;
313
314 // Clear session immediately
315 impl_->clearSession();
316 emit authStateChanged();
317
318 // Async logout call
319 QThread* thread = QThread::create([this, sessionId]() {
320 ::muscovite::security::v1::LogoutRequest request;
321 request.mutable_session_id()->set_value(sessionId.toStdString());
322
323 ::google::protobuf::Empty response;
324 grpc::ClientContext context;
325
326 impl_->authStub->Logout(&context, request, &response);
327
328 QMetaObject::invokeMethod(this, [this]() {
329 emit loggedOut();
330 }, Qt::QueuedConnection);
331 });
332
333 connect(thread, &QThread::finished, thread, &QThread::deleteLater);
334 thread->start();
335}

364void AuthManager::refreshToken() {
365 if (!impl_->currentSession || !impl_->authStub) {
366 return;
367 }
368
369 const QString refreshToken = impl_->currentSession->refreshToken;
370
371 QThread* thread = QThread::create([this, refreshToken]() {
372 ::muscovite::security::v1::RefreshTokenRequest request;
373 request.set_refresh_token(refreshToken.toStdString());
374
375 ::muscovite::security::v1::RefreshTokenResponse response;
376 grpc::ClientContext context;
377
378 const grpc::Status status = impl_->authStub->RefreshToken(&context, request, &response);
379
380 QMetaObject::invokeMethod(this, [this, status, response]() {
381 if (!status.ok() || !response.success()) {
382 const QString error = status.ok()
383 ? QString::fromStdString(response.error_message())
384 : QString::fromStdString(status.error_message());
385 impl_->lastError = error;
386 emit tokenRefreshFailed(error);
387 emit errorOccurred(error);
388
389 // Session expired - clear and notify
390 impl_->clearSession();
391 emit authStateChanged();
392 emit sessionExpired();
393 return;
394 }
395
396 // Update session
397 {
398 const QMutexLocker lock(&impl_->mutex);
399 impl_->currentSession = fromProto(response.session());
400 }
401
402 impl_->scheduleRefresh();
403 emit tokenRefreshed();
404 }, Qt::QueuedConnection);
405 });
406
407 connect(thread, &QThread::finished, thread, &QThread::deleteLater);
408 thread->start();
409}

411void AuthManager::changePassword(const QString& currentPassword, const QString& newPassword) {
412 if (!impl_->currentSession || !impl_->authStub) {
413 impl_->lastError = "Not authenticated";
414 emit passwordChangeFailed(impl_->lastError);
415 return;
416 }
417
418 const QString accessToken = impl_->currentSession->accessToken;
419
420 QThread* thread = QThread::create([this, currentPassword, newPassword, accessToken]() {
421 ::muscovite::security::v1::ChangePasswordRequest request;
422 request.set_current_password(currentPassword.toStdString());
423 request.set_new_password(newPassword.toStdString());
424
425 ::google::protobuf::Empty response;
426 grpc::ClientContext context;
427 context.AddMetadata("authorization", "Bearer " + accessToken.toStdString());
428
429 const grpc::Status status = impl_->authStub->ChangePassword(&context, request, &response);
430
431 QMetaObject::invokeMethod(this, [this, status]() {
432 if (!status.ok()) {
433 impl_->lastError = QString::fromStdString(status.error_message());
434 emit passwordChangeFailed(impl_->lastError);
435 emit errorOccurred(impl_->lastError);
436 return;
437 }
438
439 emit passwordChanged();
440 }, Qt::QueuedConnection);
441 });
442
443 connect(thread, &QThread::finished, thread, &QThread::deleteLater);
444 thread->start();
445}
