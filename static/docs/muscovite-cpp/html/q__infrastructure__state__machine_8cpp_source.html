58void QInfrastructureStateMachineBase::recordFailure()
59{
60 ++failure_count_;
61
62 if (configuration().contains(half_open_state_)) {
63 // Any failure in half-open immediately re-opens
64 emit failProbeSignal();
65 return;
66 }
67
68 if (failure_count_ >= failure_threshold_) {
69 onTrip();
70 emit tripSignal();
71 }
72}

83void QInfrastructureStateMachineBase::setupStates()
84{
85 closed_state_ = new QState{this};
86 open_state_ = new QState{this};
87 half_open_state_ = new QState{this};
88
89 // WHY: Connect entered() signals to both public signals and virtual hooks.
90 // This allows both UI binding via signals and subclass-specific behavior via overrides.
91 connect(closed_state_, &QState::entered, this, [this]() {
92 failure_count_ = 0;
93 probe_timer_->stop();
94 emit circuitClosed();
95 emit stateNameChanged();
96 });
97
98 connect(open_state_, &QState::entered, this, [this]() {
99 onOpen();
100 probe_timer_->start();
101 emit circuitOpened();
102 emit stateNameChanged();
103 });
104
105 connect(half_open_state_, &QState::entered, this, [this]() {
106 onHalfOpen();
107 emit circuitHalfOpened();
108 emit stateNameChanged();
109 });
110}

112void QInfrastructureStateMachineBase::setupTransitions()
113{
114 // Closed -> Open: on trip signal (failures >= threshold)
115 closed_state_->addTransition(this, &QInfrastructureStateMachineBase::tripSignal, open_state_);
116
117 // Open -> HalfOpen: on probe timer timeout
118 // WHY: QTimer::timeout triggers probe signal, which transitions to half-open.
119 // We use a signal relay because QState::addTransition wants a QObject* sender.
120 connect(probe_timer_, &QTimer::timeout, this, &QInfrastructureStateMachineBase::probeSignal);
121 open_state_->addTransition(this, &QInfrastructureStateMachineBase::probeSignal, half_open_state_);
122
123 // HalfOpen -> Closed: on recover signal (success in half-open)
124 half_open_state_->addTransition(this, &QInfrastructureStateMachineBase::recoverSignal, closed_state_);
125
126 // HalfOpen -> Open: on fail probe signal (failure in half-open)
127 half_open_state_->addTransition(this, &QInfrastructureStateMachineBase::failProbeSignal, open_state_);
128}
