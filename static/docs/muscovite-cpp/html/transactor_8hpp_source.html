247[[nodiscard]] inline auto has_retry_budget(const RetryPolicy& policy) -> bool {
248 if (policy.deadline == std::chrono::system_clock::time_point::max()) {
249 return true; // No deadline set — unlimited budget
250 }
251 auto remaining = std::chrono::duration_cast<std::chrono::milliseconds>(
252 policy.deadline - std::chrono::system_clock::now());
253 return remaining > policy.min_remaining_for_retry;
254}

258template <typename F>
259 requires TransactorCallable<F, std::invoke_result_t<F, pqxx::transaction<>&>> &&
260 (!std::is_void_v<std::invoke_result_t<F, pqxx::transaction<>&>>)
261auto execute(ConnectionPool& pool, F&& fn, RetryPolicy policy)
262 -> std::expected<std::invoke_result_t<F, pqxx::transaction<>&>, Error<DbError>> {
263 using R = std::invoke_result_t<F, pqxx::transaction<>&>;
264
265 for (std::size_t attempt = 0; attempt <= policy.max_retries; ++attempt) {
266 if (attempt > 0) {
267 // FTR-235: Check retry budget before sleeping
268 if (!detail::has_retry_budget(policy)) {
269 return std::unexpected(Error<DbError>{
270 .code = DbError::DeadlineExceeded,
271 .message = "Insufficient time remaining for retry (budget exhausted)",
272 .detail = {},
273 .sql_state = "57014",
274 .arc = std::nullopt});
275 }
276 auto delay = detail::next_delay(attempt, policy);
277 std::this_thread::sleep_for(delay);
278 }
279
280 auto conn_result = pool.acquire();
281 if (!conn_result) {
282 return std::unexpected(Error<DbError>{.code = DbError::ConnectionFailed,
283 .message = conn_result.error().message,
284 .detail = {},
285 .sql_state = {},
286 .arc = std::nullopt});
287 }
288
289 try {
290 pqxx::transaction<> tx{conn_result->get()};
291 R result = std::forward<F>(fn)(tx);
292 tx.commit();
293 return result;
294 } catch (const pqxx::serialization_failure& e) {
295 if (attempt < policy.max_retries && policy.retry_on_serialization_failure) {
296 continue;
297 }
298 return std::unexpected(detail::map_pqxx_error(e));
299 } catch (const pqxx::deadlock_detected& e) {
300 if (attempt < policy.max_retries && policy.retry_on_deadlock) {
301 continue;
302 }
303 return std::unexpected(detail::map_pqxx_error(e));
304 } catch (const pqxx::failure& e) {
305 // libpqxx 8: use poisons_connection() instead of catching pqxx::broken_connection
306 if (e.poisons_connection() && attempt < policy.max_retries && policy.retry_on_connection_error) {
307 continue;
308 }
309 return std::unexpected(detail::map_pqxx_error(e));
310 } catch (const std::exception& e) {
311 return std::unexpected(detail::map_pqxx_error(e));
312 }
313 }
314
315 return std::unexpected(
316 Error<DbError>{.code = DbError::Unknown, .message = "Max retries exceeded", .detail = {}, .sql_state = {}, .arc = std::nullopt});
317}

338template <typename F>
339 requires TransactorCallable<F, std::invoke_result_t<F, pqxx::transaction<>&>> &&
340 (!std::is_void_v<std::invoke_result_t<F, pqxx::transaction<>&>>)
341auto execute(ConnectionPool& pool,
342 std::chrono::system_clock::time_point deadline,
343 F&& fn,
344 std::chrono::milliseconds max_statement_timeout,
345 std::chrono::milliseconds deadline_buffer,
346 RetryPolicy policy)
347 -> std::expected<std::invoke_result_t<F, pqxx::transaction<>&>, Error<DbError>> {
348 using R = std::invoke_result_t<F, pqxx::transaction<>&>;
349
350 // Check if deadline already expired
351 auto now = std::chrono::system_clock::now();
352 if (now >= deadline) {
353 return std::unexpected(Error<DbError>{
354 .code = DbError::DeadlineExceeded,
355 .message = "gRPC deadline already expired",
356 .detail = {},
357 .sql_state = "57014"});
358 }
359
360 // Calculate statement_timeout: remaining time minus buffer, clamped to max
361 auto remaining = std::chrono::duration_cast<std::chrono::milliseconds>(deadline - now);
362 auto timeout = std::min(remaining - deadline_buffer, max_statement_timeout);
363 if (timeout <= std::chrono::milliseconds{0}) {
364 return std::unexpected(Error<DbError>{
365 .code = DbError::DeadlineExceeded,
366 .message = "Insufficient time remaining for database operation",
367 .detail = {},
368 .sql_state = "57014"});
369 }
370
371 auto timeout_ms = timeout.count();
372
373 // Wrap the user function to inject SET LOCAL statement_timeout
374 auto wrapped = [&fn, timeout_ms](pqxx::transaction<>& tx) -> R {
375 tx.exec(std::format("SET LOCAL statement_timeout = '{}'", timeout_ms)).no_rows();
376 return std::forward<F>(fn)(tx);
377 };
378
379 // FTR-235: Populate retry policy deadline so the retry loop respects time budget
380 policy.deadline = deadline;
381
382 auto result = execute(pool, wrapped, policy);
383 if (!result && result.error().code == DbError::Timeout) {
384 // Remap generic Timeout to DeadlineExceeded when we set statement_timeout
385 return std::unexpected(Error<DbError>{
386 .code = DbError::DeadlineExceeded,
387 .message = "PostgreSQL statement_timeout exceeded (propagated from gRPC deadline)",
388 .detail = result.error().detail,
389 .sql_state = "57014"});
390 }
391 return result;
392}

450template <typename F>
451 requires UnitOfWorkCallable<F, std::invoke_result_t<F, UnitOfWork&>> &&
452 (!std::is_void_v<std::invoke_result_t<F, UnitOfWork&>>)
453auto execute_uow(ConnectionPool& pool,
454 std::string_view schema,
455 F&& fn,
456 RetryPolicy policy)
457 -> std::expected<std::invoke_result_t<F, UnitOfWork&>, Error<DbError>> {
458 using R = std::invoke_result_t<F, UnitOfWork&>;
459
460 for (std::size_t attempt = 0; attempt <= policy.max_retries; ++attempt) {
461 if (attempt > 0) {
462 if (!detail::has_retry_budget(policy)) {
463 return std::unexpected(Error<DbError>{
464 .code = DbError::DeadlineExceeded,
465 .message = "Insufficient time remaining for retry (budget exhausted)",
466 .detail = {},
467 .sql_state = "57014",
468 .arc = std::nullopt});
469 }
470 auto delay = detail::next_delay(attempt, policy);
471 std::this_thread::sleep_for(delay);
472 }
473
474 try {
475 // Fresh UnitOfWork per attempt — events reset on retry
476 UnitOfWork uow{pool, schema};
477 R result = std::forward<F>(fn)(uow);
478 uow.commit_or_throw();
479 return result;
480 } catch (const pqxx::serialization_failure& e) {
481 if (attempt < policy.max_retries && policy.retry_on_serialization_failure) {
482 continue;
483 }
484 return std::unexpected(detail::map_pqxx_error(e));
485 } catch (const pqxx::deadlock_detected& e) {
486 if (attempt < policy.max_retries && policy.retry_on_deadlock) {
487 continue;
488 }
489 return std::unexpected(detail::map_pqxx_error(e));
490 } catch (const pqxx::failure& e) {
491 // libpqxx 8: use poisons_connection() instead of catching pqxx::broken_connection
492 if (e.poisons_connection() && attempt < policy.max_retries && policy.retry_on_connection_error) {
493 continue;
494 }
495 return std::unexpected(detail::map_pqxx_error(e));
496 } catch (const std::exception& e) {
497 return std::unexpected(detail::map_pqxx_error(e));
498 }
499 }
500
501 return std::unexpected(
502 Error<DbError>{.code = DbError::Unknown, .message = "Max retries exceeded", .detail = {}, .sql_state = {}, .arc = std::nullopt});
503}
