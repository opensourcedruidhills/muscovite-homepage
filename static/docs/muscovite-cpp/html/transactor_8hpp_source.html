192[[nodiscard]] inline auto has_retry_budget(const RetryPolicy& policy) -> bool {
193 if (policy.deadline == std::chrono::system_clock::time_point::max()) {
194 return true; // No deadline set — unlimited budget
195 }
196 auto remaining = std::chrono::duration_cast<std::chrono::milliseconds>(
197 policy.deadline - std::chrono::system_clock::now());
198 return remaining > policy.min_remaining_for_retry;
199}

203template <typename F>
204 requires TransactorCallable<F, std::invoke_result_t<F, pqxx::work&>> &&
205 (!std::is_void_v<std::invoke_result_t<F, pqxx::work&>>)
206auto execute(ConnectionPool& pool, F&& fn, RetryPolicy policy)
207 -> std::expected<std::invoke_result_t<F, pqxx::work&>, Error<DbError>> {
208 using R = std::invoke_result_t<F, pqxx::work&>;
209
210 for (std::size_t attempt = 0; attempt <= policy.max_retries; ++attempt) {
211 if (attempt > 0) {
212 // FTR-235: Check retry budget before sleeping
213 if (!detail::has_retry_budget(policy)) {
214 return std::unexpected(Error<DbError>{
215 .code = DbError::DeadlineExceeded,
216 .message = "Insufficient time remaining for retry (budget exhausted)",
217 .detail = {},
218 .sql_state = "57014",
219 .arc = std::nullopt});
220 }
221 auto delay = detail::next_delay(attempt, policy);
222 std::this_thread::sleep_for(delay);
223 }
224
225 auto conn_result = pool.acquire();
226 if (!conn_result) {
227 return std::unexpected(Error<DbError>{.code = DbError::ConnectionFailed,
228 .message = conn_result.error().message,
229 .detail = {},
230 .sql_state = {},
231 .arc = std::nullopt});
232 }
233
234 try {
235 pqxx::work tx{conn_result->get()};
236 R result = std::forward<F>(fn)(tx);
237 tx.commit();
238 return result;
239 } catch (const pqxx::serialization_failure& e) {
240 if (attempt < policy.max_retries && policy.retry_on_serialization_failure) {
241 continue;
242 }
243 return std::unexpected(detail::map_pqxx_error(e));
244 } catch (const pqxx::deadlock_detected& e) {
245 if (attempt < policy.max_retries && policy.retry_on_deadlock) {
246 continue;
247 }
248 return std::unexpected(detail::map_pqxx_error(e));
249 } catch (const pqxx::broken_connection& e) {
250 if (attempt < policy.max_retries && policy.retry_on_connection_error) {
251 continue;
252 }
253 return std::unexpected(detail::map_pqxx_error(e));
254 } catch (const std::exception& e) {
255 return std::unexpected(detail::map_pqxx_error(e));
256 }
257 }
258
259 return std::unexpected(
260 Error<DbError>{.code = DbError::Unknown, .message = "Max retries exceeded", .detail = {}, .sql_state = {}, .arc = std::nullopt});
261}

282template <typename F>
283 requires TransactorCallable<F, std::invoke_result_t<F, pqxx::work&>> &&
284 (!std::is_void_v<std::invoke_result_t<F, pqxx::work&>>)
285auto execute(ConnectionPool& pool,
286 std::chrono::system_clock::time_point deadline,
287 F&& fn,
288 std::chrono::milliseconds max_statement_timeout,
289 std::chrono::milliseconds deadline_buffer,
290 RetryPolicy policy)
291 -> std::expected<std::invoke_result_t<F, pqxx::work&>, Error<DbError>> {
292 using R = std::invoke_result_t<F, pqxx::work&>;
293
294 // Check if deadline already expired
295 auto now = std::chrono::system_clock::now();
296 if (now >= deadline) {
297 return std::unexpected(Error<DbError>{
298 .code = DbError::DeadlineExceeded,
299 .message = "gRPC deadline already expired",
300 .detail = {},
301 .sql_state = "57014"});
302 }
303
304 // Calculate statement_timeout: remaining time minus buffer, clamped to max
305 auto remaining = std::chrono::duration_cast<std::chrono::milliseconds>(deadline - now);
306 auto timeout = std::min(remaining - deadline_buffer, max_statement_timeout);
307 if (timeout <= std::chrono::milliseconds{0}) {
308 return std::unexpected(Error<DbError>{
309 .code = DbError::DeadlineExceeded,
310 .message = "Insufficient time remaining for database operation",
311 .detail = {},
312 .sql_state = "57014"});
313 }
314
315 auto timeout_ms = timeout.count();
316
317 // Wrap the user function to inject SET LOCAL statement_timeout
318 auto wrapped = [&fn, timeout_ms](pqxx::work& tx) -> R {
319 tx.exec0(std::format("SET LOCAL statement_timeout = '{}'", timeout_ms));
320 return std::forward<F>(fn)(tx);
321 };
322
323 // FTR-235: Populate retry policy deadline so the retry loop respects time budget
324 policy.deadline = deadline;
325
326 auto result = execute(pool, wrapped, policy);
327 if (!result && result.error().code == DbError::Timeout) {
328 // Remap generic Timeout to DeadlineExceeded when we set statement_timeout
329 return std::unexpected(Error<DbError>{
330 .code = DbError::DeadlineExceeded,
331 .message = "PostgreSQL statement_timeout exceeded (propagated from gRPC deadline)",
332 .detail = result.error().detail,
333 .sql_state = "57014"});
334 }
335 return result;
336}
