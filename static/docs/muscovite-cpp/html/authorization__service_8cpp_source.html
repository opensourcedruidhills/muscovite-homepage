22// Simple thread-safe cache with TTL
23template<typename K, typename V>
24class TTLCache {
25public:
26 explicit TTLCache(std::chrono::seconds ttl) : ttl_(ttl) {}
27
28 std::optional<V> get(const K& key) {
29 const std::shared_lock lock(mutex_);
30 auto it = cache_.find(key);
31 if (it == cache_.end()) {
32 return std::nullopt;
33 }
34
35 auto now = std::chrono::steady_clock::now();
36 if (now > it->second.expires_at) {
37 return std::nullopt; // Expired
38 }
39
40 return it->second.value;
41 }
42
43 void put(const K& key, const V& value) {
44 const std::unique_lock lock(mutex_);
45 cache_[key] = {value, std::chrono::steady_clock::now() + ttl_};
46 }
47
48 void invalidate(const K& key) {
49 const std::unique_lock lock(mutex_);
50 cache_.erase(key);
51 }
52
53 void clear() {
54 const std::unique_lock lock(mutex_);
55 cache_.clear();
56 }
57
58private:
59 struct CacheEntry {
60 V value;
61 std::chrono::steady_clock::time_point expires_at;
62 };
63
64 std::chrono::seconds ttl_;
65 std::unordered_map<K, CacheEntry> cache_;
66 mutable std::shared_mutex mutex_;
67};
68
69} // anonymous namespace

88 std::vector<std::string> fetch_privileges(const std::string& principal_id) {
89 std::vector<std::string> privileges;
90
91 try {
92 pqxx::transaction<> txn(*db);
93 auto result = txn.exec(
94 "SELECT DISTINCT p.name "
95 "FROM security.privilege p "
96 "JOIN security.groupprivilege gp ON p.id = gp.privilege_id "
97 "JOIN security.groupmembership gm ON gp.group_id = gm.group_id "
98 "JOIN security.principalgroup g ON gm.group_id = g.id "
99 "WHERE gm.principal_id = $1 AND g.is_active = true",
100 pqxx::params{principal_id}
101 );
102
103 for (const auto& row : result) {
104 privileges.push_back(row["name"].as<std::string>());
105 }
106 } catch (const std::exception&) { // NOLINT(bugprone-empty-catch) auth must not throw
107 }
108
109 return privileges;
110 }

112 std::vector<std::string> fetch_groups(const std::string& principal_id) {
113 std::vector<std::string> groups;
114
115 try {
116 pqxx::transaction<> txn(*db);
117 auto result = txn.exec(
118 "SELECT g.name "
119 "FROM security.principalgroup g "
120 "JOIN security.groupmembership m ON g.id = m.group_id "
121 "WHERE m.principal_id = $1 AND g.is_active = true",
122 pqxx::params{principal_id}
123 );
124
125 for (const auto& row : result) {
126 groups.push_back(row["name"].as<std::string>());
127 }
128 } catch (const std::exception&) { // NOLINT(bugprone-empty-catch) auth must not throw

131 return groups;

133
134 bool fetch_is_admin(const std::string& principal_id) {
135 try {
136 pqxx::transaction<> txn(*db);
137 auto result = txn.exec(
138 "SELECT 1 "
139 "FROM security.groupmembership m "
140 "JOIN security.principalgroup g ON m.group_id = g.id "
141 "WHERE m.principal_id = $1 AND g.name = $2 AND g.is_active = true",
142 pqxx::params{principal_id, config.admin_group_name}
143 );
144
145 return !result.empty();
146 } catch (const std::exception&) { // NOLINT(bugprone-empty-catch) auth must not throw
147 return false;
148 }
149 }

162bool AuthorizationService::has_privilege(
163 const std::string& principal_id,
164 const std::string& privilege_name) const {
165
166 // Check cache first
167 if (auto cached = impl_->privilege_cache.get(principal_id)) {
168 const auto& privileges = *cached;
169 return std::ranges::find(privileges, privilege_name)
170 != privileges.end();
171 }
172
173 // Fetch and cache
174 auto privileges = impl_->fetch_privileges(principal_id);
175 impl_->privilege_cache.put(principal_id, privileges);
176
177 return std::ranges::find(privileges, privilege_name)
178 != privileges.end();
179}

213AccessCheckResult AuthorizationService::check_access(
214 const std::string& principal_id,
215 const std::string& resource,
216 const std::string& action) const {
217
218 AccessCheckResult result;
219 result.granted = false;
220 result.principal_id = principal_id;
221 result.resource = resource;
222 result.action = action;
223
224 // Admins have full access
225 if (is_admin(principal_id)) {
226 result.granted = true;
227 result.reason = "User is administrator";
228 return result;
229 }
230
231 // Build privilege name from resource and action
232 // Convention: resource:action (e.g., "minerals:read", "users:write")
233 const std::string privilege_name = resource + ":" + action;
234
235 // Also check for wildcard privileges
236 const std::string wildcard_resource = resource + ":*";
237 const std::string wildcard_all = "*:*";
238
239 auto privileges = get_user_privileges(principal_id);
240
241 for (const auto& priv : privileges) {
242 if (priv == privilege_name || priv == wildcard_resource || priv == wildcard_all) {
243 result.granted = true;
244 result.reason = "Has privilege: " + priv;
245 return result;
246 }
247 }
248
249 result.reason = "Missing required privilege: " + privilege_name;
250 return result;
251}
