31void GrpcCallTask::start()
32{
33 if (!call_fn_) {
34 status_ = ::grpc::Status(::grpc::StatusCode::INTERNAL,
35 "GrpcCallTask: no call function set");
36 emit done(QtTaskTree::DoneResult::Error);
37 return;
38 }
39
40 // Capture call function by value — it owns the stub/request/response captures
41 auto fn = call_fn_;
42
43 // Run the blocking gRPC call on a worker thread
44 auto future = QtConcurrent::run([fn = std::move(fn)]() -> ::grpc::Status {
45 return fn();
46 });
47
48 // Watch for completion on the main thread
49 auto *watcher = new QFutureWatcher<::grpc::Status>(this);
50 QObject::connect(watcher, &QFutureWatcherBase::finished, this,
51 [this, watcher]() {
52 status_ = watcher->result();
53 watcher->deleteLater();
54 emit done(status_.ok() ? QtTaskTree::DoneResult::Success
55 : QtTaskTree::DoneResult::Error);
56 });
57 watcher->setFuture(future);
58}
