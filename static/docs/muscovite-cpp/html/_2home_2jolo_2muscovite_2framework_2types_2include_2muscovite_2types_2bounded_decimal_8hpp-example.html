namespace muscovite::types {
namespace detail {
[[nodiscard]] inline auto parse_decimal(std::string_view sv) -> double {
double result = 0.0;
auto [ptr, ec] = std::from_chars(sv.data(), sv.data() + sv.size(), result); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
if (ec != std::errc{} || ptr != sv.data() + sv.size()) { // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
return std::numeric_limits<double>::quiet_NaN();
}
return result;
}
} // namespace detail
template <StringLiteral Min, StringLiteral Max>
class BoundedDecimal {
public:
using value_type = double;
[[nodiscard]] static consteval auto has_lower_bound() noexcept -> bool {
return !Min.empty();
}
[[nodiscard]] static consteval auto has_upper_bound() noexcept -> bool {
return !Max.empty();
}
[[nodiscard]] static auto is_valid(value_type v) noexcept -> bool {
if constexpr (has_lower_bound() && has_upper_bound()) {
return v >= parse_min() && v <= parse_max();
} else if constexpr (has_lower_bound()) {
return v >= parse_min();
} else if constexpr (has_upper_bound()) {
return v <= parse_max();
} else {
return true;
}
}
[[nodiscard]] static auto try_from(value_type v)
-> std::expected<BoundedDecimal, std::string> {
if (!is_valid(v)) {
return std::unexpected(format_error(v));
}
return BoundedDecimal{v, trusted_tag{}};
}
[[nodiscard]] static auto try_from(std::string_view sv)
-> std::expected<BoundedDecimal, std::string> {
auto v = detail::parse_decimal(sv);
if (std::isnan(v)) {
return std::unexpected(
std::format("Invalid decimal: '{}'", sv));
}
return try_from(v);
}
[[nodiscard]] static constexpr auto from_trusted(value_type v) noexcept
-> BoundedDecimal {
return BoundedDecimal{v, trusted_tag{}};
}
[[nodiscard]] constexpr auto value() const noexcept -> value_type {
return value_;
}
[[nodiscard]] constexpr operator value_type() const noexcept {
return value_;
}
[[nodiscard]] constexpr auto operator<=>(const BoundedDecimal&) const noexcept = default;
[[nodiscard]] constexpr auto operator==(const BoundedDecimal&) const noexcept -> bool = default;
private:
struct trusted_tag {};
constexpr BoundedDecimal(value_type v, trusted_tag) noexcept
: value_(v) {}
[[nodiscard]] static auto parse_min() -> double {
static const double cached = detail::parse_decimal(Min.c_str());
return cached;
}
[[nodiscard]] static auto parse_max() -> double {
static const double cached = detail::parse_decimal(Max.c_str());
return cached;
}
[[nodiscard]] static auto format_error(value_type v) -> std::string {
if constexpr (has_lower_bound() && has_upper_bound()) {
return std::format("Value {} out of range [{}, {}]",
v, Min.c_str(), Max.c_str());
} else if constexpr (has_lower_bound()) {
return std::format("Value {} below minimum {}",
v, Min.c_str());
} else if constexpr (has_upper_bound()) {
return std::format("Value {} above maximum {}",
v, Max.c_str());
} else {
return std::format("Value {} out of range", v);
}
}
value_type value_{};
};
} // namespace muscovite::types
