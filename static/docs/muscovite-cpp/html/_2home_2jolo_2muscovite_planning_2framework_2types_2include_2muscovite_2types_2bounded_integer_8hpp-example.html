Compile-time bounded integer with [Min, Max] range enforcement.

Compile-time bounded integer with [Min, Max] range enforcement

Template Parameters
MinLower bound (inclusive). Use unbounded_min for no lower bound.
MaxUpper bound (inclusive). Use unbounded_max for no upper bound.

Construction validates that the value falls within [Min, Max]. Half-open ranges are supported via sentinel values.

using Percentage = BoundedInteger<0, 100>;
auto pct = Percentage::try_from(42); // OK
auto bad = Percentage::try_from(200); // error
auto db = Percentage::from_trusted(42); // no validation
using PositiveInteger = BoundedInteger<0, unbounded_max>;
auto pos = PositiveInteger::try_from(1); // OK

namespace muscovite::types {
inline constexpr auto unbounded_min = std::numeric_limits<std::int64_t>::min();
inline constexpr auto unbounded_max = std::numeric_limits<std::int64_t>::max();
template <std::int64_t Min, std::int64_t Max>
requires(Min <= Max)
class BoundedInteger {
public:
using value_type = std::int64_t;
static constexpr value_type min_value = Min;
static constexpr value_type max_value = Max;
[[nodiscard]] static consteval auto has_lower_bound() noexcept -> bool {
return Min != unbounded_min;
}
[[nodiscard]] static consteval auto has_upper_bound() noexcept -> bool {
return Max != unbounded_max;
}
[[nodiscard]] static constexpr auto is_valid(value_type v) noexcept -> bool {
if constexpr (has_lower_bound() && has_upper_bound()) {
return v >= Min && v <= Max;
} else if constexpr (has_lower_bound()) {
return v >= Min;
} else if constexpr (has_upper_bound()) {
return v <= Max;
} else {
return true;
}
}
[[nodiscard]] static auto try_from(value_type v)
-> std::expected<BoundedInteger, std::string> {
if (!is_valid(v)) {
return std::unexpected(format_error(v));
}
return BoundedInteger{v, trusted_tag{}};
}
[[nodiscard]] static constexpr auto from_trusted(value_type v) noexcept
-> BoundedInteger {
return BoundedInteger{v, trusted_tag{}};
}
[[nodiscard]] constexpr auto value() const noexcept -> value_type {
return value_;
}
[[nodiscard]] constexpr operator value_type() const noexcept {
return value_;
}
[[nodiscard]] constexpr auto operator<=>(const BoundedInteger&) const noexcept = default;
[[nodiscard]] constexpr auto operator==(const BoundedInteger&) const noexcept -> bool = default;
private:
struct trusted_tag {};
constexpr BoundedInteger(value_type v, trusted_tag) noexcept
: value_(v) {}
[[nodiscard]] static auto format_error(value_type v) -> std::string {
if constexpr (has_lower_bound() && has_upper_bound()) {
return std::format("Value {} out of range [{}, {}]", v, Min, Max);
} else if constexpr (has_lower_bound()) {
return std::format("Value {} below minimum {}", v, Min);
} else if constexpr (has_upper_bound()) {
return std::format("Value {} above maximum {}", v, Max);
} else {
return std::format("Value {} out of range", v);
}
}
value_type value_{};
};
