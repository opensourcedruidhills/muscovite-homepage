215 auto [ptr, ec] = std::from_chars(begin, end, value);

225[[nodiscard]] inline auto string_to_date(std::string_view str) -> std::chrono::year_month_day {
226 if (str.empty()) {
227 throw std::invalid_argument("Date string cannot be empty");
228 }
229
230 if (str.size() < 10 || str[4] != '-' || str[7] != '-') {
231 spdlog::error("Invalid date format (expected YYYY-MM-DD): {}", str);
232 throw std::invalid_argument(std::format("Invalid date format (expected YYYY-MM-DD): {}", str));
233 }
234
235 int year{};
236 int month{};
237 int day{};
238
239 if (!detail::parse_int_field(str.data(), str.data() + 4, year) ||
240 !detail::parse_int_field(str.data() + 5, str.data() + 7, month) ||
241 !detail::parse_int_field(str.data() + 8, str.data() + 10, day)) {
242 spdlog::error("Invalid date values: {}", str);
243 throw std::invalid_argument(std::format("Invalid date values: {}", str));
244 }
245
246 return std::chrono::year{year} /
247 std::chrono::month{static_cast<unsigned>(month)} /
248 std::chrono::day{static_cast<unsigned>(day)};
249}

286[[nodiscard]] inline auto string_to_timestamp(std::string_view str) -> std::chrono::system_clock::time_point {
287 if (str.empty()) {
288 throw std::invalid_argument("Timestamp string cannot be empty");
289 }
290
291 // Minimum: YYYY-MM-DDThh:mm:ss (19 chars)
292 if (str.size() < 19 || str[4] != '-' || str[7] != '-' || str[10] != 'T' ||
293 str[13] != ':' || str[16] != ':') {
294 spdlog::error("Invalid timestamp format (expected ISO 8601): {}", str);
295 throw std::invalid_argument(std::format("Invalid timestamp format (expected ISO 8601): {}", str));
296 }
297
298 int year{};
299 int month{};
300 int day{};
301 int hour{};
302 int minute{};
303 int second{};
304
305 if (!detail::parse_int_field(str.data(), str.data() + 4, year) ||
306 !detail::parse_int_field(str.data() + 5, str.data() + 7, month) ||
307 !detail::parse_int_field(str.data() + 8, str.data() + 10, day) ||
308 !detail::parse_int_field(str.data() + 11, str.data() + 13, hour) ||
309 !detail::parse_int_field(str.data() + 14, str.data() + 16, minute) ||
310 !detail::parse_int_field(str.data() + 17, str.data() + 19, second)) {
311 spdlog::error("Invalid timestamp values: {}", str);
312 throw std::invalid_argument(std::format("Invalid timestamp values: {}", str));
313 }
314
315 std::tm tm{};
316 tm.tm_year = year - 1900;
317 tm.tm_mon = month - 1;
318 tm.tm_mday = day;
319 tm.tm_hour = hour;
320 tm.tm_min = minute;
321 tm.tm_sec = second;
322 tm.tm_isdst = 0;
323
324 auto time_t_val = timegm(&tm); // POSIX — UTC conversion
325 return std::chrono::system_clock::from_time_t(time_t_val);
326}

331[[nodiscard]] inline auto timestamp_to_string(const std::chrono::system_clock::time_point& ts) -> std::string {
332 auto time_t_val = std::chrono::system_clock::to_time_t(ts);
333 std::tm tm_buf{};
334 gmtime_r(&time_t_val, &tm_buf);
335
336 return std::format("{:04d}-{:02d}-{:02d}T{:02d}:{:02d}:{:02d}Z",
337 tm_buf.tm_year + 1900, tm_buf.tm_mon + 1, tm_buf.tm_mday,
338 tm_buf.tm_hour, tm_buf.tm_min, tm_buf.tm_sec);
339}

370[[nodiscard]] inline auto string_to_time(std::string_view str) -> std::chrono::microseconds {
371 if (str.empty()) {
372 throw std::invalid_argument("Time string cannot be empty");
373 }
374
375 // Minimum: HH:MM:SS (8 chars)
376 if (str.size() < 8 || str[2] != ':' || str[5] != ':') {
377 spdlog::error("Invalid time format (expected HH:MM:SS or HH:MM:SS.ffffff): {}", str);
378 throw std::invalid_argument(
379 std::format("Invalid time format (expected HH:MM:SS or HH:MM:SS.ffffff): {}", str));
380 }
381
382 int hours{};
383 int minutes{};
384 int seconds{};
385 int microseconds{};
386
387 if (!detail::parse_int_field(str.data(), str.data() + 2, hours) ||
388 !detail::parse_int_field(str.data() + 3, str.data() + 5, minutes) ||
389 !detail::parse_int_field(str.data() + 6, str.data() + 8, seconds)) {
390 spdlog::error("Invalid time format (expected HH:MM:SS or HH:MM:SS.ffffff): {}", str);
391 throw std::invalid_argument(
392 std::format("Invalid time format (expected HH:MM:SS or HH:MM:SS.ffffff): {}", str));
393 }
394
395 // Parse optional fractional part: .ffffff
396 if (str.size() > 8 && str[8] == '.') {
397 auto frac = str.substr(9);
398 if (!frac.empty()) {
399 std::from_chars(frac.data(), frac.data() + frac.size(), microseconds);
400 }
401 }
402
403 if (hours < 0 || hours > 23 || minutes < 0 || minutes > 59 || seconds < 0 || seconds > 59) {
404 spdlog::error("Invalid time values (out of range): {}", str);
405 throw std::invalid_argument(std::format("Invalid time values (out of range): {}", str));
406 }
407
408 auto total = std::chrono::hours{hours} +
409 std::chrono::minutes{minutes} +
410 std::chrono::seconds{seconds} +
411 std::chrono::microseconds{microseconds};
412
413 return std::chrono::duration_cast<std::chrono::microseconds>(total);
414}

419[[nodiscard]] inline auto time_to_string(const std::chrono::microseconds& time) -> std::string {
420 auto total_us = time.count();
421
422 auto const hours = total_us / 3'600'000'000LL;
423 total_us %= 3'600'000'000LL;
424 auto const minutes = total_us / 60'000'000LL;
425 total_us %= 60'000'000LL;
426 auto const seconds = total_us / 1'000'000LL;
427 auto const microseconds = total_us % 1'000'000LL;
428
429 if (microseconds > 0) {
430 return std::format("{:02}:{:02}:{:02}.{:06}", hours, minutes, seconds, microseconds);
431 }
432 return std::format("{:02}:{:02}:{:02}", hours, minutes, seconds);
433}
