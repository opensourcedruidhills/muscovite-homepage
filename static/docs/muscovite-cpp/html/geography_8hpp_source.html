10// ============================================================================
11// Muscovite PostGIS GEOGRAPHY Types
12// ============================================================================
13// Compile-time SRID-safe spatial types for PostGIS GEOGRAPHY columns.
14//
15// Key Design Decisions (see docs/adr/20260126-*.md):
16// - GEOGRAPHY (spherical) not GEOMETRY (planar) for global coordinates
17// - SRID as compile-time template parameter prevents mixing coordinate systems
18// - Uses Boost.Geometry as backend for geodetic calculations
19// - All distances in meters (metric-only per ADR)
20// - No support for M coordinate (deferred)
21// - Anti-meridian crossing is a known limitation
22// ============================================================================

96template<SridTag Srid>
97class Point {
98public:
99 using srid_tag = Srid;
100 static constexpr int srid = Srid::value;
101
102private:
103 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
104 bg_point point_;
105
106public:
108 Point() : point_(0.0, 0.0) {}
109
114 Point(double lon, double lat) : point_(lon, lat) {
115 validate();
116 }
117
119 [[nodiscard]] double longitude() const noexcept {
120 return boost::geometry::get<0>(point_);
121 }
122
124 [[nodiscard]] double latitude() const noexcept {
125 return boost::geometry::get<1>(point_);
126 }
127
129 [[nodiscard]] double x() const noexcept { return longitude(); }
130
132 [[nodiscard]] double y() const noexcept { return latitude(); }
133
135 [[nodiscard]] const bg_point& native() const noexcept { return point_; }
136 [[nodiscard]] bg_point& native() noexcept { return point_; }
137
139 [[nodiscard]] bool operator==(const Point& other) const noexcept {
140 return longitude() == other.longitude() && latitude() == other.latitude();
141 }
142
144 [[nodiscard]] std::string to_wkt() const {
145 return std::format("POINT({} {})", longitude(), latitude());
146 }
147
149 [[nodiscard]] std::string to_ewkt() const {
150 return std::format("SRID={};POINT({} {})", srid, longitude(), latitude());
151 }
152
156 [[nodiscard]] static std::optional<Point> from_wkt(std::string_view wkt) {
157 // Remove leading/trailing whitespace
158 while (!wkt.empty() && std::isspace(wkt.front())) wkt.remove_prefix(1);
159 while (!wkt.empty() && std::isspace(wkt.back())) wkt.remove_suffix(1);
160
161 // Check for POINT prefix (case insensitive)
162 if (wkt.size() < 5) return std::nullopt;
163 std::string prefix(wkt.substr(0, 5));
164 std::transform(prefix.begin(), prefix.end(), prefix.begin(), ::toupper);
165 if (prefix != "POINT") return std::nullopt;
166 wkt.remove_prefix(5);
167
168 // Skip whitespace and opening paren
169 while (!wkt.empty() && std::isspace(wkt.front())) wkt.remove_prefix(1);
170 if (wkt.empty() || wkt.front() != '(') return std::nullopt;
171 wkt.remove_prefix(1);
172
173 // Skip whitespace after opening paren
174 while (!wkt.empty() && std::isspace(wkt.front())) wkt.remove_prefix(1);
175
176 // Extract longitude and latitude
177 // WHY: std::from_chars is locale-independent and non-allocating (FTR-216, CPP-PATTERNS §13)
178 {
179 // Parse longitude
180 double lon{};
181 auto [ptr1, ec1] = std::from_chars(wkt.data(), wkt.data() + wkt.size(), lon);
182 if (ec1 != std::errc{}) return std::nullopt;
183 wkt.remove_prefix(static_cast<size_t>(ptr1 - wkt.data()));
184
185 // Skip whitespace
186 while (!wkt.empty() && std::isspace(wkt.front())) wkt.remove_prefix(1);
187
188 // Parse latitude
189 double lat{};
190 auto [ptr2, ec2] = std::from_chars(wkt.data(), wkt.data() + wkt.size(), lat);
191 if (ec2 != std::errc{}) return std::nullopt;
192 wkt.remove_prefix(static_cast<size_t>(ptr2 - wkt.data()));
193
194 // Skip whitespace and closing paren
195 while (!wkt.empty() && std::isspace(wkt.front())) wkt.remove_prefix(1);
196 if (wkt.empty() || wkt.front() != ')') return std::nullopt;
197
198 return Point(lon, lat);
199 }
200 }
201
202
203private:
204 void validate() const {
205 if (longitude() < -180.0 || longitude() > 180.0) {
206 throw std::out_of_range(
207 std::format("Longitude {} out of range [-180, 180]", longitude()));
208 }
209 if (latitude() < -90.0 || latitude() > 90.0) {
210 throw std::out_of_range(
211 std::format("Latitude {} out of range [-90, 90]", latitude()));
212 }
213 }
214};

280 // Altitude can be negative (below sea level) or very high (space)

282 }

304template<SridTag Srid>
305[[nodiscard]] inline double distance_meters(const Point<Srid>& a, const Point<Srid>& b) {
306 // Use geographic strategy for accurate geodetic distance
307 using strategy = boost::geometry::strategy::distance::geographic<
308 boost::geometry::strategy::vincenty>;
309 return boost::geometry::distance(a.native(), b.native(), strategy());
310}
311
315template<SridTag Srid>
316[[nodiscard]] inline double distance_meters(const PointZ<Srid>& a, const PointZ<Srid>& b) {
317 return distance_meters(a.to_2d(), b.to_2d());
318}
319
325template<SridTag Srid>
326[[nodiscard]] inline bool within_radius(
327 const Point<Srid>& center,
328 const Point<Srid>& point,
329 double radius_meters) {
330 return distance_meters(center, point) <= radius_meters;
331}

339template<SridTag Srid>
340class BoundingBox {
341public:
342 using srid_tag = Srid;
343 static constexpr int srid = Srid::value;
344
345private:
346 double min_lon_, min_lat_, max_lon_, max_lat_;
347
348public:
354 BoundingBox(double min_lon, double min_lat, double max_lon, double max_lat)
355 : min_lon_(min_lon), min_lat_(min_lat), max_lon_(max_lon), max_lat_(max_lat) {
356 validate();
357 }
358
359 [[nodiscard]] double min_longitude() const noexcept { return min_lon_; }
360 [[nodiscard]] double min_latitude() const noexcept { return min_lat_; }
361 [[nodiscard]] double max_longitude() const noexcept { return max_lon_; }
362 [[nodiscard]] double max_latitude() const noexcept { return max_lat_; }
363
366 [[nodiscard]] bool contains(const Point<Srid>& point) const noexcept {
367 return point.longitude() >= min_lon_ && point.longitude() <= max_lon_
368 && point.latitude() >= min_lat_ && point.latitude() <= max_lat_;
369 }
370
372 [[nodiscard]] std::string to_wkt() const {
373 return std::format(
374 "POLYGON(({0} {1}, {2} {1}, {2} {3}, {0} {3}, {0} {1}))",
375 min_lon_, min_lat_, max_lon_, max_lat_);
376 }
377
379 void expand(const Point<Srid>& point) {
380 min_lon_ = std::min(min_lon_, point.longitude());
381 min_lat_ = std::min(min_lat_, point.latitude());
382 max_lon_ = std::max(max_lon_, point.longitude());
383 max_lat_ = std::max(max_lat_, point.latitude());
384 }
385
386private:
387 void validate() const {
388 if (min_lon_ > max_lon_ || min_lat_ > max_lat_) {
389 throw std::invalid_argument("BoundingBox min must be <= max");
390 }
391 }
392};
393
394using WGS84BoundingBox = BoundingBox<srid_4326>;

402template<SridTag Srid>
403class LineString {
404public:
405 using srid_tag = Srid;
406 static constexpr int srid = Srid::value;
407 using point_type = Point<Srid>;
408
409private:
410 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
411 using bg_linestring = boost::geometry::model::linestring<bg_point>;
412 bg_linestring line_;
413
414public:
416 LineString() = default;
417
419 explicit LineString(const std::vector<point_type>& points) {
420 for (const auto& p : points) {
421 line_.push_back(bg_point(p.longitude(), p.latitude()));
422 }
423 }
424
426 LineString(std::initializer_list<point_type> points) {
427 for (const auto& p : points) {
428 line_.push_back(bg_point(p.longitude(), p.latitude()));
429 }
430 }
431
433 void push_back(const point_type& point) {
434 line_.push_back(bg_point(point.longitude(), point.latitude()));
435 }
436
438 [[nodiscard]] std::size_t size() const noexcept { return line_.size(); }
439
441 [[nodiscard]] bool empty() const noexcept { return line_.empty(); }
442
444 [[nodiscard]] point_type at(std::size_t idx) const {
445 const auto& pt = line_.at(idx);
446 return point_type(boost::geometry::get<0>(pt), boost::geometry::get<1>(pt));
447 }
448
450 [[nodiscard]] const bg_linestring& native() const noexcept { return line_; }
451 [[nodiscard]] bg_linestring& native() noexcept { return line_; }
452
454 [[nodiscard]] double length_meters() const {
455 using strategy = boost::geometry::strategy::distance::geographic<
456 boost::geometry::strategy::vincenty>;
457 return static_cast<double>(boost::geometry::length(line_, strategy()));
458 }
459
461 [[nodiscard]] std::string to_wkt() const {
462 if (empty()) return "LINESTRING EMPTY";
463 std::string wkt = "LINESTRING(";
464 for (std::size_t i = 0; i < line_.size(); ++i) {
465 if (i > 0) wkt += ", ";
466 wkt += std::format("{} {}",
467 boost::geometry::get<0>(line_[i]),
468 boost::geometry::get<1>(line_[i]));
469 }
470 return wkt + ")";
471 }
472
474 [[nodiscard]] std::string to_ewkt() const {
475 return std::format("SRID={};{}", srid, to_wkt());
476 }
477
479 [[nodiscard]] BoundingBox<Srid> bounds() const {
480 if (empty()) throw std::logic_error("Cannot compute bounds of empty linestring");
481 double min_lon = 180, max_lon = -180, min_lat = 90, max_lat = -90;
482 for (const auto& pt : line_) {
483 double lon = boost::geometry::get<0>(pt);
484 double lat = boost::geometry::get<1>(pt);
485 min_lon = std::min(min_lon, lon);
486 max_lon = std::max(max_lon, lon);
487 min_lat = std::min(min_lat, lat);
488 max_lat = std::max(max_lat, lat);
489 }
490 return BoundingBox<Srid>(min_lon, min_lat, max_lon, max_lat);
491 }
492};

502template<SridTag Srid>
503class Polygon {
504public:
505 using srid_tag = Srid;
506 static constexpr int srid = Srid::value;
507 using point_type = Point<Srid>;
508
509private:
510 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
511 using bg_polygon = boost::geometry::model::polygon<bg_point>;
512 bg_polygon poly_;
513
514public:
516 Polygon() = default;
517
520 explicit Polygon(const std::vector<point_type>& outer_ring) {
521 for (const auto& p : outer_ring) {
522 boost::geometry::append(poly_.outer(), bg_point(p.longitude(), p.latitude()));
523 }
524 }
525
527 Polygon(std::initializer_list<point_type> outer_ring) {
528 for (const auto& p : outer_ring) {
529 boost::geometry::append(poly_.outer(), bg_point(p.longitude(), p.latitude()));
530 }
531 }
532
534 void add_hole(const std::vector<point_type>& hole) {
535 poly_.inners().resize(poly_.inners().size() + 1);
536 for (const auto& p : hole) {
537 boost::geometry::append(poly_.inners().back(), bg_point(p.longitude(), p.latitude()));
538 }
539 }
540
542 [[nodiscard]] std::size_t outer_size() const noexcept {
543 return poly_.outer().size();
544 }
545
547 [[nodiscard]] std::size_t num_holes() const noexcept {
548 return poly_.inners().size();
549 }
550
552 [[nodiscard]] bool empty() const noexcept {
553 return poly_.outer().empty();
554 }
555
557 [[nodiscard]] const bg_polygon& native() const noexcept { return poly_; }
558 [[nodiscard]] bg_polygon& native() noexcept { return poly_; }
559
561 [[nodiscard]] double area_sq_meters() const {
562 using strategy = boost::geometry::strategy::area::geographic<>;
563 return static_cast<double>(std::abs(boost::geometry::area(poly_, strategy())));
564 }
565
567 [[nodiscard]] bool contains(const point_type& point) const {
568 bg_point p(point.longitude(), point.latitude());
569 return boost::geometry::within(p, poly_);
570 }
571
573 [[nodiscard]] std::string to_wkt() const {
574 if (empty()) return "POLYGON EMPTY";
575 std::string wkt = "POLYGON((";
576 // Outer ring
577 for (std::size_t i = 0; i < poly_.outer().size(); ++i) {
578 if (i > 0) wkt += ", ";
579 wkt += std::format("{} {}",
580 boost::geometry::get<0>(poly_.outer()[i]),
581 boost::geometry::get<1>(poly_.outer()[i]));
582 }
583 wkt += ")";
584 // Inner rings (holes)
585 for (const auto& hole : poly_.inners()) {
586 wkt += ", (";
587 for (std::size_t i = 0; i < hole.size(); ++i) {
588 if (i > 0) wkt += ", ";
589 wkt += std::format("{} {}",
590 boost::geometry::get<0>(hole[i]),
591 boost::geometry::get<1>(hole[i]));
592 }
593 wkt += ")";
594 }
595 return wkt + ")";
596 }
597
599 [[nodiscard]] std::string to_ewkt() const {
600 return std::format("SRID={};{}", srid, to_wkt());
601 }
602
604 [[nodiscard]] BoundingBox<Srid> bounds() const {
605 if (empty()) throw std::logic_error("Cannot compute bounds of empty polygon");
606 double min_lon = 180, max_lon = -180, min_lat = 90, max_lat = -90;
607 for (const auto& pt : poly_.outer()) {
608 double lon = boost::geometry::get<0>(pt);
609 double lat = boost::geometry::get<1>(pt);
610 min_lon = std::min(min_lon, lon);
611 max_lon = std::max(max_lon, lon);
612 min_lat = std::min(min_lat, lat);
613 max_lat = std::max(max_lat, lat);
614 }
615 return BoundingBox<Srid>(min_lon, min_lat, max_lon, max_lat);
616 }
617};

723private:

725
726public:
727 MultiPolygon() = default;
728 explicit MultiPolygon(const std::vector<polygon_type>& polys) : polys_(polys) {}
729 MultiPolygon(std::initializer_list<polygon_type> polys) : polys_(polys) {}
730
731 void push_back(const polygon_type& p) { polys_.push_back(p); }
732 [[nodiscard]] std::size_t size() const noexcept { return polys_.size(); }
733 [[nodiscard]] bool empty() const noexcept { return polys_.empty(); }
734 [[nodiscard]] const polygon_type& at(std::size_t idx) const { return polys_.at(idx); }
735 [[nodiscard]] const std::vector<polygon_type>& polygons() const noexcept { return polys_; }
736
738 [[nodiscard]] double total_area_sq_meters() const {
739 double total = 0.0;

741 total += p.area_sq_meters();
742 }
743 return total;
744 }

747 [[nodiscard]] bool contains(const Point<Srid>& point) const {
748 for (const auto& p : polys_) {
749 if (p.contains(point)) return true;
750 }
751 return false;

753
754 [[nodiscard]] std::string to_wkt() const {
755 if (empty()) return "MULTIPOLYGON EMPTY";
756 std::string wkt = "MULTIPOLYGON(";
757 for (std::size_t i = 0; i < polys_.size(); ++i) {
758 if (i > 0) wkt += ", ";
759 // Extract inner part of WKT (without "POLYGON")
