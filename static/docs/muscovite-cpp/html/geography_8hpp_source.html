10// ============================================================================
11// Muscovite PostGIS GEOGRAPHY Types
12// ============================================================================
13// Compile-time SRID-safe spatial types for PostGIS GEOGRAPHY columns.
14//
15// Key Design Decisions (see docs/adr/20260126-*.md):
16// - GEOGRAPHY (spherical) not GEOMETRY (planar) for global coordinates
17// - SRID as compile-time template parameter prevents mixing coordinate systems
18// - Uses Boost.Geometry as backend for geodetic calculations
19// - All distances in meters (metric-only per ADR)
20// - No support for M coordinate (deferred)
21// - Anti-meridian crossing is a known limitation
22// ============================================================================

97template<SridTag Srid>
98class Point {
99public:
100 using srid_tag = Srid;
101 static constexpr int srid = Srid::value;
102
103private:
104 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
105 bg_point point_;
106
107public:
109 Point() : point_(0.0, 0.0) {}
110
115 Point(double lon, double lat) : point_(lon, lat) {
116 validate();
117 }
118
120 [[nodiscard]] double longitude() const noexcept {
121 return boost::geometry::get<0>(point_);
122 }
123
125 [[nodiscard]] double latitude() const noexcept {
126 return boost::geometry::get<1>(point_);
127 }
128
130 [[nodiscard]] double x() const noexcept { return longitude(); }
131
133 [[nodiscard]] double y() const noexcept { return latitude(); }
134
136 [[nodiscard]] const bg_point& native() const noexcept { return point_; }
137 [[nodiscard]] bg_point& native() noexcept { return point_; }
138
140 [[nodiscard]] bool operator==(const Point& other) const noexcept {
141 return longitude() == other.longitude() && latitude() == other.latitude();
142 }
143
145 [[nodiscard]] std::string to_wkt() const {
146 return std::format("POINT({} {})", longitude(), latitude());
147 }
148
150 [[nodiscard]] std::string to_ewkt() const {
151 return std::format("SRID={};POINT({} {})", srid, longitude(), latitude());
152 }
153
157 [[nodiscard]] static std::optional<Point> from_wkt(std::string_view wkt) {
158 // Remove leading/trailing whitespace
159 while (!wkt.empty() && (std::isspace(wkt.front()) != 0)) wkt.remove_prefix(1);
160 while (!wkt.empty() && (std::isspace(wkt.back()) != 0)) wkt.remove_suffix(1);
161
162 // Check for POINT prefix (case insensitive)
163 if (wkt.size() < 5) return std::nullopt;
164 std::string prefix(wkt.substr(0, 5));
165 std::ranges::transform(prefix, prefix.begin(), ::toupper);
166 if (prefix != "POINT") return std::nullopt;
167 wkt.remove_prefix(5);
168
169 // Skip whitespace and opening paren
170 while (!wkt.empty() && (std::isspace(wkt.front()) != 0)) wkt.remove_prefix(1);
171 if (wkt.empty() || wkt.front() != '(') return std::nullopt;
172 wkt.remove_prefix(1);
173
174 // Skip whitespace after opening paren
175 while (!wkt.empty() && (std::isspace(wkt.front()) != 0)) wkt.remove_prefix(1);
176
177 // Extract longitude and latitude
178 // WHY: std::from_chars is locale-independent and non-allocating (FTR-216, CPP-PATTERNS §13)
179 {
180 // Parse longitude
181 double lon{};
182 auto [ptr1, ec1] = std::from_chars(wkt.data(), wkt.data() + wkt.size(), lon); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
183 if (ec1 != std::errc{}) return std::nullopt;
184 wkt.remove_prefix(static_cast<size_t>(ptr1 - wkt.data())); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
185
186 // Skip whitespace
187 while (!wkt.empty() && (std::isspace(wkt.front()) != 0)) wkt.remove_prefix(1);
188
189 // Parse latitude
190 double lat{};
191 auto [ptr2, ec2] = std::from_chars(wkt.data(), wkt.data() + wkt.size(), lat); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
192 if (ec2 != std::errc{}) return std::nullopt;
193 wkt.remove_prefix(static_cast<size_t>(ptr2 - wkt.data())); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
194
195 // Skip whitespace and closing paren
196 while (!wkt.empty() && (std::isspace(wkt.front()) != 0)) wkt.remove_prefix(1);
197 if (wkt.empty() || wkt.front() != ')') return std::nullopt;
198
199 return Point(lon, lat);
200 }
201 }
202
203
204private:
205 void validate() const {
206 if (longitude() < -180.0 || longitude() > 180.0) {
207 throw std::out_of_range(
208 std::format("Longitude {} out of range [-180, 180]", longitude()));
209 }
210 if (latitude() < -90.0 || latitude() > 90.0) {
211 throw std::out_of_range(
212 std::format("Latitude {} out of range [-90, 90]", latitude()));
213 }
214 }
215};

305template<SridTag Srid>
306[[nodiscard]] inline double distance_meters(const Point<Srid>& a, const Point<Srid>& b) {
307 // Use geographic strategy for accurate geodetic distance
308 using strategy = boost::geometry::strategy::distance::geographic<
309 boost::geometry::strategy::vincenty>;
310 return boost::geometry::distance(a.native(), b.native(), strategy());
311}
312
316template<SridTag Srid>
317[[nodiscard]] inline double distance_meters(const PointZ<Srid>& a, const PointZ<Srid>& b) {
318 return distance_meters(a.to_2d(), b.to_2d());
319}
320
326template<SridTag Srid>
327[[nodiscard]] inline bool within_radius(
328 const Point<Srid>& center,
329 const Point<Srid>& point,
330 double radius_meters) {
331 return distance_meters(center, point) <= radius_meters;
332}

340template<SridTag Srid>
341class BoundingBox {
342public:
343 using srid_tag = Srid;
344 static constexpr int srid = Srid::value;
345
346private:
347 double min_lon_, min_lat_, max_lon_, max_lat_;
348
349public:
355 BoundingBox(double min_lon, double min_lat, double max_lon, double max_lat)
356 : min_lon_(min_lon), min_lat_(min_lat), max_lon_(max_lon), max_lat_(max_lat) {
357 validate();
358 }
359
360 [[nodiscard]] double min_longitude() const noexcept { return min_lon_; }
361 [[nodiscard]] double min_latitude() const noexcept { return min_lat_; }
362 [[nodiscard]] double max_longitude() const noexcept { return max_lon_; }
363 [[nodiscard]] double max_latitude() const noexcept { return max_lat_; }
364
367 [[nodiscard]] bool contains(const Point<Srid>& point) const noexcept {
368 return point.longitude() >= min_lon_ && point.longitude() <= max_lon_
369 && point.latitude() >= min_lat_ && point.latitude() <= max_lat_;
370 }
371
373 [[nodiscard]] std::string to_wkt() const {
374 return std::format(
375 "POLYGON(({0} {1}, {2} {1}, {2} {3}, {0} {3}, {0} {1}))",
376 min_lon_, min_lat_, max_lon_, max_lat_);
377 }
378
380 void expand(const Point<Srid>& point) {
381 min_lon_ = std::min(min_lon_, point.longitude());
382 min_lat_ = std::min(min_lat_, point.latitude());
383 max_lon_ = std::max(max_lon_, point.longitude());
384 max_lat_ = std::max(max_lat_, point.latitude());
385 }
386
387private:
388 void validate() const {
389 if (min_lon_ > max_lon_ || min_lat_ > max_lat_) {
390 throw std::invalid_argument("BoundingBox min must be <= max");
391 }
392 }
393};

403template<SridTag Srid>
404class LineString {
405public:
406 using srid_tag = Srid;
407 static constexpr int srid = Srid::value;
408 using point_type = Point<Srid>;
409
410private:
411 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
412 using bg_linestring = boost::geometry::model::linestring<bg_point>;
413 bg_linestring line_;
414
415public:
417 LineString() = default;
418
420 explicit LineString(const std::vector<point_type>& points) {
421 for (const auto& p : points) {
422 line_.push_back(bg_point(p.longitude(), p.latitude()));
423 }
424 }
425
427 LineString(std::initializer_list<point_type> points) {
428 for (const auto& p : points) {
429 line_.push_back(bg_point(p.longitude(), p.latitude()));
430 }
431 }
432
434 void push_back(const point_type& point) {
435 line_.push_back(bg_point(point.longitude(), point.latitude()));
436 }
437
439 [[nodiscard]] std::size_t size() const noexcept { return line_.size(); }
440
442 [[nodiscard]] bool empty() const noexcept { return line_.empty(); }
443
445 [[nodiscard]] point_type at(std::size_t idx) const {
446 const auto& pt = line_.at(idx);
447 return point_type(boost::geometry::get<0>(pt), boost::geometry::get<1>(pt));
448 }
449
451 [[nodiscard]] const bg_linestring& native() const noexcept { return line_; }
452 [[nodiscard]] bg_linestring& native() noexcept { return line_; }
453
455 [[nodiscard]] double length_meters() const {
456 using strategy = boost::geometry::strategy::distance::geographic<
457 boost::geometry::strategy::vincenty>;
458 return static_cast<double>(boost::geometry::length(line_, strategy()));
459 }
460
462 [[nodiscard]] std::string to_wkt() const {
463 if (empty()) return "LINESTRING EMPTY";
464 std::string wkt = "LINESTRING(";
465 for (std::size_t i = 0; i < line_.size(); ++i) {
466 if (i > 0) wkt += ", ";
467 wkt += std::format("{} {}",
468 boost::geometry::get<0>(line_[i]),
469 boost::geometry::get<1>(line_[i]));
470 }
471 return wkt + ")";
472 }
473
475 [[nodiscard]] std::string to_ewkt() const {
476 return std::format("SRID={};{}", srid, to_wkt());
477 }
478
480 [[nodiscard]] BoundingBox<Srid> bounds() const {
481 if (empty()) throw std::logic_error("Cannot compute bounds of empty linestring");
482 double min_lon = 180;
483 double max_lon = -180;
484 double min_lat = 90;
485 double max_lat = -90;
486 for (const auto& pt : line_) {
487 const double lon = boost::geometry::get<0>(pt);
488 const double lat = boost::geometry::get<1>(pt);
489 min_lon = std::min(min_lon, lon);
490 max_lon = std::max(max_lon, lon);
491 min_lat = std::min(min_lat, lat);
492 max_lat = std::max(max_lat, lat);
493 }
494 return BoundingBox<Srid>(min_lon, min_lat, max_lon, max_lat);
495 }
496};

506template<SridTag Srid>
507class Polygon {
508public:
509 using srid_tag = Srid;
510 static constexpr int srid = Srid::value;
511 using point_type = Point<Srid>;
512
513private:
514 using bg_point = boost::geometry::model::point<double, 2, geographic_cs>;
515 using bg_polygon = boost::geometry::model::polygon<bg_point>;
516 bg_polygon poly_;
517
518public:
520 Polygon() = default;
521
524 explicit Polygon(const std::vector<point_type>& outer_ring) {
525 for (const auto& p : outer_ring) {
526 boost::geometry::append(poly_.outer(), bg_point(p.longitude(), p.latitude()));
527 }
528 }
529
531 Polygon(std::initializer_list<point_type> outer_ring) {
532 for (const auto& p : outer_ring) {
533 boost::geometry::append(poly_.outer(), bg_point(p.longitude(), p.latitude()));
534 }
535 }
536
538 void add_hole(const std::vector<point_type>& hole) {
539 poly_.inners().resize(poly_.inners().size() + 1);
540 for (const auto& p : hole) {
541 boost::geometry::append(poly_.inners().back(), bg_point(p.longitude(), p.latitude()));
542 }
543 }
544
546 [[nodiscard]] std::size_t outer_size() const noexcept {
547 return poly_.outer().size();
548 }
549
551 [[nodiscard]] std::size_t num_holes() const noexcept {
552 return poly_.inners().size();
553 }
554
556 [[nodiscard]] bool empty() const noexcept {
557 return poly_.outer().empty();
558 }
559
561 [[nodiscard]] const bg_polygon& native() const noexcept { return poly_; }
562 [[nodiscard]] bg_polygon& native() noexcept { return poly_; }
563
565 [[nodiscard]] double area_sq_meters() const {
566 using strategy = boost::geometry::strategy::area::geographic<>;
567 return std::abs(boost::geometry::area(poly_, strategy()));
568 }
569
571 [[nodiscard]] bool contains(const point_type& point) const {
572 const bg_point p(point.longitude(), point.latitude());
573 return boost::geometry::within(p, poly_);
574 }
575
577 [[nodiscard]] std::string to_wkt() const {
578 if (empty()) return "POLYGON EMPTY";
579 std::string wkt = "POLYGON((";
580 // Outer ring
581 for (std::size_t i = 0; i < poly_.outer().size(); ++i) {
582 if (i > 0) wkt += ", ";
583 wkt += std::format("{} {}",
584 boost::geometry::get<0>(poly_.outer()[i]),
585 boost::geometry::get<1>(poly_.outer()[i]));
586 }
587 wkt += ")";
588 // Inner rings (holes)
589 for (const auto& hole : poly_.inners()) {
590 wkt += ", (";
591 for (std::size_t i = 0; i < hole.size(); ++i) {
592 if (i > 0) wkt += ", ";
593 wkt += std::format("{} {}",
594 boost::geometry::get<0>(hole[i]),
595 boost::geometry::get<1>(hole[i]));
596 }
597 wkt += ")";
598 }
599 return wkt + ")";
600 }
601
603 [[nodiscard]] std::string to_ewkt() const {
604 return std::format("SRID={};{}", srid, to_wkt());
605 }
606
608 [[nodiscard]] BoundingBox<Srid> bounds() const {
609 if (empty()) throw std::logic_error("Cannot compute bounds of empty polygon");
610 double min_lon = 180;
611 double max_lon = -180;
612 double min_lat = 90;
613 double max_lat = -90;
614 for (const auto& pt : poly_.outer()) {
615 const double lon = boost::geometry::get<0>(pt);
616 const double lat = boost::geometry::get<1>(pt);
617 min_lon = std::min(min_lon, lon);
618 max_lon = std::max(max_lon, lon);
619 min_lat = std::min(min_lat, lat);
620 max_lat = std::max(max_lat, lat);
621 }
622 return BoundingBox<Srid>(min_lon, min_lat, max_lon, max_lat);
623 }
624};

762 if (empty()) return "MULTIPOLYGON EMPTY";
763 std::string wkt = "MULTIPOLYGON(";
764 for (std::size_t i = 0; i < polys_.size(); ++i) {
765 if (i > 0) wkt += ", ";
766 // Extract inner part of WKT (without "POLYGON")
767 std::string poly_wkt = polys_[i].to_wkt();
768 auto pos = poly_wkt.find('(');
769 wkt += poly_wkt.substr(pos);
770 }
771 return wkt + ")";
772 }

775 return std::format("SRID={};{}", srid, to_wkt());
776 }
