24// Extract bearer token from authorization header
25std::optional<std::string> extract_bearer_token(
26 const std::multimap<::grpc::string_ref, ::grpc::string_ref>& metadata) {
27
28 auto it = metadata.find("authorization");
29 if (it == metadata.end()) {
30 return std::nullopt;
31 }
32
33 const std::string auth_header(it->second.data(), it->second.length());
34
35 // Check for "Bearer " prefix (case-insensitive)
36 const std::string bearer_prefix = "Bearer ";
37 if (auth_header.length() < bearer_prefix.length()) {
38 return std::nullopt;
39 }
40
41 std::string prefix = auth_header.substr(0, bearer_prefix.length());
42 // Case-insensitive comparison for "Bearer "
43 std::ranges::transform(prefix, prefix.begin(), ::tolower);
44 if (prefix != "bearer ") {
45 return std::nullopt;
46 }
47
48 return auth_header.substr(bearer_prefix.length());
49}

85// ServerInterceptor implementation
86//
87// Rejection strategy: store the denial status at POST_RECV_INITIAL_METADATA,
88// cancel the server context so the handler receives a cancelled RPC, then
89// override the outgoing status at PRE_SEND_STATUS with the proper gRPC code
90// (UNAUTHENTICATED / PERMISSION_DENIED).
91// Note: Hijack() is a client-only API and MUST NOT be called from a server
92// interceptor — it triggers a fatal gRPC assertion.
93class AuthServerInterceptor : public ::grpc::experimental::Interceptor {
94public:
95 AuthServerInterceptor(
96 ::grpc::experimental::ServerRpcInfo* info,
97 std::shared_ptr<AuthInterceptor::Impl> impl)
98 : info_(info), impl_(std::move(impl)) {}
99
100 void Intercept(::grpc::experimental::InterceptorBatchMethods* methods) override {
101 if (methods->QueryInterceptionHookPoint(
102 ::grpc::experimental::InterceptionHookPoints::POST_RECV_INITIAL_METADATA)) {
103 const auto& md = info_->server_context()->client_metadata();
104 const auto decision = impl_->owner->evaluate_request(
105 info_->method(), md);
106
107 if (!decision.allowed) {
108 denial_status_ = decision.status;
109 info_->server_context()->TryCancel();
110 } else if (decision.principal_id) {
111 // FTR-504: store caller identity for audit field propagation
112 muscovite::grpc::ActorContext::set(*decision.principal_id);
113 }
114 }
115
116 if (denial_status_.has_value() &&
117 methods->QueryInterceptionHookPoint(
118 ::grpc::experimental::InterceptionHookPoints::PRE_SEND_STATUS)) {
119 methods->ModifySendStatus(*denial_status_);
120 }
121
122 // FTR-504: clear actor identity after response is sent
123 if (methods->QueryInterceptionHookPoint(
124 ::grpc::experimental::InterceptionHookPoints::PRE_SEND_STATUS)) {
125 muscovite::grpc::ActorContext::clear();
126 }
127
128 methods->Proceed();
129 }
130
131private:
132 ::grpc::experimental::ServerRpcInfo* info_;
133 std::shared_ptr<AuthInterceptor::Impl> impl_;
134 std::optional<::grpc::Status> denial_status_;
135};

151AuthDecision AuthInterceptor::evaluate_request(
152 const std::string& method,
153 const std::multimap<::grpc::string_ref, ::grpc::string_ref>& metadata) const {
154 AuthDecision decision{};
155
156 if (impl_->is_public_method(method)) {
157 decision.allowed = true;
158 return decision;
159 }
160
161 if (!impl_->auth_service) {
162 decision.status = ::grpc::Status(
163 ::grpc::StatusCode::UNAUTHENTICATED, "Authentication service unavailable");
164 return decision;
165 }
166
167 auto token_opt = extract_bearer_token(metadata);
168 if (!token_opt) {
169 decision.status = ::grpc::Status(
170 ::grpc::StatusCode::UNAUTHENTICATED, "Missing Bearer token");
171 return decision;
172 }
173
174 auto principal = impl_->auth_service->validate_token(*token_opt);
175 if (!principal) {
176 decision.status = ::grpc::Status(
177 ::grpc::StatusCode::UNAUTHENTICATED, "Invalid or expired token");
178 return decision;
179 }
180
181 const auto required_privs = impl_->get_required_privileges(method);
182 if (!required_privs.empty()) {
183 bool has_privilege = false;
184
185 if (impl_->authz_service) {
186 has_privilege = impl_->authz_service->has_any_privilege(
187 principal->id, required_privs);
188 } else {
189 for (const auto& priv : required_privs) {
190 if (std::find(principal->privileges.begin(),
191 principal->privileges.end(),
192 priv) != principal->privileges.end()) {
193 has_privilege = true;
194 break;
195 }
196 }
197 }
198
199 if (!has_privilege) {
200 decision.status = ::grpc::Status(
201 ::grpc::StatusCode::PERMISSION_DENIED, "Missing required privilege");
202 return decision;
203 }
204 }
205
206 decision.allowed = true;
207 decision.principal_id = principal->id; // FTR-504: expose for ActorContext
208 return decision;
209}
