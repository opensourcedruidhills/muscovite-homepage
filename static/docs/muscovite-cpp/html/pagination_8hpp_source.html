91template <typename T>
92[[nodiscard]] constexpr auto make_page(std::vector<T> content, const PageRequest& request, std::size_t total_elements)
93 -> Page<T> {
94 const auto effective_size = request.is_unpaged() ? total_elements : request.size;
95 const auto total_pages = effective_size > 0 ? (total_elements + effective_size - 1) / effective_size : 0;
96
97 return Page<T>{.content = std::move(content),
98 .page_number = request.page,
99 .page_size = effective_size,
100 .total_elements = total_elements,
101 .total_pages = total_pages};
102}
