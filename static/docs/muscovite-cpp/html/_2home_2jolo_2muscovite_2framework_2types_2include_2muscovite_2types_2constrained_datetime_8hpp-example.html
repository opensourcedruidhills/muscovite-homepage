#pragma once
#include <muscovite/types/temporal_policies.hpp>
#include <chrono>
#include <compare>
#include <expected>
#include <format>
#include <string>
namespace muscovite::types {
template <typename Policy = Unconstrained>
class ConstrainedDateTime {
public:
using clock_type = std::chrono::system_clock;
using value_type = clock_type::time_point;
using duration_type = clock_type::duration;
[[nodiscard]] static auto is_valid(const value_type& tp) -> bool {
return check_policy(tp);
}
[[nodiscard]] static auto try_from(const value_type& tp)
-> std::expected<ConstrainedDateTime, std::string> {
if (!check_policy(tp)) {
return std::unexpected(format_error(tp));
}
return ConstrainedDateTime{tp, trusted_tag{}};
}
[[nodiscard]] static constexpr auto from_trusted(const value_type& tp) noexcept
-> ConstrainedDateTime {
return ConstrainedDateTime{tp, trusted_tag{}};
}
[[nodiscard]] constexpr auto value() const noexcept -> const value_type& {
return value_;
}
[[nodiscard]] constexpr auto time_since_epoch() const noexcept -> duration_type {
return value_.time_since_epoch();
}
[[nodiscard]] constexpr auto operator<=>(const ConstrainedDateTime&) const noexcept = default;
[[nodiscard]] constexpr auto operator==(const ConstrainedDateTime&) const noexcept -> bool = default;
private:
struct trusted_tag {};
constexpr ConstrainedDateTime(const value_type& v, trusted_tag) noexcept
: value_(v) {}
[[nodiscard]] static auto check_policy(const value_type& tp) -> bool {
if constexpr (std::is_same_v<Policy, Unconstrained>) {
return true;
} else {
auto now = clock_type::now();
if constexpr (std::is_same_v<Policy, PastOrPresent>) {
return tp <= now;
} else if constexpr (std::is_same_v<Policy, PastOnly>) {
return tp < now;
} else if constexpr (std::is_same_v<Policy, Future>) {
return tp > now;
} else {
static_assert(std::is_same_v<Policy, Unconstrained>,
"Unknown temporal policy");
return true;
}
}
}
[[nodiscard]] static auto format_error(const value_type& /*tp*/) -> std::string {
return std::format("Timestamp violates {} policy", Policy::name);
}
value_type value_;
};
} // namespace muscovite::types
