#pragma once
#include <compare>
#include <cstddef>
#include <expected>
#include <format>
#include <functional>
#include <limits>
#include <string>
#include <string_view>
namespace muscovite::types {
inline constexpr std::size_t unbounded_length = std::numeric_limits<std::size_t>::max();
template <std::size_t MinLen, std::size_t MaxLen>
requires(MinLen <= MaxLen)
class BoundedString {
public:
using value_type = std::string;
static constexpr std::size_t min_length = MinLen;
static constexpr std::size_t max_length = MaxLen;
[[nodiscard]] static consteval auto has_min_length() noexcept -> bool {
return MinLen > 0;
}
[[nodiscard]] static consteval auto has_max_length() noexcept -> bool {
return MaxLen != unbounded_length;
}
[[nodiscard]] static auto is_valid(std::string_view sv) noexcept -> bool {
auto len = sv.size();
if constexpr (has_min_length() && has_max_length()) {
return len >= MinLen && len <= MaxLen;
} else if constexpr (has_min_length()) {
return len >= MinLen;
} else if constexpr (has_max_length()) {
return len <= MaxLen;
} else {
return true;
}
}
[[nodiscard]] static auto try_from(std::string_view sv)
-> std::expected<BoundedString, std::string> {
if (!is_valid(sv)) {
return std::unexpected(format_error(sv));
}
return BoundedString{std::string{sv}, trusted_tag{}};
}
[[nodiscard]] static auto from_trusted(std::string value) noexcept
-> BoundedString {
return BoundedString{std::move(value), trusted_tag{}};
}
[[nodiscard]] auto value() const noexcept -> const std::string& {
return value_;
}
[[nodiscard]] auto size() const noexcept -> std::size_t {
return value_.size();
}
[[nodiscard]] auto empty() const noexcept -> bool {
return value_.empty();
}
[[nodiscard]] operator std::string_view() const noexcept {
return value_;
}
[[nodiscard]] auto operator<=>(const BoundedString&) const noexcept = default;
[[nodiscard]] auto operator==(const BoundedString&) const noexcept -> bool = default;
private:
struct trusted_tag {};
BoundedString(std::string v, trusted_tag) noexcept
: value_(std::move(v)) {}
[[nodiscard]] static auto format_error(std::string_view sv) -> std::string {
if constexpr (has_min_length() && has_max_length()) {
return std::format("String length {} out of range [{}, {}]",
sv.size(), MinLen, MaxLen);
} else if constexpr (has_min_length()) {
return std::format("String length {} below minimum {}",
sv.size(), MinLen);
} else if constexpr (has_max_length()) {
return std::format("String length {} above maximum {}",
sv.size(), MaxLen);
} else {
return std::format("String length {} out of range", sv.size());
}
}
std::string value_;
};

} // namespace muscovite::types
