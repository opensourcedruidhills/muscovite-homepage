29void MainWindow::createUI() {
30 // Central widget
31 auto* centralWidget = new QWidget(this);
32 auto* mainLayout = new QVBoxLayout(centralWidget);
33
34 // User info bar
35 userInfoLabel = new QLabel("Not authenticated");
36 userInfoLabel->setStyleSheet("padding: 10px; background-color: #f0f0f0; border-bottom: 1px solid #ccc;");
37 mainLayout->addWidget(userInfoLabel);
38
39 // Tabs for different services
40 tabWidget = new QTabWidget();
41 mainLayout->addWidget(tabWidget);
42
43 setCentralWidget(centralWidget);
44
45 // Menu bar
46 auto* menuBar = new QMenuBar(this);
47 setMenuBar(menuBar);
48
49 userMenu = menuBar->addMenu("User");
50 logoutAction = userMenu->addAction("Logout");
51 connect(logoutAction, &QAction::triggered, this, &MainWindow::onLogout);
52
53 // Status bar
54 statusBar_ = new QLabel("Ready");
55 sessionStatusLabel = new QLabel();
56 this->statusBar()->addWidget(statusBar_);
57 this->statusBar()->addPermanentWidget(sessionStatusLabel);
58}

69void MainWindow::setupConnections() {
70 if (authMgr == nullptr) return;
71
72 connect(authMgr, &AuthManager::authStateChanged,
73 this, &MainWindow::onAuthStateChanged);
74 connect(authMgr, &AuthManager::loggedOut,
75 this, [this]() {
76 statusBar_->setText("Logged out");
77 close();
78 });
79 connect(authMgr, &AuthManager::tokenRefreshed,
80 this, &MainWindow::onTokenRefreshed);
81 connect(authMgr, &AuthManager::sessionExpiring,
82 this, &MainWindow::onSessionExpiring);
83 connect(authMgr, &AuthManager::sessionExpired,
84 this, &MainWindow::onSessionExpired);
85 connect(authMgr, &AuthManager::errorOccurred,
86 this, &MainWindow::onErrorOccurred);
87}

89void MainWindow::updateUserInfo() {
90 if (authMgr == nullptr || !authMgr->isAuthenticated()) {
91 userInfoLabel->setText("Not authenticated");
92 return;
93 }
94
95 const QString displayName = authMgr->displayName();
96 const QStringList groups = authMgr->groups();
97 const QStringList privileges = authMgr->privileges();
98
99 const QString infoText = QString("User: %1 | Groups: %2 | Privileges: %3")
100 .arg(displayName,
101 groups.join(", ").isEmpty() ? "None" : groups.join(", "),
102 privileges.join(", ").isEmpty() ? "None" : privileges.join(", "));
103
104 userInfoLabel->setText(infoText);
105}
