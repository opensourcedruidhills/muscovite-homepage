24QVariant EntityListModel::data(const QModelIndex& index, int role) const {
25 if (!index.isValid()) {
26 return {};
27 }
28 const int row = index.row();
29 if (row < 0 || row >= static_cast<int>(items_.size())) {
30 return {};
31 }
32 const auto& item = items_.at(row);
33
34 // Qt::DisplayRole → "display" key (or first key if no "display")
35 if (role == Qt::DisplayRole) {
36 if (displayRoleKeyIndex_ >= 0 && displayRoleKeyIndex_ < roleKeys_.size()) {
37 return item.value(roleKeys_.at(displayRoleKeyIndex_));
38 }
39 return {};
40 }
41
42 // Custom roles: Qt::UserRole + 1..N mapped to roleKeys_
43 const int keyIndex = role - (Qt::UserRole + 1);
44 if (keyIndex >= 0 && keyIndex < roleKeys_.size()) {
45 return item.value(roleKeys_.at(keyIndex));
46 }
47
48 return {};
49}

121void EntityListModel::deriveRolesFromKeys(const QStringList& keys) {
122 roles_.clear();
123 roleKeys_ = keys;
124 displayRoleKeyIndex_ = -1;
125
126 int role = Qt::UserRole + 1;
127 for (int i = 0; i < keys.size(); ++i) {
128 const auto& key = keys.at(i);
129 roles_[role] = key.toUtf8();
130 if (key == QLatin1String("display")) {
131 displayRoleKeyIndex_ = i;
132 }
133 ++role;
134 }
135
136 // Also map Qt::DisplayRole to "display" key if present
137 if (displayRoleKeyIndex_ >= 0) {
138 roles_[Qt::DisplayRole] = "display";
139 } else if (!keys.isEmpty()) {
140 // Fallback: DisplayRole maps to first key
141 displayRoleKeyIndex_ = 0;
142 roles_[Qt::DisplayRole] = keys.first().toUtf8();
143 }
144}
