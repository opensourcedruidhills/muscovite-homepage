52auto NatsPublisher::connect() -> bool {
53 if (m_connected.load()) {
54 return true;
55 }
56
57 cleanup();
58
59 // Initialize options (FTR-208: RAII)
60 auto [opt_status, opts_ptr] = dba::make_nats_options();
61 if (opt_status != NATS_OK) {
62 m_last_error = std::string{"Failed to create NATS options: "} + natsStatus_GetText(opt_status);
63 return false;
64 }
65 m_opts = std::move(opts_ptr);
66
67 auto* opts = m_opts.get();
68
69 natsOptions_SetURL(opts, m_config.url.c_str());
70 if (!m_config.connection_name.empty()) {
71 natsOptions_SetName(opts, m_config.connection_name.c_str());
72 }
73 natsOptions_SetMaxReconnect(opts, m_config.max_reconnect);
74 natsOptions_SetReconnectWait(opts, static_cast<int64_t>(m_config.reconnect_wait.count()));
75 natsOptions_SetTimeout(opts, static_cast<int64_t>(m_config.connect_timeout.count()));
76
77 // Connect (FTR-208: factory returns RAII connection)
78 auto [conn_status, conn] = dba::make_nats_connection(opts);
79 if (conn_status != NATS_OK) {
80 m_last_error = std::string{"Failed to connect to NATS: "} + natsStatus_GetText(conn_status);
81 spdlog::error("{}", m_last_error);
82 cleanup();
83 return false;
84 }
85 m_conn = std::move(conn);
86
87 // JetStream context
88 jsOptions jsOpts;
89 jsOptions_Init(&jsOpts);
90
91 auto [js_status, js] = dba::make_js_context(m_conn.get(), &jsOpts);
92 if (js_status != NATS_OK) {
93 m_last_error = std::string{"Failed to create JetStream context: "} + natsStatus_GetText(js_status);
94 spdlog::error("{}", m_last_error);
95 cleanup();
96 return false;
97 }
98 m_js = std::move(js);
99
100 m_connected.store(true);
101 spdlog::info("NatsPublisher connected to {} (JetStream enabled)", m_config.url);
102 return true;
103}

116auto NatsPublisher::publish(const MessageEnvelope& envelope) -> PublishResult {
117 auto ack = publish_acked(envelope);
118 if (ack) {
119 return ack->duplicate ? PublishResult::Duplicate : PublishResult::Success;
120 }
121 if (!is_connected()) {
122 return PublishResult::NotConnected;
123 }
124 if (m_last_error.find("timeout") != std::string::npos) {
125 return PublishResult::Timeout;
126 }
127 return PublishResult::Error;
128}

130auto NatsPublisher::publish_acked(const MessageEnvelope& envelope)
131 -> std::optional<PublishAck> {
132 if (!m_connected.load() || !m_js) {
133 m_last_error = "Not connected to NATS";
134 return std::nullopt;
135 }
136
137 jsPubOptions pub_opts;
138 jsPubOptions_Init(&pub_opts);
139
140 // Use dedup_id as JetStream MsgId for deduplication
141 if (envelope.dedup_id.has_value()) {
142 pub_opts.MsgId = envelope.dedup_id->c_str();
143 }
144 pub_opts.MaxWait = static_cast<int64_t>(m_config.ack_timeout.count());
145
146 // Publish payload bytes
147 jsPubAck* ack = nullptr;
148 jsErrCode jerr{};
149
150 const natsStatus status = js_Publish(
151 &ack,
152 m_js.get(),
153 envelope.subject.c_str(),
154 reinterpret_cast<const void*>(envelope.payload.data()),
155 static_cast<int>(envelope.payload.size()),
156 &pub_opts,
157 &jerr);
158
159 if (status != NATS_OK) {
160 m_last_error = std::string{"JetStream publish failed: "} +
161 natsStatus_GetText(status);
162 if (static_cast<int>(jerr) != 0) {
163 m_last_error += " (JS error: " + std::to_string(static_cast<int>(jerr)) + ")";
164 }
165 spdlog::warn("{}", m_last_error);
166 return std::nullopt;
167 }
168
169 PublishAck result{
170 .stream = ack->Stream != nullptr ? ack->Stream : "",
171 .sequence = ack->Sequence,
172 .duplicate = ack->Duplicate};
173
174 ++m_messages_published;
175 jsPubAck_Destroy(ack);
176 return result;
177}

179auto NatsPublisher::ensure_infrastructure(
180 const std::string& stream_name,
181 const std::string& subject_prefix) -> bool {
182 if (!m_connected.load() || !m_js) {
183 m_last_error = "Not connected to NATS";
184 return false;
185 }
186
187 auto* js = m_js.get();
188
189 // Check if stream already exists
190 jsStreamInfo* si = nullptr;
191 jsErrCode jerr{};
192 natsStatus status = js_GetStreamInfo(&si, js, stream_name.c_str(), nullptr, &jerr);
193
194 if (status == NATS_OK && si != nullptr) {
195 spdlog::debug("JetStream stream '{}' already exists", stream_name);
196 jsStreamInfo_Destroy(si);
197 return true;
198 }
199
200 spdlog::info("Creating JetStream stream '{}' for {}.>", stream_name, subject_prefix);
201
202 jsStreamConfig cfg;
203 jsStreamConfig_Init(&cfg);
204 cfg.Name = stream_name.c_str();
205
206 const std::string subject_pattern = subject_prefix + ".>";
207 const char* subjects[] = {subject_pattern.c_str()}; // NOLINT(cppcoreguidelines-avoid-c-arrays,modernize-avoid-c-arrays,misc-const-correctness)
208 cfg.Subjects = subjects;
209 cfg.SubjectsLen = 1;
210
211 cfg.Retention = js_LimitsPolicy;
212 cfg.Storage = js_FileStorage;
213 cfg.MaxAge = static_cast<int64_t>(7 * 24 * 60 * 60) * 1000000000LL;
214 cfg.Replicas = 1;
215
216 const std::string description = "Muscovite events for " + stream_name;
217 cfg.Description = description.c_str();
218
219 jsStreamInfo* created = nullptr;
220 status = js_AddStream(&created, js, &cfg, nullptr, &jerr);
221
222 if (status != NATS_OK) {
223 m_last_error = std::string{"Failed to create stream '"} + stream_name + "': " +
224 natsStatus_GetText(status);
225 spdlog::error("{}", m_last_error);
226 return false;
227 }
228
229 spdlog::info("JetStream stream '{}' created", stream_name);
230 jsStreamInfo_Destroy(created);
231 return true;
232}
