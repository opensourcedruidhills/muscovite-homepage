template<typename Derived>
class muscovite::grpc::ServiceBase< Derived >

CRTP base class for generated gRPC service implementations

Usage:

class MyServiceImpl : public ServiceBase<MyServiceImpl>,
public myproto::MyService::Service {
public:
using ServiceBase::ServiceBase; // Inherit constructor
::grpc::Status GetById(
::grpc::ServerContext* context,
const myproto::IdRequest* request,
myproto::Entity* response) override {
return execute_read([&](pqxx::transaction<>& tx) {
// Database logic here
});
}
};
