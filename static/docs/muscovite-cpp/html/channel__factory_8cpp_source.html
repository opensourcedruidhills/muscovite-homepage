21 const ChannelConfig& config,
22 const std::shared_ptr<muscovite::grpc::oauth2::OAuth2Interceptor>& interceptor) {
23 std::shared_ptr<::grpc::ChannelCredentials> creds;
24
25 if (config.useTls) {
26 ::grpc::SslCredentialsOptions opts;
27
28 if (!config.tls.caCertPath.isEmpty()) {
29 QFile caFile(config.tls.caCertPath);
30 if (caFile.open(QIODevice::ReadOnly)) {
31 opts.pem_root_certs = caFile.readAll().toStdString();
32 }
33 }
34
35 if (!config.tls.clientCertPath.isEmpty() && !config.tls.clientKeyPath.isEmpty()) {
36 QFile certFile(config.tls.clientCertPath);
37 QFile keyFile(config.tls.clientKeyPath);
38 if (certFile.open(QIODevice::ReadOnly) && keyFile.open(QIODevice::ReadOnly)) {
39 opts.pem_cert_chain = certFile.readAll().toStdString();
40 opts.pem_private_key = keyFile.readAll().toStdString();
41 }
42 }
43
44 auto sslCreds = ::grpc::SslCredentials(opts);
45 // TODO: Wire OAuth2 call credentials when plugin lifetime is resolved
46 creds = sslCreds;
47 } else {
48 auto insecure = ::grpc::InsecureChannelCredentials();
49 creds = insecure;
50 }
51
52 return ::grpc::CreateChannel(config.target.toStdString(), creds);
53}
