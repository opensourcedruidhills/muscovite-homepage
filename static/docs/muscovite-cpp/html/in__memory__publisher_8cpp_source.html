32auto InMemoryPublisher::publish(const MessageEnvelope& envelope) -> PublishResult {
33 const std::scoped_lock lock(m_mutex);
34
35 if (!m_connected || m_force_disconnect) {
36 m_last_error = "not connected";
37 return PublishResult::NotConnected;
38 }
39
40 if (m_inject_count > 0) {
41 --m_inject_count;
42 m_last_error = "injected failure";
43 return m_inject_result;
44 }
45
46 // Deduplication
47 if (envelope.dedup_id.has_value()) {
48 if (m_seen_dedup_ids.contains(*envelope.dedup_id)) {
49 return PublishResult::Duplicate;
50 }
51 m_seen_dedup_ids.insert(*envelope.dedup_id);
52 }
53
54 m_published.push_back(envelope);
55 ++m_sequence;
56 return PublishResult::Success;
57}

59auto InMemoryPublisher::publish_acked(const MessageEnvelope& envelope)
60 -> std::optional<PublishAck> {
61 auto result = publish(envelope);
62 if (result == PublishResult::Success || result == PublishResult::Duplicate) {
63 const std::scoped_lock lock(m_mutex);
64 return PublishAck{
65 .stream = "in-memory",
66 .sequence = m_sequence,
67 .duplicate = (result == PublishResult::Duplicate)};
68 }
69 return std::nullopt;
70}
