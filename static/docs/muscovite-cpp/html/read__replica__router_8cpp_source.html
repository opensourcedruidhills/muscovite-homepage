43auto ReadReplicaRouter::create(
44 std::shared_ptr<ConnectionPool> primary,
45 ReadReplicaConfig config
46) -> std::expected<ReadReplicaRouter, Error<PoolError>> {
47
48 if (!config.enabled || config.replica_connection_string.empty()) {
49 // No replica configured, just use primary
50 ReadReplicaRouter router(std::move(primary));
51 return router;
52 }
53
54 // Create replica pool
55 const PoolConfig replica_pool_config{
56 .connection_string = config.replica_connection_string,
57 .min_connections = config.min_connections,
58 .max_connections = config.max_connections,
59 .health_check_interval = config.health_check_interval,
60 .validate_on_borrow = true
61 };
62
63 auto replica_result = ConnectionPool::create(replica_pool_config);
64 if (!replica_result) {
65 if (config.fallback_to_primary) {
66 // Replica failed but fallback is allowed
67 ReadReplicaRouter router(std::move(primary));
68 return router;
69 }
70 return std::unexpected(Error<PoolError>{
71 .code = PoolError::ConnectionFailed,
72 .message = "Failed to connect to read replica: " + replica_result.error().message,
73 .detail = {},
74 .sql_state = {}
75 });
76 }
77
78 ReadReplicaRouter router(std::move(primary));
79 router.m_impl->replica = std::make_shared<ConnectionPool>(std::move(*replica_result));
80 router.m_impl->config = std::move(config);
81 router.m_impl->replica_healthy = true;
82
83 return router;
84}

95auto ReadReplicaRouter::acquire_read() -> std::expected<PooledConnection, Error<PoolError>> {
96 // Try replica first if available and healthy
97 if (m_impl->replica && m_impl->replica_healthy) {
98 auto result = m_impl->replica->acquire();
99 if (result) {
100 return result;
101 }
102 // Replica acquisition failed, mark unhealthy
103 m_impl->replica_healthy = false;
104 }
105
106 // Fallback to primary if configured
107 if (m_impl->config.fallback_to_primary || !m_impl->replica) {
108 return m_impl->primary->acquire();
109 }
110
111 return std::unexpected(Error<PoolError>{
112 .code = PoolError::ConnectionFailed,
113 .message = "Read replica unavailable and fallback disabled",
114 .detail = {},
115 .sql_state = {}
116 });
117}

135auto ReadReplicaRouter::replica_lag_ms() const -> std::optional<std::chrono::milliseconds> {
136 if (!m_impl->replica || !m_impl->replica_healthy) {
137 return std::nullopt;
138 }
139
140 // Query replica lag from pg_stat_replication or recovery info
141 try {
142 auto conn_result = m_impl->replica->try_acquire();
143 if (!conn_result) {
144 return std::nullopt;
145 }
146
147 pqxx::transaction<> tx{conn_result->get()};
148
149 // This query gets the replay lag on a streaming replica
150 auto result = tx.exec(
151 "SELECT CASE WHEN pg_is_in_recovery() THEN "
152 " EXTRACT(EPOCH FROM (NOW() - pg_last_xact_replay_timestamp())) * 1000 "
153 "ELSE NULL END AS lag_ms"
154 );
155
156 if (result.empty() || result[0][0].is_null()) {
157 return std::nullopt;
158 }
159
160 auto lag = result[0][0].as<double>();
161 return std::chrono::milliseconds(static_cast<long long>(lag));
162
163 } catch (const std::exception&) {
164 return std::nullopt;
165 }
166}

168void ReadReplicaRouter::check_replica_health() {
169 if (!m_impl->replica) {
170 return;
171 }
172
173 const std::scoped_lock lock(m_impl->health_mutex);
174 m_impl->last_health_check = std::chrono::steady_clock::now();
175
176 try {
177 auto conn = m_impl->replica->try_acquire();
178 if (!conn) {
179 m_impl->replica_healthy = false;
180 return;
181 }
182
183 pqxx::transaction<> tx{conn->get()};
184 tx.exec("SELECT 1");
185 tx.commit();
186
187 m_impl->replica_healthy = true;
188 } catch (const std::exception&) {
189 m_impl->replica_healthy = false;
190 }
191}
