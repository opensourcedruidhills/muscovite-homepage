104 template <typename Request, typename Response, typename Callable>
105 QFuture<GrpcResult<Response>> callUnary(
106 const Request& request,
107 Callable&& callable,
108 std::chrono::milliseconds deadline = std::chrono::seconds{10}) {
109 QPromise<GrpcResult<Response>> promise;
110 auto future = promise.future();
111 promise.start();
112
113 auto channel = channel_;
114 auto* authMgr = authMgr_;
115
116 std::ignore = QtConcurrent::run([promise = std::move(promise), request, deadline,
117 callable = std::forward<Callable>(callable),
118 channel, authMgr]() mutable {
119 GrpcResult<Response> result;
120 if (!channel) {
121 result.status = ::grpc::Status(::grpc::StatusCode::UNAVAILABLE, "channel not set");
122 promise.addResult(std::move(result));
123 promise.finish();
124 return;
125 }
126
127 ::grpc::ClientContext ctx;
128 ctx.set_deadline(std::chrono::system_clock::now() + deadline);
129
130 // Add authentication header if available
131 if (authMgr && authMgr->isAuthenticated()) {
132 QString const token = authMgr->accessToken();
133 if (!token.isEmpty()) {
134 std::string const authHeader = "Bearer " + token.toStdString();
135 ctx.AddMetadata("authorization", authHeader);
136 }
137 }
138
139 result.status = callable(ctx, request, result.response);
140
141 // Extract validation error details from trailing metadata
142 // Backend sends: x-validation-code, x-validation-field, x-validation-message
143 if (!result.ok()) {
144 const auto& metadata = ctx.GetServerTrailingMetadata();
145
146 auto codeIt = metadata.find("x-validation-code");
147 auto fieldIt = metadata.find("x-validation-field");
148 auto msgIt = metadata.find("x-validation-message");
149
150 if (codeIt != metadata.end() || msgIt != metadata.end()) {
151 ValidationError error;
152 if (codeIt != metadata.end()) {
153 error.code = std::string(codeIt->second.data(), codeIt->second.size());
154 }
155 if (fieldIt != metadata.end()) {
156 error.field = std::string(fieldIt->second.data(), fieldIt->second.size());
157 }
158 if (msgIt != metadata.end()) {
159 error.message = std::string(msgIt->second.data(), msgIt->second.size());
160 }
161 result.validationError = std::move(error);
162 }
163 }
164
165 promise.addResult(std::move(result));
166 promise.finish();
167 });
168
169 return future;
170 }
