18 ::grpc::StatusCode code = ::grpc::StatusCode::UNKNOWN;
19
20 switch (error.code) {

22 code = ::grpc::StatusCode::UNAVAILABLE;
23 break;
24 case dba::DbError::QueryFailed:
25 code = ::grpc::StatusCode::INTERNAL;

78auto make_status(
79 ::grpc::StatusCode code,
80 const std::string& message,
81 const std::string& sql_state) -> ::grpc::Status {
82
83 if (sql_state.empty()) {
84 return ::grpc::Status{code, message};

86
87 // Include SQLSTATE in error details
