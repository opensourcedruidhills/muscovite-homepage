#pragma once
#include <algorithm>
#include <cctype>
#include <string_view>
#include <muscovite/types/strong_type_base.hpp>
namespace muscovite::types {
class NonEmptyString : public strong_type_base<NonEmptyString> {
public:
using strong_type_base::strong_type_base;
[[nodiscard]] static constexpr auto type_name() noexcept -> std::string_view {
return "NonEmptyString";
}
[[nodiscard]] static constexpr auto pattern() noexcept -> std::string_view {
return "non-empty, non-whitespace string";
}
[[nodiscard]] static auto is_valid(std::string_view value) noexcept -> bool {
if (value.empty()) return false;
// Check for at least one non-whitespace character
return std::ranges::any_of(value, [](unsigned char c) { return !std::isspace(c); });
}
[[nodiscard]] auto trimmed() const -> std::string_view {
auto start = value_.find_first_not_of(" \t\n\r");
if (start == std::string::npos) return {};
auto end = value_.find_last_not_of(" \t\n\r");
return std::string_view{value_}.substr(start, end - start + 1);
}
[[nodiscard]] auto length() const noexcept -> std::size_t {
return value_.size();
}
};
} // namespace muscovite::types
