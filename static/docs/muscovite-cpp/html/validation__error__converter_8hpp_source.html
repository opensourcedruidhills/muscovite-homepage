89inline validation::ValidationMessage toValidationMessage(const ValidationError& error) {
90 QString const codeStr = QString::fromStdString(error.code);
91 QString const fieldName = QString::fromStdString(error.field);
92 QString const detail = QString::fromStdString(error.message);
93
94 validation::ValidationCode const code = parseValidationCode(codeStr);
95
96 // If we parsed as VAL500_Custom, use the backend message as detail
97 // Otherwise, the standard message from validationMessage() is used
98 if (code == validation::ValidationCode::VAL500_Custom && !detail.isEmpty()) {
99 return validation::ValidationMessage::custom(fieldName, detail, false);
100 }
101
102 return {code, fieldName, detail};
103}

157inline std::optional<ValidationError> extractValidationErrorFromContext(const ::grpc::ClientContext& ctx) {
158 const auto& metadata = ctx.GetServerTrailingMetadata();
159
160 auto codeIt = metadata.find("x-validation-code");
161 auto fieldIt = metadata.find("x-validation-field");
162 auto msgIt = metadata.find("x-validation-message");
163
164 if (codeIt != metadata.end() || msgIt != metadata.end()) {
165 ValidationError error;
166 if (codeIt != metadata.end()) {
167 error.code = std::string(codeIt->second.data(), codeIt->second.size());
168 }
169 if (fieldIt != metadata.end()) {
170 error.field = std::string(fieldIt->second.data(), fieldIt->second.size());
171 }
172 if (msgIt != metadata.end()) {
173 error.message = std::string(msgIt->second.data(), msgIt->second.size());
174 }
175 return error;
176 }
177
178 return std::nullopt;
179}

193inline bool handleValidationMetadata(const ::grpc::ClientContext& ctx, const ::grpc::Status& status,
194 const std::function<void(const validation::ValidationResult&)>& showResult,
195 const std::function<void(bool)>& setLoading) {
196
197 if (status.ok()) {
198 return true; // Success, no errors
199 }
200
201 auto validationError = extractValidationErrorFromContext(ctx);
202 if (validationError) {
203 auto result = fromBackendError(*validationError);
204 showResult(result);
205 setLoading(false);
206 return false; // Validation error shown
207 }
208
209 return true; // Error, but not a validation error - let caller handle
210}
