29class EmailValidator : public QValidator {
30 Q_OBJECT
31public:
32 explicit EmailValidator(QObject* parent = nullptr) : QValidator(parent) {}
33
34 State validate(QString& input, int& /*pos*/) const override {
35 if (input.isEmpty()) return Intermediate;
36 if (QtEmail::isValid(input)) return Acceptable;
37
38 // Regex for valid email characters during typing
39 // Local part: alphanumeric, dots, hyphens, underscores, plus
40 // Domain part: alphanumeric, dots, hyphens (partial allowed)
41 static const QRegularExpression partialEmail(
42 QStringLiteral(R"(^[a-zA-Z0-9._%+-]+(@[a-zA-Z0-9.-]*)?$)"));
43
44 if (!partialEmail.match(input).hasMatch()) {
45 return Invalid;
46 }
47
48 // Must have local part before @ (reject @foo)
49 if (input.startsWith('@')) {
50 return Invalid;
51 }
52
53 // Reject multiple consecutive dots
54 if (input.contains(QStringLiteral(".."))) {
55 return Invalid;
56 }
57
58 // Reject multiple @ signs
59 if (input.count('@') > 1) {
60 return Invalid;
61 }
62
63 return Intermediate; // Valid partial email
64 }
65};

72
73 State validate(QString& input, int& /*pos*/) const override {
74 if (input.isEmpty()) return Intermediate;
75 if (QtPhone::isValid(input)) return Acceptable;
76
77 // Allow partial phone while typing
78 // Check if it only contains valid characters
79 static const QRegularExpression validChars(
80 QStringLiteral(R"(^[\+0-9\s\-\‍(\)]*$)"));
81 if (validChars.match(input).hasMatch()) {
82 return Intermediate; // Valid chars, just incomplete

109class UrlValidator : public QValidator {
110 Q_OBJECT
111public:
112 explicit UrlValidator(QObject* parent = nullptr) : QValidator(parent) {}
113
114 State validate(QString& input, int& /*pos*/) const override {
115 if (input.isEmpty()) return Intermediate;
116 if (QtUrl::isValid(input)) return Acceptable;
117
118 // Allow typing scheme prefix
119 if (QStringLiteral("http://").startsWith(input, Qt::CaseInsensitive) ||
120 QStringLiteral("https://").startsWith(input, Qt::CaseInsensitive) ||
121 QStringLiteral("ftp://").startsWith(input, Qt::CaseInsensitive)) {
122 return Intermediate;
123 }
124 // Allow partial URL after scheme
125 if (input.startsWith(QStringLiteral("http://")) ||
126 input.startsWith(QStringLiteral("https://")) ||
127 input.startsWith(QStringLiteral("ftp://"))) {
128 return Intermediate;
129 }
130 return Invalid;
131 }
132};

163 -> QValidator* {
164 if (typeName == QStringLiteral("Email")) {

174 return new UrlValidator(parent);
175 }
