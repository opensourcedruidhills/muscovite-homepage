85auto OutboxMetrics::prometheus_exposition() const -> std::string {
86 std::ostringstream oss;
87
88 // Collect schemas under lock
89 std::vector<std::pair<std::string, SchemaMetrics*>> schema_list;
90 {
91 const std::scoped_lock lock{m_mutex};
92 for (const auto& [schema, metrics] : m_schemas) {
93 schema_list.emplace_back(schema, metrics.get());
94 }
95 }
96
97 // Published counter
98 oss << "# HELP muscovite_outbox_events_published_total Total events published successfully\n";
99 oss << "# TYPE muscovite_outbox_events_published_total counter\n";
100 for (const auto& [schema, m] : schema_list) {
101 oss << std::format("muscovite_outbox_events_published_total{{schema=\"{}\"}} {}\n",
102 schema, m->published.load());
103 }
104
105 // Failed counter
106 oss << "# HELP muscovite_outbox_events_failed_total Total events failed (moved to DLQ)\n";
107 oss << "# TYPE muscovite_outbox_events_failed_total counter\n";
108 for (const auto& [schema, m] : schema_list) {
109 oss << std::format("muscovite_outbox_events_failed_total{{schema=\"{}\"}} {}\n",
110 schema, m->failed.load());
111 }
112
113 // Retried counter
114 oss << "# HELP muscovite_outbox_events_retried_total Total retry attempts\n";
115 oss << "# TYPE muscovite_outbox_events_retried_total counter\n";
116 for (const auto& [schema, m] : schema_list) {
117 oss << std::format("muscovite_outbox_events_retried_total{{schema=\"{}\"}} {}\n",
118 schema, m->retried.load());
119 }
120
121 // Lag gauge
122 oss << "# HELP muscovite_outbox_processing_lag_seconds Current processing lag in seconds\n";
123 oss << "# TYPE muscovite_outbox_processing_lag_seconds gauge\n";
124 for (const auto& [schema, m] : schema_list) {
125 oss << std::format("muscovite_outbox_processing_lag_seconds{{schema=\"{}\"}} {:.3f}\n",
126 schema, m->lag.load());
127 }
128
129 // Batch size gauge
130 oss << "# HELP muscovite_outbox_batch_size Number of events in last batch\n";
131 oss << "# TYPE muscovite_outbox_batch_size gauge\n";
132 for (const auto& [schema, m] : schema_list) {
133 oss << std::format("muscovite_outbox_batch_size{{schema=\"{}\"}} {}\n",
134 schema, m->batch_size.load());
135 }
136
137 // Queue depth gauge
138 oss << "# HELP muscovite_outbox_queue_depth Number of unpublished events\n";
139 oss << "# TYPE muscovite_outbox_queue_depth gauge\n";
140 for (const auto& [schema, m] : schema_list) {
141 oss << std::format("muscovite_outbox_queue_depth{{schema=\"{}\"}} {}\n",
142 schema, m->queue_depth.load());
143 }
144
145 return oss.str();
146}

148auto OutboxMetrics::summary() const -> std::string {
149 std::ostringstream oss;
150 oss << "Outbox Metrics Summary:\n";
151
152 std::vector<std::pair<std::string, SchemaMetrics*>> schema_list;
153 {
154 const std::scoped_lock lock{m_mutex};
155 for (const auto& [schema, metrics] : m_schemas) {
156 schema_list.emplace_back(schema, metrics.get());
157 }
158 }
159
160 for (const auto& [schema, m] : schema_list) {
161 oss << std::format(" [{}]\n", schema);
162 oss << std::format(" Published: {}\n", m->published.load());
163 oss << std::format(" Failed: {}\n", m->failed.load());
164 oss << std::format(" Retried: {}\n", m->retried.load());
165 oss << std::format(" Lag: {:.3f}s\n", m->lag.load());
166 oss << std::format(" Queue: {}\n", m->queue_depth.load());
167 oss << std::format(" Broker: {}\n", m->broker_connected.load() ? "connected" : "disconnected");
168 oss << std::format(" Running: {}\n", m->publisher_running.load() ? "yes" : "no");
169 }
170
171 return oss.str();
172}

199auto OutboxMetrics::health_status(const std::string& schema) const -> OutboxHealthStatus {
200 OutboxHealthStatus status;
201 status.schema = schema;
202
203 const auto* m = get_metrics(schema);
204 if (m != nullptr) {
205 status.broker_connected = m->broker_connected.load();
206 status.publisher_running = m->publisher_running.load();
207 status.lag_seconds = m->lag.load();
208 status.queue_depth = m->queue_depth.load();
209 status.failed_events = m->failed.load();
210 status.published_events = m->published.load();
211 }
212
213 return status;
214}

250auto OutboxMetrics::get_or_create(const std::string& schema) -> SchemaMetrics& {
251 const std::scoped_lock lock{m_mutex};
252 auto it = m_schemas.find(schema);
253 if (it == m_schemas.end()) {
254 it = m_schemas.emplace(schema, std::make_unique<SchemaMetrics>()).first;
255 }
256 return *it->second;
257}
258
259auto OutboxMetrics::get_metrics(const std::string& schema) const -> const SchemaMetrics* {
260 const std::scoped_lock lock{m_mutex};
261 auto it = m_schemas.find(schema);
262 return it != m_schemas.end() ? it->second.get() : nullptr;
263}
