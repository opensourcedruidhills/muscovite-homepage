56template <std::int64_t Min, std::int64_t Max>
57 requires(Min <= Max)
58class BoundedInteger {
59public:
60 using value_type = std::int64_t;
61
62 static constexpr value_type min_value = Min;
63 static constexpr value_type max_value = Max;
64
66 [[nodiscard]] static consteval auto has_lower_bound() noexcept -> bool {
67 return Min != unbounded_min;
68 }
69
71 [[nodiscard]] static consteval auto has_upper_bound() noexcept -> bool {
72 return Max != unbounded_max;
73 }
74
76 [[nodiscard]] static constexpr auto is_valid(value_type v) noexcept -> bool {
77 if constexpr (has_lower_bound() && has_upper_bound()) {
78 return v >= Min && v <= Max;
79 } else if constexpr (has_lower_bound()) {
80 return v >= Min;
81 } else if constexpr (has_upper_bound()) {
82 return v <= Max;
83 } else {
84 return true;
85 }
86 }
87
89 [[nodiscard]] static auto try_from(value_type v)
90 -> std::expected<BoundedInteger, std::string> {
91 if (!is_valid(v)) {
92 return std::unexpected(format_error(v));
93 }
94 return BoundedInteger{v, trusted_tag{}};
95 }
96
98 [[nodiscard]] static constexpr auto from_trusted(value_type v) noexcept
99 -> BoundedInteger {
100 return BoundedInteger{v, trusted_tag{}};
101 }
102
104 [[nodiscard]] constexpr auto value() const noexcept -> value_type {
105 return value_;
106 }
107
109 [[nodiscard]] constexpr operator value_type() const noexcept {
110 return value_;
111 }
112
114 [[nodiscard]] constexpr auto operator<=>(const BoundedInteger&) const noexcept = default;
115
117 [[nodiscard]] constexpr auto operator==(const BoundedInteger&) const noexcept -> bool = default;
118
119private:
120 struct trusted_tag {};
121
122 constexpr BoundedInteger(value_type v, trusted_tag) noexcept
123 : value_(v) {}
124
125 [[nodiscard]] static auto format_error(value_type v) -> std::string {
126 if constexpr (has_lower_bound() && has_upper_bound()) {
127 return std::format("Value {} out of range [{}, {}]", v, Min, Max);
128 } else if constexpr (has_lower_bound()) {
129 return std::format("Value {} below minimum {}", v, Min);
130 } else if constexpr (has_upper_bound()) {
131 return std::format("Value {} above maximum {}", v, Max);
132 } else {
133 return std::format("Value {} out of range", v);
134 }
135 }
136
137 value_type value_{};
138};
