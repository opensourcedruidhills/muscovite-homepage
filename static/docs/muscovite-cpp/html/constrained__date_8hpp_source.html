42public:

47 [[nodiscard]] static auto is_valid(const value_type& date) -> bool {
48 if (!date.ok()) return false;

53 [[nodiscard]] static auto try_from(const value_type& date)
54 -> std::expected<ConstrainedDate, std::string> {

102 [[nodiscard]] static auto today() -> value_type {
103 auto now = clock_type::now();
104 auto dp = std::chrono::floor<std::chrono::days>(now);
105 return std::chrono::year_month_day{dp};
106 }
107
108 [[nodiscard]] static auto check_policy(const value_type& date) -> bool {
109 if constexpr (std::is_same_v<Policy, Unconstrained>) {
110 return true;
111 } else {
112 auto t = today();
113 auto date_days = std::chrono::sys_days{date};
114 auto today_days = std::chrono::sys_days{t};
115
116 if constexpr (std::is_same_v<Policy, PastOrPresent>) {
117 return date_days <= today_days;
118 } else if constexpr (std::is_same_v<Policy, PastOnly>) {
119 return date_days < today_days;
120 } else if constexpr (std::is_same_v<Policy, Future>) {
121 return date_days > today_days;
122 } else {
123 static_assert(std::is_same_v<Policy, Unconstrained>,
124 "Unknown temporal policy");
125 return true;
126 }
127 }
128 }
129
130 [[nodiscard]] static auto format_error(const value_type& date) -> std::string {
131 return std::format("Date {} violates {} policy",
132 date, Policy::name);
133 }

135 value_type value_{};
136};
