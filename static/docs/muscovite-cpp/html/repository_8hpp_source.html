121 [[nodiscard]] auto find_by_id(const id_type& id) -> std::expected<std::optional<Entity>, Error<DbError>> override {
122 return execute(this->m_pool, [&](pqxx::work& tx) -> std::optional<Entity> {
123 auto query = static_cast<Derived*>(this)->build_select_by_id_query();
124 auto result = tx.exec(query, pqxx::params{id});
125
126 if (result.empty()) {
127 return std::nullopt;
128 }
129
130 return static_cast<Derived*>(this)->map_row(result[0]);
131 });
132 }
133
134 [[nodiscard]] auto find_all(std::size_t limit, std::size_t offset)
135 -> std::expected<std::vector<Entity>, Error<DbError>> override {
136 return execute(this->m_pool, [&](pqxx::work& tx) -> std::vector<Entity> {
137 auto query = static_cast<Derived*>(this)->build_select_all_query(limit, offset);
138 auto result = tx.exec(query);
139
140 std::vector<Entity> entities;
141 entities.reserve(result.size());
142
143 for (const auto& row : result) {
144 entities.push_back(static_cast<Derived*>(this)->map_row(row));
145 }
146
147 return entities;
148 });
149 }
150
151 [[nodiscard]] auto count() -> std::expected<std::size_t, Error<DbError>> override {
152 return execute(this->m_pool, [&](pqxx::work& tx) -> std::size_t {
153 auto query = static_cast<Derived*>(this)->build_count_query();
154 auto [cnt] = tx.query1<std::size_t>(query);
155 return cnt;
156 });
157 }

159 [[nodiscard]] auto exists(const id_type& id) -> std::expected<bool, Error<DbError>> override {
160 return execute(this->m_pool, [&](pqxx::work& tx) -> bool {
161 auto query = static_cast<Derived*>(this)->build_exists_query();
162 auto [found] = tx.query1<bool>(query, pqxx::params{id});
163 return found;
164 });

167 [[nodiscard]] auto insert(const Entity& entity) -> std::expected<Entity, Error<DbError>> override {
168 return execute(this->m_pool, [&](pqxx::work& tx) -> Entity {
169 // Derived class implements build_insert_query and bind_insert_params

175 return execute(this->m_pool,
176 [&](pqxx::work& tx) -> Entity { return static_cast<Derived*>(this)->do_update(tx, entity); });
177 }

179 [[nodiscard]] auto remove(const id_type& id) -> std::expected<bool, Error<DbError>> override {
180 return execute(this->m_pool, [&](pqxx::work& tx) -> bool {
181 auto query = static_cast<Derived*>(this)->build_delete_query();
182 auto result = tx.exec(query, pqxx::params{id});
183 return result.affected_rows() > 0;
184 });

192template <typename Entity, typename IdType>
193template <typename Result, typename... Args>
194auto Repository<Entity, IdType>::call_read_proc(std::string_view proc_name, Args&&... args)
195 -> std::expected<Result, Error<DbError>> {
196 // Build parameter placeholders using index sequence
197 constexpr auto num_args = sizeof...(Args);
198 std::string placeholders;
199 if constexpr (num_args > 0) {
200 for (std::size_t i = 1; i <= num_args; ++i) {
201 if (i > 1)
202 placeholders += ", ";
203 placeholders += fmt::format("${}", i);
204 }
205 }
206 std::string query = fmt::format("SELECT * FROM {}({})", proc_name, placeholders);
207
208 return execute(m_pool, [&, query = std::move(query)](pqxx::work& tx) -> Result {
209 return tx.query1<Result>(query, pqxx::params{std::forward<Args>(args)...});
210 });
211}
