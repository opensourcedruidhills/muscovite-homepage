47template <std::size_t MinLen, std::size_t MaxLen>
48 requires(MinLen <= MaxLen)
49class BoundedString {
50public:
51 using value_type = std::string;
52
53 static constexpr std::size_t min_length = MinLen;
54 static constexpr std::size_t max_length = MaxLen;
55
57 [[nodiscard]] static consteval auto has_min_length() noexcept -> bool {
58 return MinLen > 0;
59 }
60
62 [[nodiscard]] static consteval auto has_max_length() noexcept -> bool {
63 return MaxLen != unbounded_length;
64 }
65
67 [[nodiscard]] static auto is_valid(std::string_view sv) noexcept -> bool {
68 auto len = sv.size();
69 if constexpr (has_min_length() && has_max_length()) {
70 return len >= MinLen && len <= MaxLen;
71 } else if constexpr (has_min_length()) {
72 return len >= MinLen;
73 } else if constexpr (has_max_length()) {
74 return len <= MaxLen;
75 } else {
76 return true;
77 }
78 }
79
81 [[nodiscard]] static auto try_from(std::string_view sv)
82 -> std::expected<BoundedString, std::string> {
83 if (!is_valid(sv)) {
84 return std::unexpected(format_error(sv));
85 }
86 return BoundedString{std::string{sv}, trusted_tag{}};
87 }
88
90 [[nodiscard]] static auto from_trusted(std::string value) noexcept
91 -> BoundedString {
92 return BoundedString{std::move(value), trusted_tag{}};
93 }
94
96 [[nodiscard]] auto value() const noexcept -> const std::string& {
97 return value_;
98 }
99
101 [[nodiscard]] auto size() const noexcept -> std::size_t {
102 return value_.size();
103 }
104
106 [[nodiscard]] auto empty() const noexcept -> bool {
107 return value_.empty();
108 }
109
111 [[nodiscard]] operator std::string_view() const noexcept {
112 return value_;
113 }
114
116 [[nodiscard]] auto operator<=>(const BoundedString&) const noexcept = default;
117
119 [[nodiscard]] auto operator==(const BoundedString&) const noexcept -> bool = default;
120
121private:
122 struct trusted_tag {};
123
124 BoundedString(std::string v, trusted_tag) noexcept
125 : value_(std::move(v)) {}
126
127 [[nodiscard]] static auto format_error(std::string_view sv) -> std::string {
128 if constexpr (has_min_length() && has_max_length()) {
129 return std::format("String length {} out of range [{}, {}]",
130 sv.size(), MinLen, MaxLen);
131 } else if constexpr (has_min_length()) {
132 return std::format("String length {} below minimum {}",
133 sv.size(), MinLen);
134 } else if constexpr (has_max_length()) {
135 return std::format("String length {} above maximum {}",
136 sv.size(), MaxLen);
137 } else {
138 return std::format("String length {} out of range", sv.size());
139 }
140 }
141
142 std::string value_;
143};
