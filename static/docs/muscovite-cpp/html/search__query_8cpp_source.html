101nlohmann::json SearchQuery::to_json() const {
102 json result;
103
104 // --- Build query clause ---
105 // Determine if we need a bool query wrapper
106 const bool has_must = full_text_.has_value() || !match_clauses_.empty();
107 const bool has_filter = !filter_terms_.empty() || !filter_ranges_.empty();
108 const bool has_must_not = !must_not_terms_.empty();
109 const bool needs_bool = has_filter || has_must_not ||
110 (full_text_.has_value() && !match_clauses_.empty());
111
112 if (match_all_ && !needs_bool && !has_must) {
113 result["query"]["match_all"] = json::object();
114 } else if (needs_bool || has_must) {
115 json bool_query;
116
117 // must clauses
118 json must_clauses = json::array();
119 if (full_text_.has_value()) {
120 json multi_match;
121 multi_match["multi_match"]["query"] = *full_text_;
122 multi_match["multi_match"]["type"] = "best_fields";
123 if (!fields_.empty()) {
124 multi_match["multi_match"]["fields"] = fields_;
125 }
126 must_clauses.push_back(std::move(multi_match));
127 }
128 for (const auto& [field, text] : match_clauses_) {
129 json match_clause;
130 match_clause["match"][field] = text;
131 must_clauses.push_back(std::move(match_clause));
132 }
133 if (!must_clauses.empty()) {
134 bool_query["must"] = std::move(must_clauses);
135 }
136
137 // filter clauses
138 if (has_filter) {
139 json filter_clauses = json::array();
140 for (const auto& [field, value] : filter_terms_) {
141 json term_clause;
142 term_clause["term"][field] = value;
143 filter_clauses.push_back(std::move(term_clause));
144 }
145 for (const auto& [field, bounds] : filter_ranges_) {
146 json range_clause;
147 json range_obj;
148 if (bounds.gte) range_obj["gte"] = *bounds.gte;
149 if (bounds.gt) range_obj["gt"] = *bounds.gt;
150 if (bounds.lte) range_obj["lte"] = *bounds.lte;
151 if (bounds.lt) range_obj["lt"] = *bounds.lt;
152 range_clause["range"][field] = std::move(range_obj);
153 filter_clauses.push_back(std::move(range_clause));
154 }
155 bool_query["filter"] = std::move(filter_clauses);
156 }
157
158 // must_not clauses
159 if (has_must_not) {
160 json must_not_clauses = json::array();
161 for (const auto& [field, value] : must_not_terms_) {
162 json term_clause;
163 term_clause["term"][field] = value;
164 must_not_clauses.push_back(std::move(term_clause));
165 }
166 bool_query["must_not"] = std::move(must_not_clauses);
167 }
168
169 result["query"]["bool"] = std::move(bool_query);
170 }
171 // else: no query specified — implicit match_all handled by first branch
172
173 // --- Pagination ---
174 result["from"] = (page_ - 1) * page_size_;
175 result["size"] = page_size_;
176
177 // --- Highlighting ---
178 if (!highlight_fields_.empty()) {
179 json highlight_obj;
180 json fields_obj;
181 for (const auto& field : highlight_fields_) {
182 fields_obj[field] = json::object();
183 }
184 highlight_obj["fields"] = std::move(fields_obj);
185 result["highlight"] = std::move(highlight_obj);
186 }
187
188 // --- Sorting ---
189 if (!sort_specs_.empty()) {
190 json sort_array = json::array();
191 for (const auto& spec : sort_specs_) {
192 json sort_item;
193 sort_item[spec.field] = (spec.order == SortOrder::Asc) ? "asc" : "desc";
194 sort_array.push_back(std::move(sort_item));
195 }
196 result["sort"] = std::move(sort_array);
197 }
198
199 // --- Aggregations ---
200 if (!aggregations_.empty()) {
201 json aggs;
202 for (const auto& agg : aggregations_) {
203 json terms_agg;
204 terms_agg["terms"]["field"] = agg.field;
205 terms_agg["terms"]["size"] = agg.size;
206 aggs[agg.name] = std::move(terms_agg);
207 }
208 result["aggs"] = std::move(aggs);
209 }
210
211 // --- Minimum score ---
212 if (min_score_.has_value()) {
213 result["min_score"] = *min_score_;
214 }
215
216 return result;
217}
