Phone number with format validation.

Phone number with format validation Validates against international phone pattern: ^\+?[0-9\s\-\(\)]{7,20}$

Accepts:

  • International: +1 (555) 123-4567
  • US domestic: 555-123-4567
  • Compact: 5551234567
auto phone = Phone("+1 (555) 123-4567"); // OK
auto bad = Phone("abc"); // throws invalid_value_error

#pragma once
#include <muscovite/types/strong_type_base.hpp>
#include <algorithm>
#include <regex>
#include <string_view>
namespace muscovite::types {
class Phone : public strong_type_base<Phone> {
public:
using strong_type_base::strong_type_base;
[[nodiscard]] static constexpr auto type_name() noexcept -> std::string_view {
return "Phone";
}
[[nodiscard]] static constexpr auto pattern() noexcept -> std::string_view {
return R"(^\+?[0-9\s\-\‍(\)]{7,20}$)";
}
[[nodiscard]] static auto is_valid(std::string_view value) noexcept -> bool {
if (value.empty()) return false;
try {
static const std::regex pattern_{std::string{pattern()}, std::regex::optimize};
return std::regex_match(value.begin(), value.end(), pattern_);
} catch (...) {
return false;
}
}
[[nodiscard]] auto digits_only() const -> std::string {
std::string result;
result.reserve(value_.size());
std::copy_if(value_.begin(), value_.end(), std::back_inserter(result),
[](char c) { return c >= '0' && c <= '9'; });
return result;
}
[[nodiscard]] auto is_international() const noexcept -> bool {
return !value_.empty() && value_[0] == '+';
}
};
} // namespace muscovite::types
