29 auto create_initial_connections() -> std::expected<void, Error<PoolError>> {
30 const std::scoped_lock lock{m_mutex};
31
32 for (std::size_t i = 0; i < m_config.min_connections; ++i) {
33 auto conn = create_connection();
34 if (!conn) {
35 return std::unexpected(conn.error());
36 }
37 m_available.push(std::move(*conn));
38 }
39 return {};
40 }
41
42 auto acquire() -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
43 std::unique_lock lock{m_mutex};
44
45 if (m_shutdown) {
46 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
47 .message = "Connection pool has been shut down",
48 .detail = {},
49 .sql_state = {},
50 .arc = std::nullopt});
51 }
52
53 // Wait for an available connection or room to create one
54 m_cv.wait(lock, [this] {
55 return m_shutdown || !m_available.empty() || (m_active_count + m_available.size() < m_config.max_connections);
56 });
57
58 if (m_shutdown) {
59 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
60 .message = "Connection pool has been shut down",
61 .detail = {},
62 .sql_state = {},
63 .arc = std::nullopt});
64 }
65
66 if (!m_available.empty()) {
67 auto conn = std::move(m_available.front());
68 m_available.pop();
69 ++m_active_count;
70
71 if (m_config.validate_on_borrow && !is_connection_valid(*conn)) {
72 --m_active_count;
73 // Try to create a new connection
74 lock.unlock();
75 auto new_conn = create_connection();
76 lock.lock();
77 if (!new_conn) {
78 return std::unexpected(new_conn.error());
79 }
80 ++m_active_count;
81 return std::move(*new_conn);
82 }
83
84 return conn;
85 }
86
87 // Create new connection
88 ++m_active_count;
89 lock.unlock();
90
91 auto conn = create_connection();
92 if (!conn) {
93 lock.lock();
94 --m_active_count;
95 return std::unexpected(conn.error());
96 }
97
98 return std::move(*conn);
99 }
100
101 auto acquire(std::chrono::milliseconds timeout)
102 -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
103 std::unique_lock lock{m_mutex};
104
105 if (m_shutdown) {
106 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
107 .message = "Connection pool has been shut down",
108 .detail = {},
109 .sql_state = {},
110 .arc = std::nullopt});
111 }
112
113 auto deadline = std::chrono::steady_clock::now() + timeout;
114
115 const bool acquired = m_cv.wait_until(lock, deadline, [this] {
116 return m_shutdown || !m_available.empty() || (m_active_count + m_available.size() < m_config.max_connections);
117 });
118
119 if (!acquired) {
120 return std::unexpected(Error<PoolError>{.code = PoolError::Timeout,
121 .message = "Timed out waiting for connection",
122 .detail = {},
123 .sql_state = {},
124 .arc = std::nullopt});
125 }
126
127 if (m_shutdown) {
128 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
129 .message = "Connection pool has been shut down",
130 .detail = {},
131 .sql_state = {},
132 .arc = std::nullopt});
133 }
134
135 if (!m_available.empty()) {
136 auto conn = std::move(m_available.front());
137 m_available.pop();
138 ++m_active_count;
139 return conn;
140 }
141
142 // Create new connection
143 ++m_active_count;
144 lock.unlock();
145
146 auto conn = create_connection();
147 if (!conn) {
148 lock.lock();
149 --m_active_count;
150 return std::unexpected(conn.error());
151 }
152
153 return std::move(*conn);
154 }
155
156 auto try_acquire() -> std::optional<std::unique_ptr<pqxx::connection>> {
157 const std::scoped_lock lock{m_mutex};
158
159 if (m_shutdown || m_available.empty()) {
160 return std::nullopt;
161 }
162
163 auto conn = std::move(m_available.front());
164 m_available.pop();
165 ++m_active_count;
166 return conn;
167 }
168
169 void return_connection(std::unique_ptr<pqxx::connection> conn) {
170 const std::scoped_lock lock{m_mutex};
171 --m_active_count;
172
173 if (!m_shutdown && conn && conn->is_open()) {
174 m_available.push(std::move(conn));
175 }
176 m_cv.notify_one();
177 }
178
179 [[nodiscard]] auto active_connections() const noexcept -> std::size_t {
180 const std::scoped_lock lock{m_mutex};
181 return m_active_count;
182 }
183
184 [[nodiscard]] auto available_connections() const noexcept -> std::size_t {
185 const std::scoped_lock lock{m_mutex};
186 return m_available.size();
187 }
188
189 [[nodiscard]] auto max_connections() const noexcept -> std::size_t { return m_config.max_connections; }
190
191 void shutdown() {
192 const std::scoped_lock lock{m_mutex};
193 m_shutdown = true;
194 while (!m_available.empty()) {
195 m_available.pop();
196 }
197 m_cv.notify_all();
198 }
199
200private:
201 auto create_connection() -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
202 try {
203 auto conn = std::make_unique<pqxx::connection>(m_config.connection_string);
204 return conn;
205 } catch (const pqxx::broken_connection& e) {
206 return std::unexpected(Error<PoolError>{
207 .code = PoolError::ConnectionFailed, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt});
208 } catch (const std::exception& e) {
209 return std::unexpected(Error<PoolError>{
210 .code = PoolError::ConnectionFailed, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt});
211 }
212 }
213
214 [[nodiscard]] static auto is_connection_valid(pqxx::connection& conn) -> bool {
215 if (!conn.is_open()) {
216 return false;
217 }
218 try {
219 pqxx::nontransaction tx{conn};
220 tx.exec("SELECT 1");
221 return true;
222 } catch (...) {
223 return false;
224 }
225 }
226
227 PoolConfig m_config;
228 mutable std::mutex m_mutex;
229 std::condition_variable m_cv;
230 std::queue<std::unique_ptr<pqxx::connection>> m_available;
231 std::size_t m_active_count{0};
232 bool m_shutdown{false};
233};

295auto ConnectionPool::create(PoolConfig config) -> std::expected<ConnectionPool, Error<PoolError>> {
296 // Validate configuration
297 if (config.connection_string.empty()) {
298 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
299 .message = "Connection string cannot be empty",
300 .detail = {},
301 .sql_state = {},
302 .arc = std::nullopt});
303 }
304
305 if (config.min_connections > config.max_connections) {
306 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
307 .message = "min_connections cannot exceed max_connections",
308 .detail = {},
309 .sql_state = {},
310 .arc = std::nullopt});
311 }
312
313 if (config.max_connections == 0) {
314 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
315 .message = "max_connections must be greater than 0",
316 .detail = {},
317 .sql_state = {},
318 .arc = std::nullopt});
319 }
320
321 ConnectionPool pool;
322 pool.m_impl = std::make_unique<Impl>(std::move(config));
323
324 auto init_result = pool.m_impl->create_initial_connections();
325 if (!init_result) {
326 return std::unexpected(init_result.error());
327 }
328
329 return pool;
330}
