29 auto create_initial_connections() -> std::expected<void, Error<PoolError>> { // NOLINT(readability-convert-member-functions-to-static) FTR-465: accesses m_mutex, m_config, m_available via PIMPL
30 const std::scoped_lock lock{m_mutex};
31
32 for (std::size_t i = 0; i < m_config.min_connections; ++i) {
33 auto conn = create_connection();
34 if (!conn) {
35 return std::unexpected(conn.error());
36 }

38 }

42 auto acquire() -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
43 const auto start_time = std::chrono::steady_clock::now();
44 std::unique_lock lock{m_mutex};
45
46 if (m_shutdown) {
47 ++m_acquisition_failures;
48 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
49 .message = "Connection pool has been shut down",
50 .detail = {},
51 .sql_state = {},
52 .arc = std::nullopt});
53 }
54
55 // Track pending request
56 ++m_pending_requests;
57 const auto wait_start = std::chrono::steady_clock::now();
58
59 // Wait for an available connection or room to create one
60 m_cv.wait(lock, [this] {
61 return m_shutdown || !m_available.empty() || (m_active_count + m_available.size() < m_config.max_connections);
62 });
63
64 const auto wait_end = std::chrono::steady_clock::now();
65 --m_pending_requests;
66
67 const auto wait_duration_us = std::chrono::duration_cast<std::chrono::microseconds>(wait_end - wait_start).count();
68 m_wait_time_us += static_cast<std::uint64_t>(wait_duration_us);
69
70 // Update max wait time
71 auto current_max = m_max_wait_time_us.load(std::memory_order_relaxed);
72 while (static_cast<std::uint64_t>(wait_duration_us) > current_max &&
73 !m_max_wait_time_us.compare_exchange_weak(current_max, static_cast<std::uint64_t>(wait_duration_us), std::memory_order_relaxed)) {
74 }
75
76 if (m_shutdown) {
77 ++m_acquisition_failures;
78 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
79 .message = "Connection pool has been shut down",
80 .detail = {},
81 .sql_state = {},
82 .arc = std::nullopt});
83 }
84
85 if (!m_available.empty()) {
86 auto conn = std::move(m_available.front());
87 m_available.pop();
88 ++m_active_count;
89
90 if (m_config.validate_on_borrow && !is_connection_valid(*conn)) {
91 --m_active_count;
92 ++m_destroyed;
93 // Try to create a new connection
94 lock.unlock();
95 auto new_conn = create_connection();
96 lock.lock();
97 if (!new_conn) {
98 ++m_acquisition_failures;
99 return std::unexpected(new_conn.error());
100 }
101 ++m_active_count;
102 ++m_acquired;
103 const auto end_time = std::chrono::steady_clock::now();
104 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
105 return std::move(*new_conn);
106 }
107
108 ++m_acquired;
109 const auto end_time = std::chrono::steady_clock::now();
110 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
111 return conn;
112 }
113
114 // Create new connection
115 ++m_active_count;
116 lock.unlock();
117
118 auto conn = create_connection();
119 if (!conn) {
120 lock.lock();
121 --m_active_count;
122 ++m_acquisition_failures;
123 return std::unexpected(conn.error());
124 }
125
126 ++m_acquired;
127 const auto end_time = std::chrono::steady_clock::now();
128 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
129 return std::move(*conn);
130 }

132 auto acquire(std::chrono::milliseconds timeout)
133 -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
134 const auto start_time = std::chrono::steady_clock::now();
135 std::unique_lock lock{m_mutex};
136
137 if (m_shutdown) {
138 ++m_acquisition_failures;
139 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
140 .message = "Connection pool has been shut down",
141 .detail = {},
142 .sql_state = {},
143 .arc = std::nullopt});
144 }
145
146 auto deadline = std::chrono::steady_clock::now() + timeout;
147
148 // Track pending request
149 ++m_pending_requests;
150 const auto wait_start = std::chrono::steady_clock::now();
151
152 const bool acquired = m_cv.wait_until(lock, deadline, [this] {
153 return m_shutdown || !m_available.empty() || (m_active_count + m_available.size() < m_config.max_connections);
154 });
155
156 const auto wait_end = std::chrono::steady_clock::now();
157 --m_pending_requests;
158
159 const auto wait_duration_us = std::chrono::duration_cast<std::chrono::microseconds>(wait_end - wait_start).count();
160 m_wait_time_us += static_cast<std::uint64_t>(wait_duration_us);
161
162 // Update max wait time
163 auto current_max = m_max_wait_time_us.load(std::memory_order_relaxed);
164 while (static_cast<std::uint64_t>(wait_duration_us) > current_max &&
165 !m_max_wait_time_us.compare_exchange_weak(current_max, static_cast<std::uint64_t>(wait_duration_us), std::memory_order_relaxed)) {
166 }
167
168 if (!acquired) {
169 ++m_acquisition_failures;
170 return std::unexpected(Error<PoolError>{.code = PoolError::Timeout,
171 .message = "Timed out waiting for connection",
172 .detail = {},
173 .sql_state = {},
174 .arc = std::nullopt});
175 }
176
177 if (m_shutdown) {
178 ++m_acquisition_failures;
179 return std::unexpected(Error<PoolError>{.code = PoolError::AlreadyShutdown,
180 .message = "Connection pool has been shut down",
181 .detail = {},
182 .sql_state = {},
183 .arc = std::nullopt});
184 }
185
186 if (!m_available.empty()) {
187 auto conn = std::move(m_available.front());
188 m_available.pop();
189 ++m_active_count;
190 ++m_acquired;
191 const auto end_time = std::chrono::steady_clock::now();
192 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
193 return conn;
194 }
195
196 // Create new connection
197 ++m_active_count;
198 lock.unlock();
199
200 auto conn = create_connection();
201 if (!conn) {
202 lock.lock();
203 --m_active_count;
204 ++m_acquisition_failures;
205 return std::unexpected(conn.error());
206 }
207
208 ++m_acquired;
209 const auto end_time = std::chrono::steady_clock::now();
210 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
211 return std::move(*conn);
212 }

214 auto try_acquire() -> std::optional<std::unique_ptr<pqxx::connection>> {
215 const auto start_time = std::chrono::steady_clock::now();
216 const std::scoped_lock lock{m_mutex};
217
218 if (m_shutdown || m_available.empty()) {
219 return std::nullopt;
220 }
221
222 auto conn = std::move(m_available.front());
223 m_available.pop();
224 ++m_active_count;
225 ++m_acquired;
226 const auto end_time = std::chrono::steady_clock::now();
227 m_acquisition_time_us += static_cast<std::uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time).count());
228 return conn;

231 void return_connection(std::unique_ptr<pqxx::connection> conn) {
232 const std::scoped_lock lock{m_mutex};
233 --m_active_count;
234 ++m_released;
235
236 if (!m_shutdown && conn && conn->is_open()) {
237 m_available.push(std::move(conn));
238 } else if (conn) {
239 ++m_destroyed;
240 }
241 m_cv.notify_one();
242 }

256 void shutdown() {
257 const std::scoped_lock lock{m_mutex};
258 m_shutdown = true;
259 std::size_t destroyed_count = 0;
260 while (!m_available.empty()) {
261 m_available.pop();
262 ++destroyed_count;
263 }
264 m_destroyed += destroyed_count;
265 m_cv.notify_all();
266 }

281 [[nodiscard]] auto get_pool_status() const noexcept -> PoolStatus {
282 PoolStatus s;
283 const std::lock_guard lock{m_mutex};
284 s.max_size = m_config.max_connections;
285 s.active_connections = m_active_count;
286 s.idle_connections = m_available.size();
287 s.current_size = s.idle_connections + s.active_connections;

293 auto create_connection() -> std::expected<std::unique_ptr<pqxx::connection>, Error<PoolError>> {
294 try {
295 auto conn = std::make_unique<pqxx::connection>(m_config.connection_string);
296 ++m_created;
297 return conn;
298 } catch (const pqxx::failure& e) {
299 // libpqxx 8: pqxx::failure covers all connection failures (poisons_connection() == true)
300 return std::unexpected(Error<PoolError>{
301 .code = PoolError::ConnectionFailed, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt});
302 } catch (const std::exception& e) {
303 return std::unexpected(Error<PoolError>{
304 .code = PoolError::ConnectionFailed, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt});
305 }
306 }

308 [[nodiscard]] static auto is_connection_valid(pqxx::connection& conn) -> bool {
309 if (!conn.is_open()) {
310 return false;
311 }
312 try {
313 pqxx::nontransaction tx{conn};
314 tx.exec("SELECT 1");
315 return true;
316 } catch (...) {
317 return false;
318 }
319 }

400auto ConnectionPool::create(PoolConfig config) -> std::expected<ConnectionPool, Error<PoolError>> { // NOLINT(readability-convert-member-functions-to-static) FTR-465: static factory method correctly declared in header
401 // Validate configuration
402 if (config.connection_string.empty()) {
403 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
404 .message = "Connection string cannot be empty",
405 .detail = {},
406 .sql_state = {},
407 .arc = std::nullopt});
408 }
409
410 if (config.min_connections > config.max_connections) {
411 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
412 .message = "min_connections cannot exceed max_connections",
413 .detail = {},
414 .sql_state = {},
415 .arc = std::nullopt});
416 }
417
418 if (config.max_connections == 0) {
419 return std::unexpected(Error<PoolError>{.code = PoolError::InvalidConfig,
420 .message = "max_connections must be greater than 0",
421 .detail = {},
422 .sql_state = {},
423 .arc = std::nullopt});
424 }
425
426 ConnectionPool pool; // NOLINT(misc-const-correctness) FTR-465: modified by pool.m_impl assignment on next line
427 pool.m_impl = std::make_unique<Impl>(std::move(config));
428
429 auto init_result = pool.m_impl->create_initial_connections();
430 if (!init_result) {
431 return std::unexpected(init_result.error());
432 }
433
434 return pool;
435}
