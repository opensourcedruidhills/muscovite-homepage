#pragma once
#include <muscovite/qt/types/qt_type_base.hpp>
#include <QRegularExpression>
#include <QString>
namespace muscovite::qt {
class QtPhone : public QtTypeBase<QtPhone> {
public:
using QtTypeBase::QtTypeBase;
[[nodiscard]] static auto typeName() -> QString {
return QStringLiteral("Phone");
}
[[nodiscard]] static auto pattern() -> QString {
return QStringLiteral(R"(^\+?[0-9\s\-\‍(\)]{7,20}$)");
}
[[nodiscard]] static auto isValid(QStringView value) -> bool {
if (value.isEmpty()) return false;
static const QRegularExpression re(pattern());
return re.matchView(value).hasMatch();
}
[[nodiscard]] auto digitsOnly() const -> QString {
QString result;
result.reserve(value_.size());
for (QChar c : value_) {
if (c.isDigit()) {
result.append(c);
}
}
return result;
}
[[nodiscard]] auto isInternational() const -> bool {
return !value_.isEmpty() && value_[0] == '+';
}
};
} // namespace muscovite::qt
