136void LoginDialog::onLoginClicked() {
137 const QString serverAddr = serverEdit->text().trimmed();
138 const QString username = usernameEdit->text().trimmed();
139 const QString password = passwordEdit->text();
140
141 // Validate input
142 if (serverAddr.isEmpty()) {
143 errorLabel->setText("Server address is required");
144 serverEdit->setFocus();
145 return;
146 }
147
148 if (username.isEmpty()) {
149 errorLabel->setText("Username is required");
150 usernameEdit->setFocus();
151 return;
152 }
153
154 if (password.isEmpty()) {
155 errorLabel->setText("Password is required");
156 passwordEdit->setFocus();
157 return;
158 }
159
160 // Configure and login
161 auto& authMgr = AuthManager::instance();
162 const AuthConfig config{
163 .serverAddress = serverAddr,
164 .useTls = false,
165 .caCertPath = {},
166 .autoRefresh = true
167 };
168 authMgr.configure(config);
169
170 errorLabel->setText("");
171 statusLabel->setText("Authenticating...");
172 updateUI();
173
174 authMgr.login(username, password);
175}

182 session_ = session;
