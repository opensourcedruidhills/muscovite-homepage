66auto NatsMessageBroker::connect() -> bool {
67 if (m_cb_state->connected.load()) {
68 return true; // Already connected
69 }
70
71 cleanup(); // Clean any previous state
72
73 // Reset callback state for the new connection attempt.
74 m_cb_state->valid.store(true);
75 m_cb_state->closed_cb_done.store(false);
76
77 // Initialize options
78 if (!init_options()) {
79 return false;
80 }
81
82 // Connect to NATS (FTR-208: factory returns RAII-managed connection)
83 auto [conn_status, conn] = make_nats_connection(m_opts.get());
84
85 if (conn_status != NATS_OK) {
86 m_last_error = std::string{"Failed to connect to NATS: "} + natsStatus_GetText(conn_status);
87 spdlog::error("{}", m_last_error);
88 cleanup();
89 return false;
90 }
91 m_conn = std::move(conn);
92
93 // Create JetStream context
94 jsOptions jsOpts;
95 jsOptions_Init(&jsOpts);
96
97 if (!m_config.jetstream_domain.empty()) {
98 jsOpts.Domain = m_config.jetstream_domain.c_str();
99 }
100
101 if (!m_config.jetstream_api_prefix.empty()) {
102 jsOpts.Prefix = m_config.jetstream_api_prefix.c_str();
103 }
104
105 auto [js_status, js] = make_js_context(m_conn.get(), &jsOpts);
106
107 if (js_status != NATS_OK) {
108 m_last_error = std::string{"Failed to create JetStream context: "} + natsStatus_GetText(js_status);
109 spdlog::error("{}", m_last_error);
110 cleanup();
111 return false;
112 }
113 m_js = std::move(js);
114
115 m_cb_state->connected.store(true);
116 spdlog::info("Connected to NATS at {} (JetStream enabled)", m_config.url);
117 return true;
118}

134auto NatsMessageBroker::publish(
135 const std::string& subject,
136 const std::string& payload,
137 const std::string& event_id) -> PublishResult
138{
139 JetStreamPublishOptions opts;
140 opts.msg_id = event_id;
141
142 auto ack = publish_jetstream(subject, payload, opts);
143 if (ack) {
144 return PublishResult::Success;
145 }
146
147 // Check if error is retryable
148 // Connection errors, timeouts are retryable; auth errors are permanent
149 if (!is_connected()) {
150 return PublishResult::RetryableError;
151 }
152
153 // Check last error for hints
154 if (m_last_error.find("timeout") != std::string::npos ||
155 m_last_error.find("disconnected") != std::string::npos ||
156 m_last_error.find("no responders") != std::string::npos) {
157 return PublishResult::RetryableError;
158 }
159
160 // Default to permanent error for unknown failures
161 return PublishResult::PermanentError;
162}

164auto NatsMessageBroker::publish_jetstream(
165 const std::string& subject,
166 const std::string& payload,
167 const JetStreamPublishOptions& options) -> std::optional<JetStreamPublishAck>
168{
169 if (!m_cb_state->connected.load() || !m_js) {
170 m_last_error = "Not connected to NATS";
171 return std::nullopt;
172 }
173
174 // Set up publish options
175 jsPubOptions pub_opts;
176 jsPubOptions_Init(&pub_opts);
177 pub_opts.MsgId = options.msg_id.c_str();
178
179 if (options.expected_last_msg_id) {
180 pub_opts.ExpectLastMsgId = options.expected_last_msg_id->c_str();
181 }
182
183 if (options.expected_last_seq) {
184 pub_opts.ExpectLastSeq = *options.expected_last_seq;
185 }
186
187 pub_opts.MaxWait = static_cast<int64_t>(options.ack_timeout.count());
188
189 // Publish
190 jsPubAck* ack = nullptr;
191 jsErrCode jerr{}; // Default-initialized; values start at 10002
192
193 const natsStatus status = js_Publish(
194 &ack,
195 m_js.get(),
196 subject.c_str(),
197 payload.data(),
198 static_cast<int>(payload.size()),
199 &pub_opts,
200 &jerr
201 );
202
203 if (status != NATS_OK) {
204 m_last_error = std::string{"JetStream publish failed: "} +
205 natsStatus_GetText(status);
206 if (static_cast<int>(jerr) != 0) {
207 m_last_error += " (JS error: " + std::to_string(static_cast<int>(jerr)) + ")";
208 }
209 spdlog::warn("{}", m_last_error);
210 return std::nullopt;
211 }
212
213 // Extract acknowledgment info
214 JetStreamPublishAck result;
215 result.stream = ack->Stream != nullptr ? ack->Stream : "";
216 result.sequence = ack->Sequence;
217 result.duplicate = ack->Duplicate;
218
219 spdlog::debug("JetStream publish to '{}' succeeded: stream={}, seq={}, dup={}",
220 subject, result.stream, result.sequence, result.duplicate);
221
222 // Update stats
223 ++m_messages_published;
224 m_bytes_published += payload.size();
225
226 // Free the ack
227 jsPubAck_Destroy(ack);
228
229 return result;
230}

236auto NatsMessageBroker::ensure_stream(
237 const std::string& stream_name,
238 const std::vector<std::string>& subjects) -> bool
239{
240 if (!m_cb_state->connected.load() || !m_js) {
241 m_last_error = "Not connected to NATS";
242 return false;
243 }
244
245 if (subjects.empty()) {
246 m_last_error = "At least one subject pattern is required";
247 return false;
248 }
249
250 auto* js = m_js.get();
251
252 // First, try to get the existing stream info
253 jsStreamInfo* si = nullptr;
254 jsErrCode jerr{};
255 natsStatus status = js_GetStreamInfo(&si, js, stream_name.c_str(), nullptr, &jerr);
256
257 if (status == NATS_OK && si != nullptr) {
258 // Stream already exists
259 spdlog::debug("JetStream stream '{}' already exists (messages: {})",
260 stream_name, si->State.Msgs);
261 jsStreamInfo_Destroy(si);
262 return true;
263 }
264
265 // Stream doesn't exist, create it
266 // Build subjects string for logging
267 std::string subjects_str;
268 for (size_t i = 0; i < subjects.size(); ++i) {
269 if (i > 0) subjects_str += ", ";
270 subjects_str += subjects[i];
271 }
272 spdlog::info("Creating JetStream stream '{}' for subjects: {}",
273 stream_name, subjects_str);
274
275 jsStreamConfig cfg;
276 jsStreamConfig_Init(&cfg);
277
278 cfg.Name = stream_name.c_str();
279
280 // Convert subjects to C-style array
281 std::vector<const char*> subject_ptrs;
282 subject_ptrs.reserve(subjects.size());
283 for (const auto& s : subjects) {
284 subject_ptrs.push_back(s.c_str());
285 }
286 cfg.Subjects = subject_ptrs.data();
287 cfg.SubjectsLen = static_cast<int>(subjects.size());
288
289 // Use sensible defaults for domain events:
290 // - Limits retention: keep events until explicitly deleted or max age reached
291 // - File storage: durable
292 // - 7 day max age: events older than this are automatically deleted
293 cfg.Retention = js_LimitsPolicy;
294 cfg.Storage = js_FileStorage;
295 cfg.MaxAge = static_cast<int64_t>(7 * 24 * 60 * 60) * 1000000000LL; // 7 days in nanoseconds
296 cfg.Replicas = 1; // Single replica for local dev; production should use 3+
297
298 // Optional: set descriptions
299 const std::string description = "Muscovite domain events for " + stream_name;
300 cfg.Description = description.c_str();
301
302 // Create the stream
303 jsStreamInfo* created_si = nullptr;
304 status = js_AddStream(&created_si, js, &cfg, nullptr, &jerr);
305
306 if (status != NATS_OK) {
307 m_last_error = std::string{"Failed to create JetStream stream '"} + stream_name + "': " +
308 natsStatus_GetText(status);
309 if (static_cast<int>(jerr) != 0) {
310 m_last_error += " (JS error: " + std::to_string(static_cast<int>(jerr)) + ")";
311 }
312 spdlog::error("{}", m_last_error);
313 return false;
314 }
315
316 spdlog::info("JetStream stream '{}' created successfully", stream_name);
317 jsStreamInfo_Destroy(created_si);
318 return true;
319}

342auto NatsMessageBroker::init_options() -> bool {
343 // FTR-208: factory returns RAII-managed options
344 auto [status, opts_ptr] = make_nats_options();
345 if (status != NATS_OK) {
346 m_last_error = std::string{"Failed to create NATS options: "} + natsStatus_GetText(status);
347 return false;
348 }
349 m_opts = std::move(opts_ptr);
350
351 auto* opts = m_opts.get();
352
353 // Set URL
354 status = natsOptions_SetURL(opts, m_config.url.c_str());
355 if (status != NATS_OK) {
356 m_last_error = std::string{"Failed to set NATS URL: "} + natsStatus_GetText(status);
357 return false;
358 }
359
360 // Set connection name
361 if (!m_config.connection_name.empty()) {
362 status = natsOptions_SetName(opts, m_config.connection_name.c_str());
363 if (status != NATS_OK) {
364 spdlog::warn("Failed to set connection name: {}", natsStatus_GetText(status));
365 }
366 }
367
368 // Set reconnect options
369 status = natsOptions_SetMaxReconnect(opts, m_config.max_reconnect);
370 if (status != NATS_OK) {
371 spdlog::warn("Failed to set max reconnect: {}", natsStatus_GetText(status));
372 }
373
374 status = natsOptions_SetReconnectWait(opts, static_cast<int64_t>(m_config.reconnect_wait.count()));
375 if (status != NATS_OK) {
376 spdlog::warn("Failed to set reconnect wait: {}", natsStatus_GetText(status));
377 }
378
379 status = natsOptions_SetTimeout(opts, static_cast<int64_t>(m_config.connect_timeout.count()));
380 if (status != NATS_OK) {
381 spdlog::warn("Failed to set connect timeout: {}", natsStatus_GetText(status));
382 }
383
384 // Set TLS if configured
385 if (m_config.use_tls) {
386 status = natsOptions_SetSecure(opts, true);
387 if (status != NATS_OK) {
388 m_last_error = std::string{"Failed to enable TLS: "} + natsStatus_GetText(status);
389 return false;
390 }
391
392 if (m_config.ca_cert_path) {
393 status = natsOptions_LoadCATrustedCertificates(opts, m_config.ca_cert_path->c_str());
394 if (status != NATS_OK) {
395 m_last_error = std::string{"Failed to load CA cert: "} + natsStatus_GetText(status);
396 return false;
397 }
398 }
399
400 if (m_config.client_cert_path && m_config.client_key_path) {
401 status = natsOptions_LoadCertificatesChain(opts,
402 m_config.client_cert_path->c_str(),
403 m_config.client_key_path->c_str());
404 if (status != NATS_OK) {
405 m_last_error = std::string{"Failed to load client cert: "} + natsStatus_GetText(status);
406 return false;
407 }
408 }
409 }
410
411 // BFX-001: Callbacks use a heap-allocated CallbackState (via raw ptr
412 // to shared_ptr-managed object). This survives broker destruction and
413 // avoids dangling `this`. spdlog is intentionally NOT called here
414 // because the async-callback thread may fire after spdlog statics are
415 // destroyed during process exit.
416 auto* cb = m_cb_state.get();
417
418 status = natsOptions_SetDisconnectedCB(opts, [](natsConnection*, void* closure) {
419 auto* state = static_cast<CallbackState*>(closure);
420 state->connected.store(false);
421 }, cb);
422
423 status = natsOptions_SetReconnectedCB(opts, [](natsConnection*, void* closure) {
424 auto* state = static_cast<CallbackState*>(closure);
425 state->connected.store(true);
426 state->reconnects.fetch_add(1, std::memory_order_relaxed);
427 }, cb);
428
429 status = natsOptions_SetClosedCB(opts, [](natsConnection*, void* closure) {
430 auto* state = static_cast<CallbackState*>(closure);
431 state->connected.store(false);
432 // BFX-001: Signal that the ClosedCB has completed. cleanup()
433 // spin-waits on this so that no callback fires after the broker
434 // (and its CallbackState shared_ptr) is destroyed.
435 state->closed_cb_done.store(true);
436 }, cb);
437
438 return true;
439}

441void NatsMessageBroker::cleanup() {
442 // BFX-001: Mark callback state invalid FIRST. After this, the async
443 // callback thread will see valid==false and skip all work. This must
444 // happen before Close / Destroy so that callbacks fired during teardown
445 // don't touch broker state or spdlog.
446 if (m_cb_state) {
447 m_cb_state->valid.store(false);
448 m_cb_state->connected.store(false);
449 }
450
451 // FTR-208: RAII reset() calls the custom deleter automatically
452 m_js.reset();
453
454 if (m_conn) {
455 // BFX-001: Close the connection then wait for BOTH:
456 // (a) natsConnection_IsClosed — close operation complete
457 // (b) closed_cb_done — ClosedCB has executed and returned
458 // Only after (b) is it safe to Destroy the connection and free
459 // the CallbackState, because the cnats async-callback thread no
460 // longer holds a raw pointer into it.
461 natsConnection_Close(m_conn.get());
462
463 for (int attempt = 0; attempt < 200; ++attempt) {
464 const bool is_closed = natsConnection_IsClosed(m_conn.get());
465 const bool cb_done = m_cb_state && m_cb_state->closed_cb_done.load();
466 if (is_closed && cb_done) {
467 break;
468 }
469 std::this_thread::sleep_for(std::chrono::milliseconds(10));
470 }
471
472 // reset() invokes NatsConnectionDeleter → natsConnection_Destroy
473 m_conn.reset();
474 }
475
476 m_opts.reset();
477}
