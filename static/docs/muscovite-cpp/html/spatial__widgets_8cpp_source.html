35QValidator::State CoordinateValidator::validate(QString& input, int& /*pos*/) const {
36 if (input.isEmpty()) {
37 return Intermediate;
38 }
39
40 bool ok = false;
41 double const value = input.toDouble(&ok);
42
43 if (!ok) {
44 return Invalid;
45 }
46
47 if (isValid(value)) {
48 return Acceptable;
49 }
50
51 return Invalid;
52}

54bool CoordinateValidator::isValid(double value) const
55{
56 switch (type_) {
57 case Type::Latitude:
58 return value >= MIN_LATITUDE && value <= MAX_LATITUDE;
59 case Type::Longitude:
60 return value >= MIN_LONGITUDE && value <= MAX_LONGITUDE;
61 case Type::Radius:
62 return value >= MIN_RADIUS && value <= MAX_RADIUS;
63 }
64 return false;
65}
66
67std::optional<validation::ValidationCode> CoordinateValidator::validationCode(double value) const
68{
69 if (isValid(value)) {
70 return std::nullopt;
71 }
72
73 switch (type_) {
74 case Type::Latitude:
75 if (value < MIN_LATITUDE) {
76 return validation::ValidationCode::VAL050_BelowMinimum;
77 }
78 if (value > MAX_LATITUDE) {
79 return validation::ValidationCode::VAL051_AboveMaximum;
80 }
81 break;
82 case Type::Longitude:
83 if (value < MIN_LONGITUDE) {
84 return validation::ValidationCode::VAL050_BelowMinimum;
85 }
86 if (value > MAX_LONGITUDE) {
87 return validation::ValidationCode::VAL051_AboveMaximum;
88 }
89 break;
90 case Type::Radius:
91 if (value < MIN_RADIUS) {
92 return validation::ValidationCode::VAL050_BelowMinimum;
93 }
94 if (value > MAX_RADIUS) {
95 return validation::ValidationCode::VAL051_AboveMaximum;
96 }
97 break;
98 }
99
100 return validation::ValidationCode::VAL018_InvalidDecimal;
101}

502validation::ValidationResult BoundsInput::validate() const
503{
504 validation::ValidationResult result;
505 auto bnds = bounds();
506
507 if (!latValidator_->isValid(bnds.minLatitude)) {
508 auto code = latValidator_->validationCode(bnds.minLatitude);
509 if (code) {
510 result.addMessage(validation::ValidationMessage(*code, QObject::tr("Min Latitude")));
511 }
512 }
513
514 if (!latValidator_->isValid(bnds.maxLatitude)) {
515 auto code = latValidator_->validationCode(bnds.maxLatitude);
516 if (code) {
517 result.addMessage(validation::ValidationMessage(*code, QObject::tr("Max Latitude")));
518 }
519 }
520
521 if (!lonValidator_->isValid(bnds.minLongitude)) {
522 auto code = lonValidator_->validationCode(bnds.minLongitude);
523 if (code) {
524 result.addMessage(validation::ValidationMessage(*code, QObject::tr("Min Longitude")));
525 }
526 }
527
528 if (!lonValidator_->isValid(bnds.maxLongitude)) {
529 auto code = lonValidator_->validationCode(bnds.maxLongitude);
530 if (code) {
531 result.addMessage(validation::ValidationMessage(*code, QObject::tr("Max Longitude")));
532 }
533 }
534
535 if (bnds.minLatitude > bnds.maxLatitude) {
536 result.addMessage(validation::ValidationMessage(
537 validation::ValidationCode::VAL050_BelowMinimum,
538 QObject::tr("Bounds"),
539 QObject::tr("Min Latitude must be <= Max Latitude")
540 ));
541 }
542
543 if (bnds.minLongitude > bnds.maxLongitude) {
544 result.addMessage(validation::ValidationMessage(
545 validation::ValidationCode::VAL050_BelowMinimum,
546 QObject::tr("Bounds"),
547 QObject::tr("Min Longitude must be <= Max Longitude")
548 ));
549 }
550
551 return result;
552}
