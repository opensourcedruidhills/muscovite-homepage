#pragma once
#include <muscovite/types/temporal_policies.hpp>
#include <chrono>
#include <compare>
#include <expected>
#include <format>
#include <string>
namespace muscovite::types {
template <typename Policy = Unconstrained>
class ConstrainedDate {
public:
using value_type = std::chrono::year_month_day;
using clock_type = std::chrono::system_clock;
[[nodiscard]] static auto is_valid(const value_type& date) -> bool {
if (!date.ok()) return false;
return check_policy(date);
}
[[nodiscard]] static auto try_from(const value_type& date)
-> std::expected<ConstrainedDate, std::string> {
if (!date.ok()) {
return std::unexpected(std::string{"Invalid calendar date"});
}
if (!check_policy(date)) {
return std::unexpected(format_error(date));
}
return ConstrainedDate{date, trusted_tag{}};
}
[[nodiscard]] static constexpr auto from_trusted(const value_type& date) noexcept
-> ConstrainedDate {
return ConstrainedDate{date, trusted_tag{}};
}
[[nodiscard]] constexpr auto value() const noexcept -> const value_type& {
return value_;
}
[[nodiscard]] constexpr auto year() const noexcept -> std::chrono::year {
return value_.year();
}
[[nodiscard]] constexpr auto month() const noexcept -> std::chrono::month {
return value_.month();
}
[[nodiscard]] constexpr auto day() const noexcept -> std::chrono::day {
return value_.day();
}
[[nodiscard]] constexpr auto operator<=>(const ConstrainedDate&) const noexcept = default;
[[nodiscard]] constexpr auto operator==(const ConstrainedDate&) const noexcept -> bool = default;
private:
struct trusted_tag {};
constexpr ConstrainedDate(const value_type& v, trusted_tag) noexcept
: value_(v) {}
[[nodiscard]] static auto today() -> value_type {
auto now = clock_type::now();
auto dp = std::chrono::floor<std::chrono::days>(now);
return std::chrono::year_month_day{dp};
}
[[nodiscard]] static auto check_policy(const value_type& date) -> bool {
if constexpr (std::is_same_v<Policy, Unconstrained>) {
return true;
} else {
auto t = today();
auto date_days = std::chrono::sys_days{date};
auto today_days = std::chrono::sys_days{t};
if constexpr (std::is_same_v<Policy, PastOrPresent>) {
return date_days <= today_days;
} else if constexpr (std::is_same_v<Policy, PastOnly>) {
return date_days < today_days;
} else if constexpr (std::is_same_v<Policy, Future>) {
return date_days > today_days;
} else {
static_assert(std::is_same_v<Policy, Unconstrained>,
"Unknown temporal policy");
return true;
}
}
}
[[nodiscard]] static auto format_error(const value_type& date) -> std::string {
return std::format("Date {} violates {} policy",
date, Policy::name);
}
value_type value_{};
};
} // namespace muscovite::types
