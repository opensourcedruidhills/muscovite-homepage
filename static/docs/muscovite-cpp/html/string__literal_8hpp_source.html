34template <std::size_t N>
35struct StringLiteral {
36 char data[N]{}; // NOLINT(cppcoreguidelines-avoid-c-arrays,modernize-avoid-c-arrays)
37 static constexpr std::size_t size = N - 1;
38
39 // NOLINTNEXTLINE(google-explicit-constructor,cppcoreguidelines-avoid-c-arrays,modernize-avoid-c-arrays)
40 consteval StringLiteral(const char (&str)[N]) {
41 std::copy_n(str, N, data);
42 }
43
44 [[nodiscard]] consteval auto empty() const -> bool {
45 return N <= 1 || data[0] == '\0';
46 }
47
48 [[nodiscard]] consteval auto c_str() const -> const char* {
49 return data;
50 }
51
52 [[nodiscard]] consteval auto operator==(const StringLiteral& other) const -> bool {
53 for (std::size_t i = 0; i < N; ++i) {
54 if (data[i] != other.data[i]) return false;
55 }
56 return true;
57 }
58};
