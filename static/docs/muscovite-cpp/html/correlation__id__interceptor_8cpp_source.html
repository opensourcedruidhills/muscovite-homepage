24class CorrelationIdInterceptor : public ::grpc::experimental::Interceptor {
25public:
26 explicit CorrelationIdInterceptor(::grpc::experimental::ServerRpcInfo* info)
27 : m_method_name{info != nullptr ? info->method() : "unknown"} {}
28
29 void Intercept(::grpc::experimental::InterceptorBatchMethods* methods) override {
30 // Extract or generate correlation ID at the start of RPC processing
31 if (methods->QueryInterceptionHookPoint(
32 ::grpc::experimental::InterceptionHookPoints::POST_RECV_INITIAL_METADATA)) {
33
34 auto* metadata = methods->GetRecvInitialMetadata();
35 std::string correlation_id;
36
37 if (metadata != nullptr) {
38 auto it = metadata->find(std::string{CorrelationContext::kMetadataKey});
39 if (it != metadata->end()) {
40 correlation_id = std::string{it->second.begin(), it->second.end()};
41 }
42 }
43
44 if (correlation_id.empty()) {
45 correlation_id = CorrelationContext::generate();
46 spdlog::debug("Generated correlation ID {} for {}", correlation_id, m_method_name);
47 } else {
48 CorrelationContext::set(correlation_id);
49 spdlog::debug("Propagated correlation ID {} for {}", correlation_id, m_method_name);
50 }
51
52 m_correlation_id = correlation_id;
53 }
54
55 // Add correlation ID to response trailers
56 if (methods->QueryInterceptionHookPoint(
57 ::grpc::experimental::InterceptionHookPoints::PRE_SEND_STATUS)) {
58
59 auto* trailing_metadata = methods->GetSendTrailingMetadata();
60 if (trailing_metadata != nullptr && !m_correlation_id.empty()) {
61 trailing_metadata->insert({std::string{CorrelationContext::kMetadataKey},
62 m_correlation_id});
63 }
64
65 // Clear context at end of request
66 CorrelationContext::clear();
67 }
68
69 methods->Proceed();
70 }
71
72private:
73 std::string m_method_name;
74 std::string m_correlation_id;
75};

77} // anonymous namespace
