36[[nodiscard]] inline auto parse_decimal(std::string_view sv) -> double {
37 double result = 0.0;
38 auto [ptr, ec] = std::from_chars(sv.data(), sv.data() + sv.size(), result); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
39 if (ec != std::errc{} || ptr != sv.data() + sv.size()) { // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
40 return std::numeric_limits<double>::quiet_NaN();
41 }
42 return result;
43}

62template <StringLiteral Min, StringLiteral Max>
63class BoundedDecimal {
64public:
65 using value_type = double;
66
68 [[nodiscard]] static consteval auto has_lower_bound() noexcept -> bool {
69 return !Min.empty();
70 }
71
73 [[nodiscard]] static consteval auto has_upper_bound() noexcept -> bool {
74 return !Max.empty();
75 }
76
78 [[nodiscard]] static auto is_valid(value_type v) noexcept -> bool {
79 if constexpr (has_lower_bound() && has_upper_bound()) {
80 return v >= parse_min() && v <= parse_max();
81 } else if constexpr (has_lower_bound()) {
82 return v >= parse_min();
83 } else if constexpr (has_upper_bound()) {
84 return v <= parse_max();
85 } else {
86 return true;
87 }
88 }

91 [[nodiscard]] static auto try_from(value_type v)
92 -> std::expected<BoundedDecimal, std::string> {
93 if (!is_valid(v)) {
94 return std::unexpected(format_error(v));
95 }
96 return BoundedDecimal{v, trusted_tag{}};
97 }

100 [[nodiscard]] static auto try_from(std::string_view sv)
101 -> std::expected<BoundedDecimal, std::string> {
102 auto v = detail::parse_decimal(sv);
103 if (std::isnan(v)) {
104 return std::unexpected(
105 std::format("Invalid decimal: '{}'", sv));
106 }
107 return try_from(v);
108 }

140 return cached;

145 return cached;

147
148 [[nodiscard]] static auto format_error(value_type v) -> std::string {
149 if constexpr (has_lower_bound() && has_upper_bound()) {
150 return std::format("Value {} out of range [{}, {}]",
151 v, Min.c_str(), Max.c_str());
152 } else if constexpr (has_lower_bound()) {
153 return std::format("Value {} below minimum {}",
154 v, Min.c_str());
155 } else if constexpr (has_upper_bound()) {
156 return std::format("Value {} above maximum {}",
157 v, Max.c_str());
158 } else {
159 return std::format("Value {} out of range", v);
160 }
161 }
