26auto Theme::instance() -> Theme const& {
27 std::call_once(s_initFlag, [] {
28 s_instance = new Theme();
29 if (s_pendingConfig.has_value()) {
30 s_instance->applyConfig(*s_pendingConfig);
31 s_pendingConfig.reset();
32 }
33 s_configured = true;
34 });
35 return *s_instance;
36}
37
38void Theme::configure(ThemeConfig config) {
39 if (s_configured) {
40 throw std::logic_error(
41 "Theme::configure() must be called before the first Theme::instance() call");
42 }
43 s_pendingConfig = config;
44}

104void Theme::applyConfig(ThemeConfig const& config) {
105 if (config.primaryColor) primaryColor_ = *config.primaryColor;
106 if (config.backgroundColor) backgroundColor_ = *config.backgroundColor;
107 if (config.surfaceColor) surfaceColor_ = *config.surfaceColor;
108 if (config.errorColor) errorColor_ = *config.errorColor;
109 if (config.successColor) successColor_ = *config.successColor;
110 if (config.warningColor) warningColor_ = *config.warningColor;
111 if (config.textColor) textColor_ = *config.textColor;
112 if (config.textOnPrimaryColor) textOnPrimaryColor_ = *config.textOnPrimaryColor;
113 if (config.baseSpacing) baseSpacing_ = *config.baseSpacing;
114 if (config.borderRadius) borderRadius_ = *config.borderRadius;
115 if (config.darkMode) darkMode_ = *config.darkMode;
116}
