61QFuture<bool> OAuth2AuthManager::completeAuthorization(const QString& code) {
62 QPromise<bool> promise;
63 auto future = promise.future();
64 promise.start();
65
66 auto codeStr = code.toStdString();
67 QtConcurrent::run([this, promise = std::move(promise), codeStr]() mutable {
68 if (!interceptor_) {
69 promise.addResult(false);
70 promise.finish();
71 return;
72 }
73 try {
74 interceptor_->completeOAuth2Flow(codeStr);
75
76 OAuth2Session sess;
77 auto access = interceptor_->getAccessToken();
78 sess.accessToken = access ? QString::fromStdString(*access) : QString();
79 auto user = interceptor_->getUsername();
80 sess.subject = user ? QString::fromStdString(*user) : QString();
81
82 setSession(sess);
83 promise.addResult(true);
84 } catch (const std::exception& ex) {
85 emit authError(QString::fromUtf8(ex.what()));
86 promise.addResult(false);
87 }
88 promise.finish();
89 });
90
91 return future;
92}

108QFuture<void> OAuth2AuthManager::logout() {
109 QPromise<void> promise;
110 auto future = promise.future();
111 promise.start();
112
113 QtConcurrent::run([this, promise = std::move(promise)]() mutable {
114 try {
115 interceptor_->logout();
116 } catch (...) {
117 // best effort
118 }
119 setSession(std::nullopt);
120 promise.finish();
121 });
122
123 return future;
