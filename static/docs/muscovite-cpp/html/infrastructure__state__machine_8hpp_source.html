35template<std::int64_t BaseMsV = 500, std::int64_t MaxMsV = 30000, int JitterPctV = 10>
36class ExponentialBackoffPolicy {
37public:
38 static constexpr std::int64_t base_ms = BaseMsV;
39 static constexpr std::int64_t max_ms = MaxMsV;
40 static constexpr int jitter_pct = JitterPctV;
41
45 [[nodiscard]] auto next_delay(int attempt) -> std::chrono::milliseconds {
46 // 2^attempt * base, capped at max
47 auto raw = std::min(
48 BaseMsV << std::min(attempt, 30),
49 MaxMsV);
50 return apply_jitter(raw);
51 }
52
54 void reset() noexcept {}
55
56private:
57 [[nodiscard]] auto apply_jitter(std::int64_t delay_ms) -> std::chrono::milliseconds {
58 if constexpr (JitterPctV == 0) {
59 return std::chrono::milliseconds{delay_ms};
60 } else {
61 auto jitter_range = delay_ms * JitterPctV / 100;
62 std::uniform_int_distribution<std::int64_t> dist(-jitter_range, jitter_range);
63 auto jittered = std::clamp(delay_ms + dist(rng_), BaseMsV, MaxMsV);
64 return std::chrono::milliseconds{jittered};
65 }
66 }
67
68 std::mt19937 rng_{std::random_device{}()};
69};

162template<typename BackoffPolicy>
163class InfrastructureStateMachineBase {
164public:
166 using State = CircuitState;
167
170 explicit InfrastructureStateMachineBase(int failure_threshold = 3)
171 : failure_threshold_(failure_threshold) {}
172
173 virtual ~InfrastructureStateMachineBase() = default;
174
175 // Non-copyable, movable
176 InfrastructureStateMachineBase(const InfrastructureStateMachineBase&) = delete;
177 auto operator=(const InfrastructureStateMachineBase&) -> InfrastructureStateMachineBase& = delete;
178 InfrastructureStateMachineBase(InfrastructureStateMachineBase&&) = default;
179 auto operator=(InfrastructureStateMachineBase&&) -> InfrastructureStateMachineBase& = default;
180
187 template<typename F>
188 auto call(F&& fn) -> bool {
189 const std::scoped_lock lock(mutex_);
190
191 switch (state_) {
192 case State::Closed: {
193 const bool ok = std::invoke(std::forward<F>(fn));
194 if (ok) {
195 on_success_internal();
196 } else {
197 on_failure_internal();
198 }
199 return ok;
200 }
201 case State::Open: {
202 auto now = std::chrono::steady_clock::now();
203 auto delay = backoff_.next_delay(open_attempt_);
204 if (now - open_since_ >= delay) {
205 // Transition to Half-Open
206 state_ = State::HalfOpen;
207 ++open_attempt_;
208 on_half_open();
209 // Allow single probe
210 const bool ok = std::invoke(std::forward<F>(fn));
211 if (ok) {
212 on_success_internal();
213 } else {
214 on_failure_internal();
215 }
216 return ok;
217 }
218 // Still in backoff — reject
219 return false;
220 }
221 case State::HalfOpen: {
222 // Should not normally reach here — Half-Open is transient.
223 // But if concurrent calls arrive, reject extras.
224 return false;
225 }
226 }
227 return false; // unreachable
228 }
229
231 [[nodiscard]] auto state() const noexcept -> State {
232 return state_;
233 }
234
236 [[nodiscard]] auto failure_count() const noexcept -> int {
237 return failure_count_;
238 }
239
241 [[nodiscard]] auto failure_threshold() const noexcept -> int {
242 return failure_threshold_;
243 }
244
246 void reset() {
247 const std::scoped_lock lock(mutex_);
248 state_ = State::Closed;
249 failure_count_ = 0;
250 open_attempt_ = 0;
251 backoff_.reset();
252 }
253
254protected:
257 virtual void on_open() {}
258
260 virtual void on_half_open() {}
261
263 virtual void on_trip() {}
264
265private:
266 void on_success_internal() {
267 // Reset failure count and return to Closed
268 state_ = State::Closed;
269 failure_count_ = 0;
270 open_attempt_ = 0;
271 backoff_.reset();
272 }
273
274 void on_failure_internal() {
275 ++failure_count_;
276 if (failure_count_ >= failure_threshold_) {
277 on_trip();
278 state_ = State::Open;
279 open_since_ = std::chrono::steady_clock::now();
280 failure_count_ = 0;
281 on_open();
282 }
283 }
284
285 State state_{State::Closed};
286 int failure_count_{0};
287 int failure_threshold_;
288 int open_attempt_{0};
289 std::chrono::steady_clock::time_point open_since_;
290 BackoffPolicy backoff_{};
291 std::mutex mutex_;
292};
