40template <typename Policy = Unconstrained>
41class ConstrainedDateTime {
42public:

87 : value_(v) {}
88
89 [[nodiscard]] static auto check_policy(const value_type& tp) -> bool {
90 if constexpr (std::is_same_v<Policy, Unconstrained>) {
91 return true;
92 } else {
93 auto now = clock_type::now();
94
95 if constexpr (std::is_same_v<Policy, PastOrPresent>) {
96 return tp <= now;
97 } else if constexpr (std::is_same_v<Policy, PastOnly>) {
98 return tp < now;
99 } else if constexpr (std::is_same_v<Policy, Future>) {
100 return tp > now;
101 } else {
102 static_assert(std::is_same_v<Policy, Unconstrained>,
103 "Unknown temporal policy");
104 return true;
105 }
106 }
107 }
108
109 [[nodiscard]] static auto format_error(const value_type& /*tp*/) -> std::string {
110 return std::format("Timestamp violates {} policy", Policy::name);
111 }

113 value_type value_;
114};
