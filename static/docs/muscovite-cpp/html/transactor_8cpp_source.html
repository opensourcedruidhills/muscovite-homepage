17auto map_pqxx_error(const std::exception& e) -> Error<DbError> {
18 // Try to identify specific pqxx exception types
19
20 if (dynamic_cast<const pqxx::serialization_failure*>(&e) != nullptr) {
21 return Error<DbError>{
22 .code = DbError::SerializationFailure, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
23 }
24
25 if (dynamic_cast<const pqxx::deadlock_detected*>(&e) != nullptr) {
26 return Error<DbError>{.code = DbError::DeadlockDetected, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
27 }
28
29 if (dynamic_cast<const pqxx::unique_violation*>(&e) != nullptr) {
30 return Error<DbError>{.code = DbError::UniqueViolation, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
31 }
32
33 if (dynamic_cast<const pqxx::foreign_key_violation*>(&e) != nullptr) {
34 return Error<DbError>{.code = DbError::ForeignKeyViolation, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
35 }
36
37 if (dynamic_cast<const pqxx::check_violation*>(&e) != nullptr) {
38 return Error<DbError>{.code = DbError::CheckViolation, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
39 }
40
41 if (dynamic_cast<const pqxx::not_null_violation*>(&e) != nullptr) {
42 return Error<DbError>{.code = DbError::NotNullViolation, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
43 }
44
45 if (dynamic_cast<const pqxx::broken_connection*>(&e) != nullptr) {
46 return Error<DbError>{.code = DbError::ConnectionFailed, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
47 }
48
49 // FTR-224: Check SQLSTATE 57014 (query_canceled / statement_timeout) before generic sql_error
50 if (const auto* sql_err = dynamic_cast<const pqxx::sql_error*>(&e); sql_err != nullptr) {
51 if (sql_err->sqlstate() == "57014") {
52 return Error<DbError>{
53 .code = DbError::DeadlineExceeded, .message = e.what(), .detail = {}, .sql_state = "57014", .arc = std::nullopt};
54 }
55 return Error<DbError>{.code = DbError::QueryFailed, .message = e.what(), .detail = {}, .sql_state = sql_err->sqlstate(), .arc = std::nullopt};
56 }
57
58 // Generic error
59 return Error<DbError>{.code = DbError::Unknown, .message = e.what(), .detail = {}, .sql_state = {}, .arc = std::nullopt};
60}

62auto should_retry(DbError code, const RetryPolicy& policy) -> bool {
63 switch (code) {
64 case DbError::SerializationFailure:
65 return policy.retry_on_serialization_failure;
66 case DbError::DeadlockDetected:
67 return policy.retry_on_deadlock;
68 case DbError::ConnectionFailed:
69 return policy.retry_on_connection_error;
70 default:
71 return false;
72 }
73}

75auto next_delay(std::size_t attempt, const RetryPolicy& policy) -> std::chrono::milliseconds {
76 using namespace std::chrono;
77
78 if (attempt == 0) {
79 return policy.initial_delay;
80 }
81
82 // Calculate exponential backoff
83 double multiplier = 1.0;
84 for (std::size_t i = 0; i < attempt; ++i) {
85 multiplier *= policy.backoff_multiplier;
86 }
87
88 auto delay = duration_cast<milliseconds>(
89 duration<double, std::milli>(static_cast<double>(policy.initial_delay.count()) * multiplier));
90
91 // Cap at max_delay
92 if (delay > policy.max_delay) {
93 delay = policy.max_delay;
94 }
95
96 // FTR-235: Cap backoff at remaining_time / 2 when deadline is set
97 if (policy.deadline != system_clock::time_point::max()) {
98 auto remaining = duration_cast<milliseconds>(policy.deadline - system_clock::now());
99 auto half_remaining = remaining / 2;
100 if (half_remaining > milliseconds{0} && delay > half_remaining) {
101 delay = half_remaining;
102 }
103 }
104
105 return delay;
106}
