40auto OutboxPublisher::start() -> bool {
41 if (m_running.load()) {
42 spdlog::warn("OutboxPublisher for {} is already running", m_config.schema_name);
43 return false;
44 }
45
46 auto& metrics = OutboxMetrics::instance();
47
48 // Connect to broker
49 if (!m_broker->connect()) {
50 spdlog::error("OutboxPublisher: Failed to connect to message broker for {}", m_config.schema_name);
51 metrics.record_broker_connected(m_config.schema_name, false);
52 metrics.record_publisher_running(m_config.schema_name, false);
53 return false;
54 }
55
56 metrics.record_broker_connected(m_config.schema_name, true);
57
58 // Ensure message broker infrastructure exists (e.g., JetStream streams)
59 const std::string subject_prefix = "muscovite." + m_config.schema_name;
60 if (!m_broker->ensure_infrastructure(m_config.schema_name, subject_prefix)) {
61 spdlog::error("OutboxPublisher: Failed to ensure broker infrastructure for {}",
62 m_config.schema_name);
63 m_broker->disconnect();
64 metrics.record_broker_connected(m_config.schema_name, false);
65 metrics.record_publisher_running(m_config.schema_name, false);
66 return false;
67 }
68
69 m_stop_requested.store(false);
70 m_running.store(true);
71
72 metrics.record_publisher_running(m_config.schema_name, true);
73
74 m_poll_thread = std::thread{[this] { poll_loop(); }};
75
76 spdlog::info("OutboxPublisher started for schema '{}'", m_config.schema_name);
77 return true;
78}

80void OutboxPublisher::stop() {
81 if (!m_running.load()) {
82 return;
83 }
84
85 spdlog::info("OutboxPublisher stopping for schema '{}'...", m_config.schema_name);
86
87 m_stop_requested.store(true);
88
89 if (m_poll_thread.joinable()) {
90 m_poll_thread.join();
91 }
92
93 m_broker->disconnect();
94 m_running.store(false);
95
96 auto& metrics = OutboxMetrics::instance();
97 metrics.record_publisher_running(m_config.schema_name, false);
98 metrics.record_broker_connected(m_config.schema_name, false);
99
100 spdlog::info("OutboxPublisher stopped for schema '{}'. Published: {}, Failed: {}",
101 m_config.schema_name, m_events_published.load(), m_events_failed.load());
102}

120void OutboxPublisher::poll_loop() {
121 spdlog::debug("OutboxPublisher poll loop started for '{}'", m_config.schema_name);
122
123 auto& metrics = OutboxMetrics::instance();
124
125 while (!m_stop_requested.load()) {
126 try {
127 // Fetch batch of unpublished events
128 auto events = fetch_unpublished_events();
129
130 // Record batch size and queue depth metrics
131 metrics.record_batch_size(m_config.schema_name, events.size());
132 metrics.record_queue_depth(m_config.schema_name, events.size());
133
134 if (events.empty()) {
135 // No events, sleep before next poll
136 std::this_thread::sleep_for(m_config.poll_interval);
137 continue;
138 }
139
140 spdlog::debug("OutboxPublisher fetched {} events from '{}'",
141 events.size(), m_config.schema_name);
142
143 // Process each event
144 for (const auto& event : events) {
145 if (m_stop_requested.load()) {
146 break; // Graceful shutdown during batch
147 }
148
149 // Calculate and record lag (time from event creation to now)
150 auto now = std::chrono::system_clock::now();
151 auto lag = std::chrono::duration<double>(now - event.occurred_at).count();
152 metrics.record_lag(m_config.schema_name, lag);
153
154 // Check if broker is still connected
155 if (!m_broker->is_connected()) {
156 spdlog::warn("OutboxPublisher broker disconnected, attempting reconnect...");
157 if (!m_broker->connect()) {
158 spdlog::error("OutboxPublisher failed to reconnect, will retry next poll");
159 std::this_thread::sleep_for(m_config.poll_interval);
160 break;
161 }
162 }
163
164 // Build subject and publish
165 auto subject = build_subject(event.event_type);
166 auto result = m_broker->publish(subject, event.payload, event.id);
167
168 // Handle result
169 switch (result) {
170 case PublishResult::Success:
171 mark_processed(event.id);
172 ++m_events_published;
173 metrics.record_published(m_config.schema_name);
174 spdlog::debug("Published event {} to {}", event.id, subject);
175 break;
176
177 case PublishResult::RetryableError:
178 mark_failed(event.id, event.retry_count);
179 metrics.record_retry(m_config.schema_name);
180 spdlog::warn("Retryable error publishing event {}, retry count: {}",
181 event.id, event.retry_count + 1);
182 break;
183
184 case PublishResult::PermanentError:
185 move_to_dlq(event, "Permanent publish error");
186 ++m_events_failed;
187 metrics.record_failed(m_config.schema_name);
188 spdlog::error("Permanent error publishing event {}, moved to DLQ", event.id);
189 break;
190 }
191
192 // Invoke callback if set
193 if (m_callback) {
194 m_callback(event, result);
195 }
196 }
197
198 } catch (const pqxx::sql_error& ex) {
199 spdlog::error("OutboxPublisher SQL error: {} (SQLSTATE: {})",
200 ex.what(), ex.sqlstate());
201 std::this_thread::sleep_for(m_config.poll_interval);
202 } catch (const std::exception& ex) {
203 spdlog::error("OutboxPublisher error: {}", ex.what());
204 std::this_thread::sleep_for(m_config.poll_interval);
205 }
206 }
207
208 spdlog::debug("OutboxPublisher poll loop ended for '{}'", m_config.schema_name);
209}

211auto OutboxPublisher::fetch_unpublished_events() -> std::vector<OutboxEvent> {
212 std::vector<OutboxEvent> events;
213
214 auto conn_result = m_pool.acquire();
215 if (!conn_result) {
216 spdlog::error("OutboxPublisher failed to acquire connection: {}",
217 conn_result.error().message);
218 return events;
219 }
220
221 auto& conn = *conn_result;
222
223 try {
224 pqxx::work txn{conn.get()};
225
226 // Query for unpublished events with retry backoff filtering
227 // Events with retry_count > 0 are only fetched if enough time has passed
228 auto query = std::format(R"(
229 SELECT id, event_type, aggregate_id, payload, occurred_at, retry_count
230 FROM {}.domain_events
231 WHERE processed_at IS NULL
232 AND (
233 retry_count = 0
234 OR last_retry_at IS NULL
235 OR last_retry_at < NOW() - INTERVAL '1 millisecond' * {}
236 )
237 ORDER BY occurred_at ASC
238 LIMIT {}
239 FOR UPDATE SKIP LOCKED
240 )",
241 txn.quote_name(m_config.schema_name),
242 m_config.initial_retry_delay.count(), // Base delay for simplicity
243 m_config.batch_size
244 );
245
246 auto result = txn.exec(query);
247
248 events.reserve(static_cast<std::size_t>(result.size()));
249 for (const auto& row : result) {
250 OutboxEvent event;
251 event.id = row["id"].as<std::string>();
252 event.event_type = row["event_type"].as<std::string>();
253 event.aggregate_id = row["aggregate_id"].as<std::string>();
254
255 // ADR 20260202: Payload is stored as TEXT (JSON string)
256 event.payload = row["payload"].as<std::string>();
257
258 // Parse timestamp — TODO: proper parsing with date library
259 [[maybe_unused]] auto occurred_str = row["occurred_at"].as<std::string>();
260 event.occurred_at = std::chrono::system_clock::now();
261
262 event.retry_count = row["retry_count"].as<std::size_t>(0);
263
264 events.push_back(std::move(event));
265 }
266
267 txn.commit();
268
269 } catch (const pqxx::sql_error& ex) {
270 spdlog::error("Failed to fetch unpublished events: {} (SQLSTATE: {})",
271 ex.what(), ex.sqlstate());
272 }
273
274 return events;
275}

277void OutboxPublisher::mark_processed(const std::string& event_id) {
278 auto conn_result = m_pool.acquire();
279 if (!conn_result) {
280 spdlog::error("Failed to acquire connection for mark_processed");
281 return;
282 }
283
284 auto& conn = *conn_result;
285
286 try {
287 pqxx::work txn{conn.get()};
288
289 auto query = std::format(
290 "UPDATE {}.domain_events SET processed_at = NOW() WHERE id = {}",
291 txn.quote_name(m_config.schema_name),
292 txn.quote(event_id)
293 );
294
295 txn.exec(query);
296 txn.commit();
297
298 } catch (const std::exception& ex) {
299 spdlog::error("Failed to mark event {} as processed: {}", event_id, ex.what());

303void OutboxPublisher::mark_failed(const std::string& event_id, std::size_t current_retry_count) {
304 auto conn_result = m_pool.acquire();
305 if (!conn_result) {
306 spdlog::error("Failed to acquire connection for mark_failed");
307 return;
308 }
309
310 auto& conn = *conn_result;
311
312 try {
313 pqxx::work txn{conn.get()};
314
315 auto new_retry_count = current_retry_count + 1;
316
317 if (new_retry_count >= m_config.max_retries) {
318 // Will be moved to DLQ on next fetch (or we could do it here)
319 spdlog::warn("Event {} exceeded max retries ({}), will be moved to DLQ",
320 event_id, m_config.max_retries);
321 }
322
323 auto query = std::format(
324 "UPDATE {}.domain_events SET retry_count = {}, last_retry_at = NOW() WHERE id = {}",
325 txn.quote_name(m_config.schema_name),
326 new_retry_count,
327 txn.quote(event_id)
328 );
329
330 txn.exec(query);
331 txn.commit();
332
333 } catch (const std::exception& ex) {
334 spdlog::error("Failed to mark event {} as failed: {}", event_id, ex.what());
335 }

338void OutboxPublisher::move_to_dlq(const OutboxEvent& event, const std::string& reason) {
339 auto conn_result = m_pool.acquire();
340 if (!conn_result) {
341 spdlog::error("Failed to acquire connection for move_to_dlq");
342 return;
343 }
344
345 auto& conn = *conn_result;
346
347 try {
348 pqxx::work txn{conn.get()};
349
350 // Insert into dead-letter queue table
351 auto insert_query = std::format(R"(
352 INSERT INTO {}.domain_events_dlq (
353 id, event_type, aggregate_id, payload, occurred_at, retry_count, failed_at, failure_reason
354 )
355 SELECT id, event_type, aggregate_id, payload, occurred_at, retry_count, NOW(), {}
356 FROM {}.domain_events
357 WHERE id = {}
358 )",
359 txn.quote_name(m_config.schema_name),
360 txn.quote(reason),
361 txn.quote_name(m_config.schema_name),
362 txn.quote(event.id)
363 );
364
365 txn.exec(insert_query);
366
367 // Delete from main outbox
368 auto delete_query = std::format(
369 "DELETE FROM {}.domain_events WHERE id = {}",
370 txn.quote_name(m_config.schema_name),
371 txn.quote(event.id)
372 );
373
374 txn.exec(delete_query);
375 txn.commit();
376
377 spdlog::info("Moved event {} to dead-letter queue: {}", event.id, reason);
378
379 } catch (const std::exception& ex) {
380 spdlog::error("Failed to move event {} to DLQ: {}", event.id, ex.what());
381 }

384auto OutboxPublisher::calculate_retry_delay(std::size_t retry_count) const -> std::chrono::milliseconds {
385 // Exponential backoff: delay = initial * 2^retry_count, capped at max
386 auto delay = m_config.initial_retry_delay * (1 << retry_count);
387 return std::min(delay, m_config.max_retry_delay);
388}
