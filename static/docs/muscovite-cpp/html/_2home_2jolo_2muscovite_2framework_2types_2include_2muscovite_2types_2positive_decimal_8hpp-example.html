Positive decimal number with validation.

Positive decimal number with validation Validates that the value is a valid decimal > 0. Stores as string to preserve exact representation (no floating-point errors).

auto price = PositiveDecimal("99.99"); // OK
auto qty = PositiveDecimal("1"); // OK
auto bad = PositiveDecimal("-5"); // throws (not positive)
auto bad2 = PositiveDecimal("0"); // throws (must be > 0)

#pragma once
#include <muscovite/types/strong_type_base.hpp>
#include <charconv>
#include <regex>
#include <string_view>
namespace muscovite::types {
class PositiveDecimal : public strong_type_base<PositiveDecimal> {
public:
using strong_type_base::strong_type_base;
[[nodiscard]] static constexpr auto type_name() noexcept -> std::string_view {
return "PositiveDecimal";
}
[[nodiscard]] static constexpr auto pattern() noexcept -> std::string_view {
return R"(^[0-9]+(\.[0-9]+)?$)";
}
[[nodiscard]] static auto is_valid(std::string_view value) noexcept -> bool {
if (value.empty()) return false;
// Check format matches decimal pattern
try {
static const std::regex pattern_{std::string{pattern()}, std::regex::optimize};
if (!std::regex_match(value.begin(), value.end(), pattern_)) {
return false;
}
} catch (...) {
return false;
}
// Parse and check > 0
double d = 0.0;
auto [ptr, ec] = std::from_chars(value.data(), value.data() + value.size(), d); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
if (ec != std::errc{} || ptr != value.data() + value.size()) { // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
return false;
}
return d > 0.0;
}
[[nodiscard]] auto to_double() const -> double {
double result = 0.0;
std::from_chars(value_.data(), value_.data() + value_.size(), result); // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
return result;
}
explicit PositiveDecimal(double value)
: strong_type_base(std::format("{:.6g}", value)) {
if (value <= 0.0) {
throw invalid_value_error(type_name(), value_);
}
}
};
} // namespace muscovite::types
