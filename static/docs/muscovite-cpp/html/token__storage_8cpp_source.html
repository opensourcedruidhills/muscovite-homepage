35// Simple XOR encryption for fallback file storage
36// Note: This is NOT secure, just obfuscation. Use proper keychain when available.
37QByteArray xorEncrypt(const QByteArray& data, const QByteArray& key) {
38 QByteArray result = data;
39 for (qsizetype i = 0; i < result.size(); ++i) {
40 result[i] = static_cast<char>(result[i] ^ key[i % key.size()]);
41 }
42 return result;
43}

45QByteArray generateMachineKey() {
46 // Generate a key based on machine-specific data
47 // This is still not secure, but better than nothing
48 const QString machineData = QSysInfo::machineUniqueId() + QSysInfo::prettyProductName();
49 return QCryptographicHash::hash(machineData.toUtf8(), QCryptographicHash::Sha256);
50}

67#ifdef Q_OS_MACOS
68 bool saveToKeychain(const StoredTokens& tokens) {
69 // Save to macOS Keychain
70 QByteArray data = serializeTokens(tokens);
71
72 SecKeychainItemRef itemRef = nullptr;
73 OSStatus status = SecKeychainFindGenericPassword(
74 nullptr,
75 serviceName.toUtf8().length(),
76 serviceName.toUtf8().constData(),
77 serviceName.toUtf8().length(),
78 serviceName.toUtf8().constData(),
79 nullptr, nullptr,
80 &itemRef
81 );
82
83 if (status == errSecSuccess && itemRef) {
84 // Update existing
85 status = SecKeychainItemModifyContent(
86 itemRef,
87 nullptr,
88 data.length(),
89 data.constData()
90 );
91 CFRelease(itemRef);
92 } else {
93 // Create new
94 status = SecKeychainAddGenericPassword(
95 nullptr,
96 serviceName.toUtf8().length(),
97 serviceName.toUtf8().constData(),
98 serviceName.toUtf8().length(),
99 serviceName.toUtf8().constData(),
100 data.length(),
101 data.constData(),
102 nullptr
103 );
104 }
105
106 return status == errSecSuccess;
107 }
108
109 std::optional<StoredTokens> loadFromKeychain() {
110 void* data = nullptr;
111 UInt32 dataLength = 0;
112
113 OSStatus status = SecKeychainFindGenericPassword(
114 nullptr,
115 serviceName.toUtf8().length(),
116 serviceName.toUtf8().constData(),
117 serviceName.toUtf8().length(),
118 serviceName.toUtf8().constData(),
119 &dataLength,
120 &data,
121 nullptr
122 );
123
124 if (status != errSecSuccess || !data) {
125 return std::nullopt;
126 }
127
128 QByteArray bytes(static_cast<const char*>(data), dataLength);
129 SecKeychainItemFreeContent(nullptr, data);
130
131 return deserializeTokens(bytes);
132 }
133
134 bool clearFromKeychain() {
135 SecKeychainItemRef itemRef = nullptr;
136 OSStatus status = SecKeychainFindGenericPassword(
137 nullptr,
138 serviceName.toUtf8().length(),
139 serviceName.toUtf8().constData(),
140 serviceName.toUtf8().length(),
141 serviceName.toUtf8().constData(),
142 nullptr, nullptr,
143 &itemRef
144 );
145
146 if (status == errSecSuccess && itemRef) {
147 status = SecKeychainItemDelete(itemRef);
148 CFRelease(itemRef);
149 }
150
151 return status == errSecSuccess || status == errSecItemNotFound;
152 }

155#ifdef Q_OS_WIN
156 bool saveToCredentialManager(const StoredTokens& tokens) {
157 QByteArray data = serializeTokens(tokens);
158
159 CREDENTIALW cred = {};
160 cred.Type = CRED_TYPE_GENERIC;
161 cred.TargetName = reinterpret_cast<LPWSTR>(
162 const_cast<ushort*>(serviceName.utf16()));
163 cred.CredentialBlobSize = data.size();
164 cred.CredentialBlob = reinterpret_cast<LPBYTE>(data.data());
165 cred.Persist = CRED_PERSIST_LOCAL_MACHINE;
166 cred.UserName = cred.TargetName;
167
168 return CredWriteW(&cred, 0) == TRUE;
169 }
170
171 std::optional<StoredTokens> loadFromCredentialManager() {
172 PCREDENTIALW cred = nullptr;
173 if (!CredReadW(reinterpret_cast<LPCWSTR>(serviceName.utf16()),
174 CRED_TYPE_GENERIC, 0, &cred)) {
175 return std::nullopt;
176 }
177
178 QByteArray data(reinterpret_cast<const char*>(cred->CredentialBlob),
179 cred->CredentialBlobSize);
180 CredFree(cred);
181
182 return deserializeTokens(data);
183 }
184
185 bool clearFromCredentialManager() {
186 return CredDeleteW(reinterpret_cast<LPCWSTR>(serviceName.utf16()),
187 CRED_TYPE_GENERIC, 0) == TRUE;
188 }
189#endif

191 // Fallback file storage (encrypted with machine key)
192 static bool saveToFile(const StoredTokens& tokens) {
193 const QByteArray data = serializeTokens(tokens);
194 const QByteArray encrypted = xorEncrypt(data, generateMachineKey());
195
196 QFile file(getStoragePath());
197 if (!file.open(QIODevice::WriteOnly)) {
198 return false;
199 }
200 file.write(encrypted);
201 return true;
202 }
203
204 static std::optional<StoredTokens> loadFromFile() {
205 QFile file(getStoragePath());
206 if (!file.open(QIODevice::ReadOnly)) {
207 return std::nullopt;
208 }
209
210 const QByteArray encrypted = file.readAll();
211 const QByteArray data = xorEncrypt(encrypted, generateMachineKey());
212
213 return deserializeTokens(data);
214 }
215
216 static bool clearFromFile() {
217 return QFile::remove(getStoragePath());
218 }
219
220 static QByteArray serializeTokens(const StoredTokens& tokens) {
221 QByteArray data;
222 QDataStream stream(&data, QIODevice::WriteOnly);
223 stream << tokens.accessToken << tokens.refreshToken
224 << tokens.principalId << tokens.username;
225 return data;
226 }
227
228 static std::optional<StoredTokens> deserializeTokens(const QByteArray& data) {
229 QDataStream stream(data);
230 StoredTokens tokens;
231 stream >> tokens.accessToken >> tokens.refreshToken
232 >> tokens.principalId >> tokens.username;
233
234 if (stream.status() != QDataStream::Ok) {
235 return std::nullopt;
236 }
237 return tokens;
238 }

252bool TokenStorage::saveTokens(const StoredTokens& tokens) {
253 bool success = false;
254
255#ifdef Q_OS_MACOS
256 success = impl_->saveToKeychain(tokens);
257#elifdef Q_OS_WIN
258 success = impl_->saveToCredentialManager(tokens);
259#endif
260
261 // Fallback to file storage
262 if (!success) {
263 success = Impl::saveToFile(tokens);
264 }
265
266 if (success) {
267 emit tokensSaved();
268 } else {
269 emit errorOccurred("Failed to save tokens");
270 }
271
272 return success;
273}

275std::optional<StoredTokens> TokenStorage::loadTokens() {
276#ifdef Q_OS_MACOS
277 auto tokens = impl_->loadFromKeychain();
278 if (tokens) return tokens;
279#elifdef Q_OS_WIN
280 auto tokens = impl_->loadFromCredentialManager();
281 if (tokens) return tokens;
282#endif
283
284 // Fallback to file storage
285 return Impl::loadFromFile();
286}
